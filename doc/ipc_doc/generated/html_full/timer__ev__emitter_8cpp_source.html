45Timer_event_emitter::Timer_fired_read_end* Timer_event_emitter::create_timer_signal_pipe()
46{
47 using boost::asio::writable_pipe;
48 using boost::asio::connect_pipe;
49 using boost::movelib::make_unique;
50
51 auto read_end = make_unique<Timer_fired_read_end>(*(m_worker.task_engine()));
52 auto write_end = make_unique<writable_pipe>(*(m_worker.task_engine()));
53 Error_code sys_err_code;
54 connect_pipe(*read_end, *write_end, sys_err_code);
55
56 if (sys_err_code)
57 {
58 // @todo Someday we should report this as an error, etc. Low-priority, as at this stage everything is kaput.
59 FLOW_LOG_FATAL("connect_pipe() failed; this should never happen. Details follow.");
60 FLOW_ERROR_SYS_ERROR_LOG_FATAL();
61 assert(false && "connect_pipe() failed; this should never happen. Details follow."); std::abort();
62 return nullptr;
63 }
64 // else
65
66 auto& read_end_ref = m_signal_pipe_readers.emplace_back(std::move(read_end));
67
68#ifndef NDEBUG
69 const auto result =
70#endif
71 m_signal_pipe_writers.insert({ read_end_ref.get(),
72 std::move(write_end) });
73 assert(result.second);
74
75 FLOW_LOG_TRACE("Timer_event_emitter [" << *this << "]: Pipe created; read-end ptr = [" << read_end_ref.get() << "].");
76
77 return read_end_ref.get();
78} // Timer_event_emitter::create_timer_signal_pipe()

80void Timer_event_emitter::timer_async_wait(flow::util::Timer* timer, Timer_fired_read_end* read_end)
81{
82 FLOW_LOG_TRACE("Timer_event_emitter [" << *this << "]: Starting timer async-wait; when/if it fires, "
83 "we will write to write-end corresponding to read-end ptr = [" << read_end << "].");
84
85 /* Careful: this is a user thread, but the handler below is in our worker thread.
86 * Accessing m_signal_pipe_writers inside there would be not thread-safe, as if some timer happens to fire,
87 * while they (say) create_timer_signal_pipe() for another future timer's purposes, things could explode.
88 * Save the value now; it is accurate and guaranteed valid until `*this` dies (which can only happen
89 * after thread is joined, meaning handler couldn't possibly be executing). */
90 const auto it = m_signal_pipe_writers.find(read_end);
91 assert((it != m_signal_pipe_writers.end()) && "timer_async_wait() invoked on read_end from another *this?");
92 const auto write_end = it->second.get();
93
94 timer->async_wait([this, write_end, read_end](const Error_code& async_err_code)
95 {
96 auto sys_err_code = async_err_code;
97
98 if (sys_err_code == boost::asio::error::operation_aborted)
99 {
100 // As promised we only report actual firings. Stuff is shutting down, or timer canceled. GTFO.
101 return;
102 }
103 // else
104
105 if (sys_err_code)
106 {
107 /* This decision (to simply pretend it was a success code) is borrowed from flow::async::schedule_*().
108 * assert() is another option, but use Flow wisdom. Note, e.g., snd_auto_ping_now() does the same. */
109 FLOW_ERROR_SYS_ERROR_LOG_WARNING();
110 FLOW_LOG_WARNING("Timer_event_emitter [" << *this << "]: "
111 "Timer system error; just logged; totally unexpected; pretending it fired normally.");
112 }
113
114 FLOW_LOG_TRACE("Timer_event_emitter [" << *this << "]: Timer fired. Pushing byte to pipe; "
115 "read_end ptr = [" << read_end << "]. User code should detect event soon and cause byte "
116 "to be popped.");
117
118 util::pipe_produce(get_logger(), write_end);
119 }); // timer->async_wait()
120} // Timer_event_emitter::timer_async_wait()

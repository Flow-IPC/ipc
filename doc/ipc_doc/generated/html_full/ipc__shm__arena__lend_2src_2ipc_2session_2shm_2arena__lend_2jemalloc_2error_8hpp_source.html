39/**
40 * All possible errors returned (via `Error_code` arguments) by ipc::session functions/methods *outside of*
41 * ipc::transport-triggered errors involved in transport involved in doing session-related work; and possibly
42 * system-triggered errors.
43 *
44 * All notes from transport::error::Code doc header apply here.
45 */
46enum class Code
47{
48 /**
49 * Session opening: While setting up the session's locally-managed SHM-arena(s) of an arena-lending SHM-provider,
50 * an error occurred thus hosing the session before it could be opened. Logs may indicate the reason.
51 *
52 * @internal
53 * @todo If ipc::shm::arena_lend arena-creation API(s) are modified to output an `Error_code` instead of
54 * reporting just success-versus-failure, then ipc::session::error::Code::S_SHM_ARENA_CREATION_FAILED should go
55 * away.
56 */
57 S_SHM_ARENA_CREATION_FAILED,
58
59 /**
60 * Session opening: While registering the session's locally-managed SHM-arena(s) of an arena-lending SHM-provider
61 * with the lend/borrow engine, an error occurred thus hosing the session before it could be opened. Logs may
62 * indicate the reason. Most likely it was an internal-IPC failure when sending arena info.
63 *
64 * @internal
65 * @todo If ipc::shm::arena_lend arena-lending API(s) are modified to output an `Error_code` instead of
66 * reporting just success-versus-failure and reporting problem through an async callback, then
67 * ipc::session::error::Code::S_SHM_ARENA_CREATION_FAILED should go away. See
68 * shm::arena_lend::jemalloc::init_shm() body for more discussion.
69 */
70 S_SHM_ARENA_LEND_FAILED,
71
72 /**
73 * Session opening: While trying to transmit resource for internal-IPC use in an arena-lending SHM-provider,
74 * server encountered incoming-direction channel error whose exact nature could not be determined at that
75 * exact moment. Logs will indicate that exact nature; meanwhile session opening failed.
76 */
77 S_SERVER_MASTER_SHM_UNEXPECTED_TRANSPORT_ERROR,
78
79 /// SENTINEL: Not an error. This Code must never be issued by an error/success-emitting API; I/O use only.
80 S_END_SENTINEL
81}; // enum class Code

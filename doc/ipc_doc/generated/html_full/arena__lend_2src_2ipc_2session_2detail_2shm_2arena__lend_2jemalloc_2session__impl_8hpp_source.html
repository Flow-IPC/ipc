45/**
46 * Common data and logic for shm::arena_lend::jemalloc::Server_session_impl and
47 * shm::arena_lend::jemalloc::Client_session_impl. Arena-lending SHM-providers are almost exactly the same,
48 * once everything is open; the only difference is session-server provides an extra `Arena` shared among
49 * all sessions with the same opposing Client_app (created the first time such a session-client establishes
50 * a session); jemalloc::Session_server tracks these `Arena`s and coordinates with jemalloc::Server_session.
51 * So all the in-common stuff is here in this internally-used super-class of the 2 `*_impl`. The setup
52 * procedure is asymmetrical, and the two sides of that asymmetry are in those 2 `*_impl`s -- including
53 * the differing APIs (e.g., Client_session_impl::sync_connect()).
54 *
55 * @tparam Session_impl_t
56 * Our base that our sub-class wants to build on-top-of. E.g., `Server_session_impl<...>`.
57 */

197 /**
198 * To be invoked at most once, synchronously sets up nearly-all aspects of this arena-lending-SHM-capable `Session`;
199 * the prerequisite from sub-class `_impl` being that the possibly-asynchronous steps leading up to this have been
200 * completed successfully. Namely the arg `local_hndl` is the local half of the connected socket-pair created
201 * by the server as needed to create the internal-use IPC channel for the SHM-provider; and
202 * - Client_session_impl: it has async-received this item, after opposing Server_session_impl generated it;
203 * - Server_session_impl: it has generated it and successfully sent the other (of 2) handle to opposing
204 * Client_session_impl.
205 *
206 * That says "nearly-all aspects." That is because Server_session_impl must do one extra thing; namely also create
207 * the `.app_shm()`-returned #Arena and pass that to us as `app_shm_or_null`.
208 *
209 * @param local_hndl
210 * See above. Must not be `.null()`; else behavior undefined (assertion may trip).
211 * `local_hndl` becomes `.null()` at exit. Regardless of init_shm() success or failure, it shall ensure
212 * return of the contained native handle to the OS at the proper time (the caller can rest easy).
213 * @param pool_name_fragment_or_empty
214 * Function that shall help compute SHM-pool names as needed by the SHM-provider.
215 * In particular the final absolute pool name, for each pool, shall be /P/X/R, where P and R will
216 * be determined by `*this` internally; while X -- which shall exclude the leading and trailing
217 * util::Shared_name::S_SEPARATOR (represented as forward-slash in this explanation) -- is
218 * supplied in `pool_name_fragment_or_empty`. If the arg is `.empty()`, then the /X/ part shall be
219 * omitted. Rationale: for cleanup purposes Client_session_impl needs to encode certain info
220 * (spoiler alert: session-client process ID) that can be later parsed-out upon performing a
221 * `for_each_persistent()` (spoiler alert: directory-listing of /dev/shm in Linux at least).
222 * More generally perhaps other info could be encoded there so as to be acquired from an existing
223 * pool-name later. So this here provides the opportunity for our sub-classes to do that.
224 * @param app_shm_or_null
225 * See above. It'll be registered with shm_session() unless null.
226 * @return Whether it was successful (creation of, at least, the #Arena may fail); falsy if so,
227 * the reason describing failure if not. In the former case, post-conditions:
228 * session_shm() and shm_session() are not null.
229 */
230 Error_code init_shm(util::Native_handle&& local_hndl,
231 const Shared_name& pool_name_fragment_or_empty,
232 const std::shared_ptr<Arena>& app_shm_or_null = {});
233
234 /**
235 * Undoes init_shm_arenas(), namely destroying shm_session() pointee and session_shm() pointee. Note
236 * that any other arena (perhaps Server_session_impl::app_shm() pointee?) registered with
237 * shm_session() continues to exist; just objects from it cannot be `lend_object()`ed through that
238 * shm_session() anymore (but may well be lendable through other sessions -- to other processes).
239 *
240 * Intended as of this writing as a one-time resource clean in case of subsequent failure.
241 */
242 void reset_shm();

268TEMPLATE_JEM_SESSION_IMPL
269Error_code CLASS_JEM_SESSION_IMPL::init_shm
270 (util::Native_handle&& local_hndl,
271 const Shared_name& pool_name_fragment_or_empty,
272 const std::shared_ptr<Arena>& app_shm_or_null)
273{
274 using ipc::shm::arena_lend::jemalloc::Memory_manager;
275 using flow::util::ostream_op_string;
276 using std::make_shared;
277 using std::string;
278 using std::to_string;
279
280 // We are in thread W (of vanilla Server/Client_session_impl).
281
282 assert(!local_hndl.null());
283
284 assert(!m_session_shm);
285 assert(!m_shm_session);
286
287 // First m_session_shm (per-session-scope Arena). This can fail.
288
289 /* Arena needs a pool namer it will invoke each time the memory manager decides it wants a new mmap()ped vaddr
290 * space (in the SHM context, <-> SHM-pool). We use our standard semantics for per-session items; but
291 * possibly add pool_name_fragment_or_empty for reasons explained in our method doc header. To save some compute
292 * cycles pre-compute the prefix and remember it via capture. */
293 auto shm_pool_name_prefix = build_conventional_shared_name(Shared_name::S_RESOURCE_TYPE_ID_SHM,
294 Shared_name::ct(Base::Base::m_srv_app_ref.m_name),
295 Base::Base::srv_namespace(),
296 Shared_name::ct(Base::Base::cli_app_ptr()->m_name),
297 Base::Base::cli_namespace());
298 shm_pool_name_prefix /= SHM_SUBTYPE_PREFIX;
299 if (!pool_name_fragment_or_empty.empty())
300 {
301 shm_pool_name_prefix /= pool_name_fragment_or_empty;
302 }
303 shm_pool_name_prefix /= ""; // Might as well remember the last separator too.
304
305 auto pool_naming_func = [shm_pool_name_prefix = std::move(shm_pool_name_prefix)](auto pool_id) -> string
306 {
307 // @todo Arena expects string at the moment; might/should change to Shared_name at some point.
308
309 string shm_pool_name = shm_pool_name_prefix.str();
310 // @todo to_string() deals with locales which may be slow. Consider std::to_chars().
311 return shm_pool_name += to_string(pool_id);
312 }; // auto pool_naming_func =
313
314 m_session_shm
315 = Arena::create(get_logger(),
316 /* @todo At the moment this is fine: other that storing Logger* this is data-free in practice.
317 * TBD whether "formally" this is correct, or we should be storing a singleton or let user
318 * pass-in a thing, or what; particularly if this goes beyond jemalloc specifically.
319 * At the moment it really is fine; even though it derives from a generic-looking Memory_manager,
320 * echan confirms that's at the moment a basically-abandoned idea. Nor is there a practical plan
321 * to un-abandon it. */
322 make_shared<Memory_manager>(get_logger()),
323 typename Arena::Shm_object_name_generator(std::move(pool_naming_func)),
324 /* This is clearly correct if *this super-classes Server_session_impl; but what about
325 * Client_session_impl, given that the setting comes from Server_app m_srv_app_ref?
326 * Answer: Yes, still; see discussion in Server_app::m_permissions_level_for_client_apps doc
327 * header. */
328 util::shared_resource_permissions
329 (Base::Base::m_srv_app_ref.m_permissions_level_for_client_apps));
330 if (!m_session_shm)
331 {
332 FLOW_LOG_WARNING("Session [" << * this << "]: Failed to create session-scope Arena; "
333 "details may be found above. Session will not open.");
334
335 /* We promised to dispose of this at the proper time. Since we won't create the Channel that would do it,
336 * let's do it directly. */
337 transport::asio_local_stream_socket::release_native_peer_socket(std::move(local_hndl));
338
339 // See @todo on this Code; in short if create() emitted an Error_code, we'd just emit that instead here.
340 return error::Code::S_SHM_ARENA_CREATION_FAILED;
341 }
342 // else
343
344 auto setup_done = make_shared<bool>(false);
345
346 /* Next m_shm_session (the lender from any locally-managed arena(s), including m_session_shm; the borrower
347 * from opposing-process-managed arena(s)). */
348 const auto nickname = ostream_op_string("jem-", *this);
349 m_shm_channel.emplace(get_logger(),
350 transport::Socket_stream_channel_of_blobs<true>
351 (get_logger(), nickname,
352 transport::sync_io::Native_socket_stream
353 (get_logger(), nickname, std::move(local_hndl))),
354 transport::struc::Channel_base::S_SERIALIZE_VIA_HEAP,
355 Base::master_channel_const().session_token());
356 m_shm_channel->start([this, setup_done](const Error_code& err_code) mutable
357 {
358 // We are in a struc::Channel thread (not our thread W).
359
360 /* Shm_session got an incoming-direction channel-hosing error. I don't think there should be spurious
361 * traffic, until we give this thing (SHM-session, SHM-arena) to actually use to user, so whatever is
362 * captured here before *setup_done=true should lead to Shm_session::create() or Shm_session::lend_arena()
363 * failing which we handle properly. So eat it until *setup_done is true.
364 *
365 * Come to think of it, I (ygoldfel) decided to more-or-less just eat it after *setup_done is true too.
366 * Basically, once in PEER state, we would just do `if !Base::hosed() { Base::hose(err_code; }`. Until then,
367 * though (but after *setup_done) what to do depends on whether this is Server_session or Client_session.
368 * Like for Server_session we could probably do something like on_master_channel_error(); for
369 * Client_session similarly. Thing is, it's painful to set up that plumbing just for this unlikely
370 * corner case. (That might be because of how I've set up this non-virtual hierarchy. Could probably be
371 * rejiggered to by very slick. @todo maybe.) So what to do? For now I feel this:
372 * - If this added socket-stream-based channel dies, so would m_master_channel (also socket-stream-based).
373 * - So it would hose the session anyway, maybe even significantly sooner if it's due to zombification
374 * (it has keep-alive/idle-timer in both directions, unlike at this time this new channel).
375 * Thus eating it seems all right in that highly catastrophic scenario, as the session-hosing sensors do
376 * their best already.
377 *
378 * @todo Do set up the plumbing, and don't eat the error. Consider also @todo above about perhaps
379 * rejiggering the Session_impl hierarchy to make that plumbing easy-peasy to set up. If we develop
380 * more Session_impl guys -- for other SHM-providers perhaps -- this would rise in priority. For now this
381 * seems reasonable. */
382 Base::async_worker()->post([this, setup_done = std::move(setup_done), err_code]()
383 {
384 if (*setup_done)
385 {
386 FLOW_LOG_WARNING("Session [" << * this << "]: Internal-use (for SHM) channel reported incoming-direction "
387 "error [" << err_code << "] [" << err_code.message() << "]. This occured after SHM-setup; "
388 "almost certainly the session master channel will catch a problem or has caught it; "
389 "session will be hosed, or session opening will fail, depending on the situation.");
390 }
391 else
392 {
393 FLOW_LOG_WARNING("Session [" << * this << "]: Internal-use (for SHM) channel reported incoming-direction "
394 "error [" << err_code << "] [" << err_code.message() << "]. This occured during SHM-setup, "
395 "so we will catch it or have caught it and will fail to open session.");
396 }
397 });
398 });
399
400 m_shm_session = Shm_session::create(get_logger(),
401 /* This is correct/vanilla value create() expects; but see note below
402 * regarding this guy's logging. */
403 Borrower_shm_pool_collection_repository_singleton::get_instance(),
404 *m_shm_channel,
405 /* During its operation (not the init we're doing right now!) a synchronous
406 * Shm_channel op (send(), sync_request()) may yield a channel-hosing error --
407 * at most once -- and then this will be called: */
408 [this, setup_done]
409 (const Error_code& err_code) mutable
410 {
411 /* We are in thread... I (ygoldfel) am not 100% sure at the moment (@todo would be nice to write it down here).
412 * But actually it does not matter too much; at worst we're already in our thread W somehow,
413 * and we're about to do an unnecessary post(). It is fine.
414 *
415 * What is important it that Shm_session tried to do a struc::Channel::send() (or sync_request()), and it
416 * failed indicating that channel just got hosed; which means the *this Session can't really continue operating;
417 * which means, similarly to the above incoming-direction handler do the following.
418 *
419 * Same comment and @todo(s) apply as in the handler above. This is a little different, as it's a send error,
420 * which wouldn't be reported as a channel error; but still we figure it something went this wrong with this
421 * channel, the session master channel will be hosed too. And even that caveat (again) only matters
422 * after *setup_done; before *setup_done we will absolutely catch the problem before exiting the present method. */
423 Base::async_worker()->post([this, setup_done = std::move(setup_done), err_code]()
424 {
425 if (*setup_done)
426 {
427 FLOW_LOG_WARNING("Session [" << * this << "]: Internal-use (for SHM) channel reported outgoing-direction "
428 "error [" << err_code << "] [" << err_code.message() << "]. This occured after SHM-setup; "
429 "almost certainly the session master channel will catch a problem or has caught it; "
430 "session will be hosed, or session opening will fail, depending on the situation.");
431 }
432 else
433 {
434 FLOW_LOG_WARNING("Session [" << * this << "]: Internal-use (for SHM) channel reported outgoing-direction "
435 "error [" << err_code << "] [" << err_code.message() << "]. This occured during SHM-setup, "
436 "so we will catch it or have caught it and will fail to open session.");
437 }
438 });
439 });
440 /* @todo Shm_session::create() takes an optional timeout for any Shm_channel::sync_request() (send, await response)
441 * op it might do; if omitted it's infinite. Assuming no unfathomable Shm_channel IPC error, a long duration
442 * for such a .sync_request() in this context implies some kind of pathologically bad SHM-related operation.
443 * For the time being we have no guidance on what that would look like, so we're assuming things will be fine
444 * to not worry about limiting this. Reconsider (perhaps after observing in the field). */
445
446 /* Regarding Borrower_shm_pool_collection_repository logging:
447 * Borrower_shm_pool_collection_repository is used in singleton fashion across the entire process, period.
448 * Hence it's a glorified global. That's fine; not our problem; but one aspect that's potentially our problem
449 * is its logging. It has a thread-unsafe .set_logger(Logger*) API subsequent to which it will do mostly-TRACE
450 * logging -- though in unlikely error situations also WARNINGs which would be important (as usual for WARNINGs)
451 * to see. If not called, it just won't log (the usual behavior in flow::log when a Logger* is null) which is not
452 * ideal. So should we call it here? Or somewhere? Answer: Well, no. In particular if *this is
453 * Server_session_impl, then there can be 2+ of us; and while in production there shouldn't be 2+
454 * `Client_session_impl`s around, it is allowed for test/debug/profile purposes. Now, we could make some inferences
455 * and do something like:
456 * - Session_server -- of which there should be (in production anyway) only 1 ever in a given process -- can
457 * do it in its ctor.
458 * - Client_session -- of which (as just noted) there should be at most 1 at a time (in production anyway) -- can
459 * do it here; meaning the "active" Client_session is always sharing its get_logger() with this singleton guy.
460 * It's too janky. Can I guarantee the singleton guy isn't logging *something* while we do our "heuristic"
461 * .set_logger() call on it? Not really. For that matter these "in production anyway" claims aren't exactly
462 * iron-clas guarantees either. So what to do?
463 *
464 * There may be something smarter we can do to make this automagic (@todo look into it; discuss with echan), but
465 * in the meantime it seems all-right to ask the user to make the proper
466 *
467 * Borrower_shm_pool_collection_repository_singleton::get_instance().set_logger()
468 *
469 * call in a truly pre-session-work, single-thread context -- at start of main() or something -- to set up
470 * this special guy's logging independently of (prior to) various ipc::session lifecycles. So we do that
471 * in public docs elsewhere.
472 *
473 * @todo Again -- it would be nice to make this automagic but not exactly high-priority. It *would* be nice, as
474 * literally everything else arena_lend-related is not all global-like like this, and we set it up internally
475 * without making the user do *anything*. Just this particular ask is fairly timid, so it's all right. */
476
477 assert(m_shm_session && "It's not supposed to fail according to contract.");
478
479 // Next register one with the other, so stuff constructed in session_shm() can be lend_object()ed.
480
481 if (!shm_session()->lend_arena(session_shm_ptr()))
482 {
483 FLOW_LOG_WARNING("Session [" << * this << "]: Registering session-scope local Arena with borrow/lend engine: "
484 "failed (details likely above). Assuming we are bug-free in using the SHM-provider API "
485 "this would occur only on internal SHM-provider IPC channel error; but due to certain internal "
486 "reasons the triggering Error_code has not reached this thread yet; reporting a more general "
487 "Error_code. That detail aside: Session will not open.");
488 /* Like it says, it'd be nice to report the triggering Error_code. As of this writing lend_arena() does not
489 * report such a thing. In reality, as we just said, probably the just-created (successfully!) Channel
490 * reported error on send() or sync_request() or maybe incoming-direction error somehow;
491 * so it will be reported through an on-error guy up above.
492 * We cannot wait for that though -- I mean we could, but in the meantime what do we return, success? No.
493 * So we just eat that error in that handler and emit the following catch-all thing.
494 * @todo Ideally lend_arena(), being a synchronous API that does internal IPC (synchronously, meaning it sends)
495 * would just emit an Error_code and not (in that case) send it through the post_hose()ing guy. Then we'd just
496 * emit it here and not need the setup_done thing either. What we do, in fact, do is decent enough though. */
497
498 m_shm_session.reset(); // Be somewhat aggressive, though they'd go away in destructor anyway.
499 m_shm_channel.reset();
500 m_session_shm.reset();
501 return error::Code::S_SHM_ARENA_LEND_FAILED;
502 } // if (!shm_session()->lend_arena(session_shm()))
503 // else
504
505 // Lastly register per-app-scope one if so asked (if Server_session_impl is the sub-class invoking us).
506
507 if (app_shm_or_null && (!shm_session()->lend_arena(app_shm_or_null)))
508 {
509 FLOW_LOG_WARNING("Session [" << * this << "]: Registering app-scope local Arena with borrow/lend engine: "
510 "failed (details likely above). Assuming we are bug-free in using the SHM-provider API "
511 "this would occur only on internal SHM-provider IPC channel error; but due to certain internal "
512 "reasons the triggering Error_code has not reached this thread yet; reporting a more general "
513 "Error_code. That detail aside: Session will not open.");
514
515 m_shm_session.reset();
516 m_shm_channel.reset();
517 m_session_shm.reset();
518 return error::Code::S_SHM_ARENA_LEND_FAILED;
519 }
520 // else
521
522 *setup_done = true;
523
524 FLOW_LOG_INFO("Session [" << * this << "]: Successfully created session-scope Arena; and "
525 "borrow/lend engine Shm_session; registered the former to the latter; "
526 "registered app-scope arena too? = [" << bool(app_shm_or_null) << "].");
527
528 return Error_code();
529} // Session_impl::init_shm()

79Error_code make_error_code(Code err_code)
80{
81 /* Assign Category as the category for flow::error::Code-cast error_codes;

93std::string Category::message(int val) const // Virtual.
94{
95 using std::string;
96
97 // KEEP THESE STRINGS IN SYNC WITH COMMENT IN error.hpp ON THE INDIVIDUAL ENUM MEMBERS!
98
99 // See notes in transport::error in same spot.
100 switch (static_cast<Code>(val))
101 {
102 case Code::S_INVALID_ARGUMENT:
103 return "User called an API with 1 or more arguments against the API spec.";
104 case Code::S_OBJECT_SHUTDOWN_ABORTED_COMPLETION_HANDLER:
105 return "Async completion handler is being called prematurely, because underlying object is shutting down, "
106 "as user desires.";
107 case Code::S_CLIENT_NAMESPACE_STORE_BAD_FORMAT:
108 return "Session master channel: log-in as client: The Client Namespace Store (CNS) file (a/k/a PID file) lacks "
109 "leading newline-terminated line with numeric PID.";
110 case Code::S_CLIENT_MASTER_LOG_IN_RESPONSE_BAD:
111 return "Session master channel: log-in as client: received log-in response is not the expected internal message.";
112 case Code::S_SERVER_MASTER_LOG_IN_REQUEST_CLIENT_APP_DISALLOWED_OR_UNKNOWN:
113 return "Session master channel: log-in as server: client identifies self as unknown client-app or one that is not "
114 "registered as an allowed partner to this server-app.";
115 case Code::S_SERVER_MASTER_LOG_IN_REQUEST_CLIENT_APP_INCONSISTENT_CREDS:
116 return "Session master channel: log-in as server: client application-level process credentials (UID, etc.) "
117 "do not match the registered values and/or the OS-reported client peer values from socket stream.";
118 case Code::S_SESSION_OPEN_CHANNEL_REMOTE_PEER_REJECTED_PASSIVE_OPEN:
119 return "Session channel opening: User code chose to reject all passive-open requests.";
120 case Code::S_SESSION_OPEN_CHANNEL_SERVER_CANNOT_PROCEED_RESOURCE_UNAVAILABLE:
121 return "Session channel opening: Server peer is responsible for low-level resource acquisition but failed "
122 "to do so. Session continues; this is not a fatal error.";
123 case Code::S_SESSION_OPEN_CHANNEL_ACTIVE_TIMEOUT:
124 return "Session channel opening: "
125 "Other peer must reply quickly enough as-if non-blocking but did not do so in time.";
126 case Code::S_MUTEX_BIPC_MISC_LIBRARY_ERROR:
127 return "Low-level boost.ipc.mutex: boost.interprocess emitted miscellaneous library exception sans a system code; "
128 "a WARNING message at throw-time should contain all possible details.";
129 case Code::S_SERVER_MASTER_POST_LOG_IN_SEND_FAILED_MISC:
130 return "Session master channel: post-log-in setup (such as SHM setup): send failed because channel hosed "
131 "around that time. Timing was such that exact reason for this could not be obtained, but "
132 "logs will show it.";
133 case Code::S_SERVER_MASTER_LOG_IN_REQUEST_CONFIG_MISMATCH:
134 return "Session master channel: log-in as server: opposing client's compile-time type config differs from "
135 "local server's counterpart config; did you use the proper Server/Client_session paired templates and "
136 "mutually equal template parameters?";
137 case Code::S_RESOURCE_OWNER_UNEXPECTED:
138 return "A resource in the file system (file, SHM pool, MQ, etc.) has or could have unexpected owner; "
139 "ipc::session may emit this when acquiring resources and/or opening session if the server process "
140 "is not running as the configured user, or a previous iteration was not running as that user.";
141 case Code::S_SESSION_FINISHED:
142 return "The opposing end of the session in question has been closed gracefully by the user invoking the "
143 "end-session API.";
144
145 case Code::S_END_SENTINEL:
146 assert(false && "SENTINEL: Not an error. "
147 "This Code must never be issued by an error/success-emitting API; I/O use only.");
148 }
149 assert(false);
150 return "";
151} // Category::message()

153util::String_view Category::code_symbol(Code code) // Static.
154{
155 // Note: Must satisfy istream_to_enum() requirements.
156
157 switch (code)
158 {
159 case Code::S_INVALID_ARGUMENT:
160 return "INVALID_ARGUMENT";
161 case Code::S_OBJECT_SHUTDOWN_ABORTED_COMPLETION_HANDLER:
162 return "OBJECT_SHUTDOWN_ABORTED_COMPLETION_HANDLER";
163 case Code::S_CLIENT_NAMESPACE_STORE_BAD_FORMAT:
164 return "CLIENT_NAMESPACE_STORE_BAD_FORMAT";
165 case Code::S_CLIENT_MASTER_LOG_IN_RESPONSE_BAD:
166 return "CLIENT_MASTER_LOG_IN_RESPONSE_BAD";
167 case Code::S_SERVER_MASTER_LOG_IN_REQUEST_CLIENT_APP_DISALLOWED_OR_UNKNOWN:
168 return "SERVER_MASTER_LOG_IN_REQUEST_CLIENT_APP_DISALLOWED_OR_UNKNOWN";
169 case Code::S_SERVER_MASTER_LOG_IN_REQUEST_CLIENT_APP_INCONSISTENT_CREDS:
170 return "SERVER_MASTER_LOG_IN_REQUEST_CLIENT_APP_INCONSISTENT_CREDS";
171 case Code::S_SESSION_OPEN_CHANNEL_REMOTE_PEER_REJECTED_PASSIVE_OPEN:
172 return "SESSION_OPEN_CHANNEL_REMOTE_PEER_REJECTED_PASSIVE_OPEN";
173 case Code::S_SESSION_OPEN_CHANNEL_SERVER_CANNOT_PROCEED_RESOURCE_UNAVAILABLE:
174 return "SESSION_OPEN_CHANNEL_SERVER_CANNOT_PROCEED_RESOURCE_UNAVAILABLE";
175 case Code::S_SESSION_OPEN_CHANNEL_ACTIVE_TIMEOUT:
176 return "SESSION_OPEN_CHANNEL_ACTIVE_TIMEOUT";
177 case Code::S_MUTEX_BIPC_MISC_LIBRARY_ERROR:
178 return "MUTEX_BIPC_MISC_LIBRARY_ERROR";
179 case Code::S_SERVER_MASTER_POST_LOG_IN_SEND_FAILED_MISC:
180 return "SERVER_MASTER_POST_LOG_IN_SEND_FAILED_MISC";
181 case Code::S_SERVER_MASTER_LOG_IN_REQUEST_CONFIG_MISMATCH:
182 return "SERVER_MASTER_LOG_IN_REQUEST_CONFIG_MISMATCH";
183 case Code::S_RESOURCE_OWNER_UNEXPECTED:
184 return "RESOURCE_OWNER_UNEXPECTED";
185 case Code::S_SESSION_FINISHED:
186 return "SESSION_FINISHED";
187
188 case Code::S_END_SENTINEL:
189 return "END_SENTINEL";
190 }
191 assert(false);
192 return "";
193}

195std::ostream& operator<<(std::ostream& os, Code val)
196{
197 // Note: Must satisfy istream_to_enum() requirements.
198 return os << Category::code_symbol(val);

201std::istream& operator>>(std::istream& is, Code& val)
202{
203 /* Range [<1st Code>, END_SENTINEL); no match => END_SENTINEL;
204 * allow for number instead of ostream<< string; case-insensitive. */
205 val = flow::util::istream_to_enum(&is, Code::S_END_SENTINEL, Code::S_END_SENTINEL, true, false,
206 Code(S_CODE_LOWEST_INT_VALUE));
207 return is;
208}

27/**
28 * The boost.system category for errors returned by the ipc::transport module. Think of it as the polymorphic
29 * counterpart of error::Code, and it kicks in when, for `Error_code ec`, something
30 * like `ec.message()` is invoked.
31 *
32 * Note that this class's declaration is not available outside this translation unit (.cpp
33 * file), and its logic is accessed indirectly through standard boost.system machinery
34 * (`Error_code::name()` and `Error_code::message()`). This class enables those basic
35 * features (e.g., `err_code.message()` returns the string description of the error `err_code`) to
36 * work.
37 */
38class Category :
39 public boost::system::error_category
40{
41public:
42 // Constants.
43
44 /// The one Category.
45 static const Category S_CATEGORY;
46
47 // Methods.
48
49 /**
50 * Implements super-class API: returns a `static` string representing this `error_category` (which,
51 * for example, shows up in the `ostream` representation of any Category-belonging
52 * #Error_code).
53 *
54 * @return A `static` string that's a brief description of this error category.
55 */
56 const char* name() const noexcept override;
57
58 /**
59 * Implements super-class API: given the integer error code of an error in this category, returns a description of
60 * that error (similarly in spirit to `std::strerror()`). This is the "engine" that allows `ec.message()`
61 * to work, where `ec` is an #Error_code with an error::Code value.
62 *
63 * @param val
64 * Error code of an Category error (realistically, a #Code `enum` value cast to
65 * `int`).
66 * @return String describing the error.
67 */
68 std::string message(int val) const override;
69
70 /**
71 * The guts of the `ostream << Code` operation: outputs, e.g., Code::S_INVALID_ARGUMENT => `"INVALID_ARGUMENT"`.
72 * @param code
73 * A Code.
74 * @return What would/should be printed to an `ostream` given `code`.
75 */
76 static util::String_view code_symbol(Code code);
77
78private:
79 // Constructors.
80
81 /// Boring constructor.
82 explicit Category();
83}; // class Category

91Error_code make_error_code(Code err_code)
92{
93 /* Assign Category as the category for flow::error::Code-cast error_codes;
94 * this basically glues together Category::name()/message() with the Code enum. */
95 return Error_code(static_cast<int>(err_code), Category::S_CATEGORY);
96}

105std::string Category::message(int val) const // Virtual.
106{
107 using std::string;
108
109 // KEEP THESE STRINGS IN SYNC WITH COMMENT IN error.hpp ON THE INDIVIDUAL ENUM MEMBERS!
110
111 /* Just create a string (unless compiler is smart enough to do this only once and reuse later)
112 * based on a static char* which is rapidly indexed from val by the switch() statement.
113 * Since the error_category interface requires that message() return a string by value, this
114 * is the best we can do speed-wise... but it should be fine.
115 *
116 * Some error messages can be fancier by specifying outside values (e.g., see
117 * net_flow's S_INVALID_SERVICE_PORT_NUMBER). */
118 switch (static_cast<Code>(val))
119 {
120 case Code::S_SENDS_FINISHED_CANNOT_SEND:
121 return "Will not send message: local user already ended sending via API marking this.";
122 case Code::S_RECEIVES_FINISHED_CANNOT_RECEIVE:
123 return "Will not receive message: opposing user sent graceful-close via API.";
124 case Code::S_LOW_LVL_TRANSPORT_HOSED_CANNOT_SEND:
125 return "Unable to send outgoing traffic: an earlier-reported, or at least logged, system error had hosed the "
126 "underlying transport mechanism.";
127 case Code::S_LOW_LVL_TRANSPORT_HOSED_CANNOT_RECEIVE:
128 return "Unable to receive incoming traffic: an earlier-reported, or at least logged, system error had hosed the "
129 "underlying transport mechanism.";
130 case Code::S_LOW_LVL_TRANSPORT_HOSED:
131 return "Unable to access low-level details: an earlier-reported system error had hosed the underlying "
132 "transport mechanism.";
133 case Code::S_LOW_LVL_UNEXPECTED_STREAM_PAYLOAD_BEYOND_HNDL:
134 return "Unable to receive incoming traffic: message contains more than: 1 blob plus 0-1 native handles.";
135 case Code::S_MESSAGE_SIZE_EXCEEDS_USER_STORAGE:
136 return "User protocol-code mismatch: local user-provided storage cannot fit entire message received from "
137 "opposing peer.";
138 case Code::S_BLOB_RECEIVER_GOT_NON_BLOB:
139 return "User protocol-code mismatch: local user expected blob only and no native handle; received at "
140 "least the latter.";
141 case Code::S_RECEIVER_IDLE_TIMEOUT:
142 return "No messages (optional auto-pings or otherwise) have been received; optionally configured timeout "
143 "exceeded. If a session::Session emitted this error, the session is now hosed due to master channel "
144 "idle timeout indicating poor health of the opposing side or the system.";
145 case Code::S_INVALID_ARGUMENT:
146 return "User called an API with 1 or more arguments against the API spec.";
147 case Code::S_TIMEOUT:
148 return "A (usually user-specified) timeout period has elapsed before a blocking operation completed.";
149 case Code::S_INTERRUPTED:
150 return "A blocking operation was intentionally interrupted or preemptively canceled.";
151 case Code::S_OBJECT_SHUTDOWN_ABORTED_COMPLETION_HANDLER:
152 return "Async completion handler is being called prematurely, because underlying object is shutting down, "
153 "as user desires.";
154 case Code::S_MQ_MESSAGE_SIZE_OVER_OR_UNDERFLOW:
155 return "Low-level message queue send-op buffer overflow (> max size) or receive-op buffer underflow (< max size).";
156 case Code::S_MQ_BIPC_MISC_LIBRARY_ERROR:
157 return "Low-level message queue: boost.interprocess emitted miscellaneous library exception sans a system code; "
158 "a WARNING message at throw-time should contain all possible details.";
159 case Code::S_BLOB_STREAM_MQ_SENDER_EXISTS:
160 return "Message-queue blob stream outgoing-direction peer could not be created, because one already "
161 "exists at that name.";
162 case Code::S_BLOB_STREAM_MQ_RECEIVER_EXISTS:
163 return "Message-queue blob stream incoming-direction peer could not be created, because one already "
164 "exists at that name.";
165 case Code::S_BLOB_STREAM_MQ_RECEIVER_BAD_CTL_CMD:
166 return "Message-queue blob stream peer received a malformed internal-use message from opposing side.";
167 case Code::S_SYNC_IO_WOULD_BLOCK:
168 return
169 "A sync_io operation could not immediately complete; it will complete contingent on active async-wait event(s).";
170 case Code::S_PROTOCOL_NEGOTIATION_OPPOSING_VER_INVALID:
171 return "In protocol negotiation, opposing side sent invalid version value (not positive, not a number, etc.); "
172 "the comm pathway must close.";
173 case Code::S_PROTOCOL_NEGOTIATION_OPPOSING_VER_TOO_OLD:
174 return "In protocol negotiation, opposing side reported its newest protocol version is even older than the most "
175 "backwards-compatible (oldest, smallest) version we speak; the comm pathway must close.";
176
177 case Code::S_END_SENTINEL:
178 assert(false && "SENTINEL: Not an error. "
179 "This Code must never be issued by an error/success-emitting API; I/O use only.");
180 }
181 assert(false);
182 return "";
183} // Category::message()

241std::ostream& operator<<(std::ostream& os, Code val)
242{
243 // Note: Must satisfy istream_to_enum() requirements.
244 return os << Category::code_symbol(val);
245}

247std::istream& operator>>(std::istream& is, Code& val)
248{
249 /* Range [<1st Code>, END_SENTINEL); no match => END_SENTINEL;
250 * allow for number instead of ostream<< string; case-insensitive. */
251 val = flow::util::istream_to_enum(&is, Code::S_END_SENTINEL, Code::S_END_SENTINEL, true, false,
252 Code(S_CODE_LOWEST_INT_VALUE));
253 return is;
254}

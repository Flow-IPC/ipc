33/**
34 * Internal, non-movable pImpl-lite implementation of Blob_stream_mq_sender class template.
35 * In and of itself it would have been directly and publicly usable; however Blob_stream_mq_sender adds move semantics
36 * which are essential to cooperation with the rest of the API, Channel in particular.
37 *
38 * @see All discussion of the public API is in Blob_stream_mq_sender doc header; that class template forwards to this
39 * one. All discussion of pImpl-lite-related notions is also there. See that doc header first please. Then come
40 * back here.
41 *
42 * ### Impl design ###
43 * The actual logic in this class is very simple, because (1) it's either in PEER state or NULL state, and there
44 * is no CONNECTING to speak of -- the only way to go between them is move-assignment; and (2) while in PEER state
45 * it is only a matter of adapting a sync_io::Blob_sender -- namely sync_io::Blob_stream_mq_sender -- *core* into
46 * the requirements of async-I/O-pattern Blob_sender concept. PEER state has 100% of the logic; and
47 * 99% of that logic is adapting which is accomplished by class template sync_io::Async_adapter_sender.
48 *
49 * Therefore:
50 *
51 * @see sync_io::Blob_stream_mq_sender for the core logic having to do with managing MQs.
52 * @see sync_io::Async_adapter_sender for the transport-agnostic adapting of the latter (as a sync_io::Blob_sender)
53 * into a `*this`. We just forward to it.
54 *
55 * @tparam Persistent_mq_handle
56 * See Persistent_mq_handle concept doc header.
57 */

168 /**
169 * Single-thread worker pool for all internal async work. Referred to as thread W in comments.
170 *
171 * Ordering: Must be either declared after mutex(es), or `.stop()`ed explicitly in dtor: Thread must be joined,
172 * before mutex possibly-locked-in-it destructs.
173 */
174 flow::async::Single_thread_task_loop m_worker;
175
176 /**
177 * The core `Blob_stream_mq_sender` engine, implementing the `sync_io` pattern (see util::sync_io doc header).
178 * See our class doc header for overview of how we use it (the aforementioned `sync_io` doc header talks about
179 * the `sync_io` pattern generally).
180 *
181 * Thus, #m_sync_io is the synchronous engine that we use to perform our work in our asynchronous boost.asio
182 * loop running in thread W (#m_worker) while collaborating with user thread(s) a/k/a thread U.
183 * (Recall that the user may choose to set up their own event loop/thread(s) --
184 * boost.asio-based or otherwise -- and use their own equivalent of an #m_sync_io instead.)
185 *
186 * ### Order subtlety versus `m_worker` ###
187 * When constructing #m_sync_io, we need the `Task_engine` from #m_worker. On the other hand tasks operating
188 * in #m_worker access #m_sync_io. So in destructor it is important to `m_worker.stop()` explicitly, so that
189 * the latter is no longer a factor. Then when automatic destruction occurs in the opposite order of
190 * creation, the fact that #m_sync_io is destroyed before #m_worker has no bad effect.
191 */
192 sync_io::Blob_stream_mq_sender<Mq> m_sync_io;
193
194 /**
195 * This handles ~all logic in that state. sync_io::Async_adapter_sender adapts
196 * any sync_io::Blob_sender and makes available ~all necessary async-I/O Blob_sender APIs.
197 * So we forward ~everything to this guy.
198 *
199 * ### Creation ###
200 * By its contract, this guy's ctor will handle what it needs to, as long as #m_worker (to which it stores a pointer)
201 * has been `.start()`ed by that time, and #m_sync_io (to which it stores... ditto) has been
202 * `.replace_event_wait_handles()`ed as required.
203 *
204 * ### Destruction ###
205 * By its contract, this guy's dtor will handle what it needs to, as long as #m_worker (to which it stores a pointer)
206 * has been `.stop()`ed by that time, and any queued-up (ready to execute) handlers on it have been
207 * `Task_enginer::poll()`ed-through by that time as well.
208 *
209 * ### Why `optional`? ###
210 * The only reason is we have to invoke a certain method on #m_sync_io before we can construct this guy.
211 */
212 std::optional<sync_io::Async_adapter_sender<decltype(m_sync_io)>> m_sync_io_adapter;

232template<typename Persistent_mq_handle>
233Blob_stream_mq_sender_impl<Persistent_mq_handle>::Blob_stream_mq_sender_impl
234 (sync_io::Blob_stream_mq_sender<Mq>&& sync_io_core_in_peer_state_moved) :
235 flow::log::Log_context(sync_io_core_in_peer_state_moved.get_logger(), Log_component::S_TRANSPORT),
236 m_worker(sync_io_core_in_peer_state_moved.get_logger(), sync_io_core_in_peer_state_moved.nickname()),
237 // Adopt the just-cted, idle sync_io:: core.
238 m_sync_io(std::move(sync_io_core_in_peer_state_moved))
239 // m_sync_io_adapter is null but is set-up shortly below.
240{
241 using util::sync_io::Asio_waitable_native_handle;
242 using util::sync_io::Task_ptr;
243 using flow::util::ostream_op_string;
244
245 m_worker.start();
246
247 // We're using a boost.asio event loop, so we need to base the async-waited-on handles on our Task_engine.
248#ifndef NDEBUG
249 bool ok =
250#endif
251 m_sync_io.replace_event_wait_handles([this]() -> Asio_waitable_native_handle
252 { return Asio_waitable_native_handle(*(m_worker.task_engine())); });
253 assert(ok && "Did you break contract by passing-in a non-fresh sync_io core object to ctor?");
254
255 /* Have to do this after .replace_event_wait_handles() by the adapter's ctor's contract.
256 * As of this writing that's the only reason m_sync_io_adapter is optional<>. */
257 m_sync_io_adapter.emplace(get_logger(), ostream_op_string("Blob_stream_mq_sender [", *this, ']'),
258 &m_worker, &m_sync_io);
259} // Blob_stream_mq_sender_impl::Blob_stream_mq_sender_impl()

261template<typename Persistent_mq_handle>
262Blob_stream_mq_sender_impl<Persistent_mq_handle>::~Blob_stream_mq_sender_impl()
263{
264 using flow::async::Single_thread_task_loop;
265 using flow::util::ostream_op_string;
266
267 // We are in thread U. By contract in doc header, they must not call us from a completion handler (thread W).
268
269 FLOW_LOG_INFO("Blob_stream_mq_sender [" << *this << "]: Shutting down. All our "
270 "internal async handlers will be canceled; and worker thread will be joined.");
271
272 /* This (1) stop()s the Task_engine thus possibly
273 * preventing any more handlers from running at all (any handler possibly running now is the last one to run); (2)
274 * at that point Task_engine::run() exits, hence thread W exits; (3) joins thread W (waits for it to
275 * exit); (4) returns. That's a lot, but it's non-blocking. */
276 m_worker.stop();
277 // Thread W is (synchronously!) no more.
278
279 FLOW_LOG_INFO("Blob_stream_mq_sender [" << *this << "]: Continuing shutdown. "
280 "Next we will run user handler (if any) from some other thread. "
281 "In this user thread we will await those handlers' completion and then return.");
282
283 // See comment in similar spot in Native_socket_stream::~Impl() regarding the following.
284
285 FLOW_LOG_INFO("Blob_stream_mq_sender [" << *this << "]: Continuing shutdown. Next we will run pending handlers "
286 "from some other thread. In this user thread we will await those handlers' completion and then "
287 "return.");
288
289 Single_thread_task_loop one_thread(get_logger(), ostream_op_string(nickname(), "-temp_deinit"));
290 one_thread.start([&]()
291 {
292 FLOW_LOG_INFO("Blob_stream_mq_sender [" << *this << "]: "
293 "In transient finisher thread: Shall run all pending internal handlers (typically none).");
294
295 const auto task_engine = m_worker.task_engine();
296 task_engine->restart();
297 const auto count = task_engine->poll();
298 if (count != 0)
299 {
300 FLOW_LOG_INFO("Blob_stream_mq_sender [" << *this << "]: "
301 "In transient finisher thread: Ran [" << count << "] internal handlers after all.");
302 }
303 task_engine->stop();
304
305 FLOW_LOG_INFO("Transient finisher exiting. (Send-ops de-init may follow.)");
306 });
307 // Here thread exits/joins synchronously.
308} // Blob_stream_mq_sender_impl::~Blob_stream_mq_sender_impl()

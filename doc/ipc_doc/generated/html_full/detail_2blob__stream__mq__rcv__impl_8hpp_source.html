33/**
34 * Internal, non-movable pImpl-lite implementation of Blob_stream_mq_receiver class template.
35 * In and of itself it would have been directly and publicly usable; however Blob_stream_mq_receiver adds move semantics
36 * which are essential to cooperation with the rest of the API, Channel in particular.
37 *
38 * @see All discussion of the public API is in Blob_stream_mq_receiver doc header; that class template forwards to this
39 * one. All discussion of pImpl-lite-related notions is also there. See that doc header first please. Then come
40 * back here.
41 *
42 * ### Impl design ###
43 * The actual logic in this class is very simple, because (1) it's either in PEER state or NULL state, and there
44 * is no CONNECTING to speak of -- the only way to go between them is move-assignment; and (2) while in PEER state
45 * it is only a matter of adapting a sync_io::Blob_receiver -- namely sync_io::Blob_stream_mq_receiver -- *core* into
46 * the requirements of async-I/O-pattern Blob_receiver concept. PEER state has 100% of the logic; and
47 * 99% of that logic is adapting which is accomplished by class template sync_io::Async_adapter_receiver.
48 *
49 * Therefore:
50 *
51 * @see sync_io::Blob_stream_mq_receiver for the core logic having to do with managing MQs.
52 * @see sync_io::Async_adapter_receiver for the transport-agnostic adapting of the latter (as a sync_io::Blob_receiver)
53 * into a `*this`. We just forward to it.
54 *
55 * @tparam Persistent_mq_handle
56 * See Persistent_mq_handle concept doc header.
57 */

151 /**
152 * Single-thread worker pool for all internal async work. Referred to as thread W in comments.
153 *
154 * Ordering: Must be either declared after mutex(es), or `.stop()`ed explicitly in dtor: Thread must be joined,
155 * before mutex possibly-locked-in-it destructs.
156 */
157 flow::async::Single_thread_task_loop m_worker;
158
159 /**
160 * The core `Blob_stream_mq_receiver` engine, implementing the `sync_io` pattern (see util::sync_io doc header).
161 * See our class doc header for overview of how we use it (the aforementioned `sync_io` doc header talks about
162 * the `sync_io` pattern generally).
163 *
164 * Thus, #m_sync_io is the synchronous engine that we use to perform our work in our asynchronous boost.asio
165 * loop running in thread W (#m_worker) while collaborating with user thread(s) a/k/a thread U.
166 * (Recall that the user may choose to set up their own event loop/thread(s) --
167 * boost.asio-based or otherwise -- and use their own equivalent of an #m_sync_io instead.)
168 *
169 * ### Order subtlety versus `m_worker` ###
170 * When constructing #m_sync_io, we need the `Task_engine` from #m_worker. On the other hand tasks operating
171 * in #m_worker access #m_sync_io. So in destructor it is important to `m_worker.stop()` explicitly, so that
172 * the latter is no longer a factor. Then when automatic destruction occurs in the opposite order of
173 * creation, the fact that #m_sync_io is destroyed before #m_worker has no bad effect.
174 */
175 sync_io::Blob_stream_mq_receiver<Mq> m_sync_io;
176
177 /**
178 * This handles ~all logic in that state. sync_io::Async_adapter_receiver adapts
179 * any sync_io::Blob_receiver and makes available ~all necessary async-I/O Blob_receiver APIs.
180 * So we forward ~everything to this guy.
181 *
182 * ### Creation ###
183 * By its contract, this guy's ctor will handle what it needs to, as long as #m_worker (to which it stores a pointer)
184 * has been `.start()`ed by that time, and #m_sync_io (to which it stores... ditto) has been
185 * `.replace_event_wait_handles()`ed as required.
186 *
187 * ### Destruction ###
188 * By its contract, this guy's dtor will handle what it needs to, as long as #m_worker (to which it stores a pointer)
189 * has been `.stop()`ed by that time, and any queued-up (ready to execute) handlers on it have been
190 * `Task_enginer::poll()`ed-through by that time as well.
191 *
192 * ### Why `optional`? ###
193 * The only reason is we have to invoke a certain method on #m_sync_io before we can construct this guy.
194 */
195 std::optional<sync_io::Async_adapter_receiver<decltype(m_sync_io)>> m_sync_io_adapter;

215template<typename Persistent_mq_handle>
216Blob_stream_mq_receiver_impl<Persistent_mq_handle>::Blob_stream_mq_receiver_impl
217 (sync_io::Blob_stream_mq_receiver<Mq>&& sync_io_core_in_peer_state_moved) :
218 flow::log::Log_context(sync_io_core_in_peer_state_moved.get_logger(), Log_component::S_TRANSPORT),
219 m_worker(sync_io_core_in_peer_state_moved.get_logger(), sync_io_core_in_peer_state_moved.nickname()),
220 // Adopt the just-cted, idle sync_io:: core.
221 m_sync_io(std::move(sync_io_core_in_peer_state_moved))
222 // m_sync_io_adapter is null but is set-up shortly below.
223{
224 using util::sync_io::Asio_waitable_native_handle;
225 using util::sync_io::Task_ptr;
226 using flow::util::ostream_op_string;
227
228 m_worker.start();
229
230 // We're using a boost.asio event loop, so we need to base the async-waited-on handles on our Task_engine.
231#ifndef NDEBUG
232 bool ok =
233#endif
234 m_sync_io.replace_event_wait_handles([this]() -> Asio_waitable_native_handle
235 { return Asio_waitable_native_handle(*(m_worker.task_engine())); });
236 assert(ok && "Did you break contract by passing-in a non-fresh sync_io core object to ctor?");
237
238 /* Have to do this after .replace_event_wait_handles() by the adapter's ctor's contract.
239 * As of this writing that's the only reason m_sync_io_adapter is optional<>. */
240 m_sync_io_adapter.emplace(get_logger(), ostream_op_string("Blob_stream_mq_receiver [", *this, ']'),
241 &m_worker, &m_sync_io);
242} // Blob_stream_mq_receiver_impl::Blob_stream_mq_receiver_impl()

244template<typename Persistent_mq_handle>
245Blob_stream_mq_receiver_impl<Persistent_mq_handle>::~Blob_stream_mq_receiver_impl()
246{
247 using flow::async::Single_thread_task_loop;
248 using flow::util::ostream_op_string;
249
250 // We are in thread U. By contract in doc header, they must not call us from a completion handler (thread W).
251
252 FLOW_LOG_INFO("Blob_stream_mq_receiver [" << *this << "]: Shutting down. All our "
253 "internal async handlers will be canceled; and worker thread will be joined.");
254
255 /* This (1) stop()s the Task_engine thus possibly
256 * preventing any more handlers from running at all (any handler possibly running now is the last one to run); (2)
257 * at that point Task_engine::run() exits, hence thread W exits; (3) joins thread W (waits for it to
258 * exit); (4) returns. That's a lot, but it's non-blocking. */
259 m_worker.stop();
260 // Thread W is (synchronously!) no more.
261
262 FLOW_LOG_INFO("Blob_stream_mq_receiver [" << *this << "]: Continuing shutdown. "
263 "Next we will run user handler (if any) from some other thread. "
264 "In this user thread we will await those handlers' completion and then return.");
265
266 // See comment in similar spot in Native_socket_stream::~Impl() regarding the following.
267
268 FLOW_LOG_INFO("Blob_stream_mq_receiver [" << *this << "]: Continuing shutdown. Next we will run pending handlers "
269 "from some other thread. In this user thread we will await those handlers' completion and then "
270 "return.");
271
272 Single_thread_task_loop one_thread(get_logger(), ostream_op_string(nickname(), "-temp_deinit"));
273 one_thread.start([&]()
274 {
275 FLOW_LOG_INFO("Blob_stream_mq_receiver [" << *this << "]: "
276 "In transient finisher thread: Shall run all pending internal handlers (typically none).");
277
278 const auto task_engine = m_worker.task_engine();
279 task_engine->restart();
280 const auto count = task_engine->poll();
281 if (count != 0)
282 {
283 FLOW_LOG_INFO("Blob_stream_mq_receiver [" << *this << "]: "
284 "In transient finisher thread: Ran [" << count << "] internal handlers after all.");
285 }
286 task_engine->stop();
287
288 FLOW_LOG_INFO("Transient finisher exiting. (Receive-ops de-init may follow.)");
289 });
290 // Here thread exits/joins synchronously. (But the adapter might run its own similar ones.)
291} // Blob_stream_mq_receiver_impl::~Blob_stream_mq_receiver_impl()

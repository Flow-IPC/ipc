29/**
30 * Core internally-used implementation of shm::classic::Client_session: it is to the latter what its `public`
31 * super-class Client_session_impl is to #Client_session.
32 *
33 * @see shm::classic::Client_session doc header which covers both its public behavior/API and a detailed sketch
34 * of the entire implementation.
35 *
36 * ### Impl notes ###
37 * The use of app_shm() and session_shm(), and various public APIs around them, is totally symmetrical with
38 * Server_session_impl; so that all is in super-class Session_impl. We do what's asymmetrical:
39 * assigning session_shm() and app_shm() values; and provide the client-specific APIs (notably
40 * async_connect()).
41 *
42 * @tparam S_MQ_TYPE_OR_NONE
43 * See shm::classic::Client_session counterpart.

137TEMPLATE_CLSC_CLI_SESSION_IMPL
138template<typename Task_err>
139bool CLASS_CLSC_CLI_SESSION_IMPL::async_connect
140 (const typename Base::Base::Base::Mdt_builder_ptr& mdt,
141 typename Base::Base::Base::Channels* init_channels_by_cli_req_pre_sized,
142 typename Base::Base::Base::Mdt_reader_ptr* mdt_from_srv_or_null,
143 typename Base::Base::Base::Channels* init_channels_by_srv_req,
144 Task_err&& on_done_func)
145{
146 using boost::movelib::make_unique;
147 using boost::make_shared;
148 using boost::shared_ptr;
149 using Channels = typename Base::Base::Base::Channels;
150 using Mdt_reader_ptr = typename Base::Base::Base::Mdt_reader_ptr;
151
152 /* Very simple really: server is in charge of setting up the SHM arenas; we just open them.
153 * Unlike on the server side, no one but the us-calling user needs to be gated on our successfully opening
154 * them; so we can simply perform the entire vanilla Base::Base::async_connect(), and if that went fine then
155 * open the arenas.
156 *
157 * The only wrinkle is, what (unlikely though it is) the opening fails. *(Base*)this is already in PEER
158 * state by then. Well, cancel_peer_state_to_null() exists for that purpose. A bit cheesy? One could say that...
159 * but it's not that bad.
160 *
161 * Okay, another wrinkle; don't set the out-args, until we are really successful; so put stuff
162 * into some intermediate targets first. */
163
164 auto temp_init_channels_by_cli_req_pre_sized
165 = init_channels_by_cli_req_pre_sized ? make_shared<Channels>(init_channels_by_cli_req_pre_sized->size())
166 : shared_ptr<Channels>();
167 auto temp_mdt_from_srv_or_null
168 = mdt_from_srv_or_null ? make_shared<Mdt_reader_ptr>() : shared_ptr<Mdt_reader_ptr>();
169 auto temp_init_channels_by_srv_req
170 = init_channels_by_srv_req ? make_shared<Channels>() : shared_ptr<Channels>();
171
172 return Base::Base::async_connect(mdt,
173 temp_init_channels_by_cli_req_pre_sized.get(),
174 temp_mdt_from_srv_or_null.get(),
175 temp_init_channels_by_srv_req.get(),
176 [this, on_done_func = std::move(on_done_func),
177 init_channels_by_cli_req_pre_sized, mdt_from_srv_or_null, init_channels_by_srv_req,
178 temp_init_channels_by_cli_req_pre_sized, // @todo Avoid s_p<> copies here.
179 temp_mdt_from_srv_or_null,
180 temp_init_channels_by_srv_req]
181 (const Error_code& async_err_code)
182 {
183 if (async_err_code)
184 {
185 // No questions asked. GTFO.
186 on_done_func(async_err_code);
187 return;
188 }
189 /* else: Cool; just do what shm::classic::Server_session_impl did except much simpler:
190 * - Merely open, no creating.
191 * - Lifetime of the Arena (pool handle) equals `*this`; so no need to interface with any daddy
192 * server object; just do it. */
193
194 auto err_code = async_err_code; // We may need to modify it anyway below, so make a copy.
195 const auto srv_app_name = Shared_name::ct(Base::Base::Base::m_srv_app_ref.m_name);
196 const auto cli_app_name = Shared_name::ct(Base::Base::Base::cli_app_ptr()->m_name);
197 const auto& srv_namespace = Base::Base::Base::srv_namespace();
198
199 decltype(m_app_shm) session_shm;
200 auto shm_pool_name = build_conventional_shared_name(Shared_name::S_RESOURCE_TYPE_ID_SHM,
201 srv_app_name, srv_namespace, cli_app_name,
202 Base::Base::Base::cli_namespace())
203 / SHM_SUBTYPE_PREFIX
204 / Shared_name::S_1ST_OR_ONLY;
205 session_shm = make_unique<typename Base::Arena>
206 (get_logger(), shm_pool_name, util::OPEN_ONLY, false, &err_code);
207
208 if (!err_code)
209 {
210 shm_pool_name = build_conventional_shared_name(Shared_name::S_RESOURCE_TYPE_ID_SHM,
211 srv_app_name, srv_namespace, cli_app_name)
212 / SHM_SUBTYPE_PREFIX
213 / Shared_name::S_1ST_OR_ONLY;
214
215 assert(!m_app_shm);
216 m_app_shm = make_unique<typename Base::Arena>
217 (get_logger(), shm_pool_name, util::OPEN_ONLY, false, &err_code);
218 } // if (!err_code) (but may have become truthy inside)
219
220 if (err_code)
221 {
222 // Get back to NULL state all around. Note: do *not* ever clean underlying shared resources (server's job!).
223 assert(!m_app_shm);
224 Base::Base::cancel_peer_state_to_null();
225 }
226 // else { Well... great! Stay in PEER state. }
227 Base::init_shm_arenas(std::move(session_shm), m_app_shm.get()); // Might be saving nulls (if err_code truthy).
228
229 if (!err_code)
230 {
231 // Still good! Finalize out-args.
232 if (init_channels_by_cli_req_pre_sized)
233 {
234 *init_channels_by_cli_req_pre_sized = std::move(*temp_init_channels_by_cli_req_pre_sized);
235 }
236 if (mdt_from_srv_or_null)
237 {
238 *mdt_from_srv_or_null = std::move(*temp_mdt_from_srv_or_null);
239 }
240 if (init_channels_by_srv_req)
241 {
242 *init_channels_by_srv_req = std::move(*temp_init_channels_by_srv_req);
243 }
244 } // if (!err_code)
245
246 on_done_func(err_code); // err_code could be truthy.
247 }); // auto real_pre_rsp_setup_func =
248} // Client_session_impl::async_connect()

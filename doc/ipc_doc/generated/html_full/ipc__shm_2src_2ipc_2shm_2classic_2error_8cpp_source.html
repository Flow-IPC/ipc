93std::string Category::message(int val) const // Virtual.
94{
95 using std::string;
96
97 // KEEP THESE STRINGS IN SYNC WITH COMMENT IN error.hpp ON THE INDIVIDUAL ENUM MEMBERS!
98
99 // See notes in transport::error in same spot.
100 switch (static_cast<Code>(val))
101 {
102 case Code::S_SHM_BIPC_MISC_LIBRARY_ERROR:
103 return "Low-level boost.ipc.shm: boost.interprocess emitted miscellaneous library exception sans a system code; "
104 "a WARNING message at throw-time should contain all possible details.";
105
106 case Code::S_END_SENTINEL:
107 assert(false && "SENTINEL: Not an error. "
108 "This Code must never be issued by an error/success-emitting API; I/O use only.");
109 }
110 assert(false);
111 return "";
112} // Category::message()

114util::String_view Category::code_symbol(Code code) // Static.
115{
116 // Note: Must satisfy istream_to_enum() requirements.
117
118 switch (code)
119 {
120 case Code::S_SHM_BIPC_MISC_LIBRARY_ERROR:
121 return "SHM_BIPC_MISC_LIBRARY_ERROR";
122
123 case Code::S_END_SENTINEL:
124 return "END_SENTINEL";
125 }
126 assert(false);
127 return "";
128}

136std::istream& operator>>(std::istream& is, Code& val)
137{
138 /* Range [<1st Code>, END_SENTINEL); no match => END_SENTINEL;
139 * allow for number instead of ostream<< string; case-insensitive. */
140 val = flow::util::istream_to_enum(&is, Code::S_END_SENTINEL, Code::S_END_SENTINEL, true, false,
141 Code(S_CODE_LOWEST_INT_VALUE));
142 return is;
143}

38/**
39 * Internal-use workhorse containing common elements of Client_session_adapter and Server_session_adapter.
40 * As such it contains the machinery for:
41 * - storing (deriving from) the adapted `Session` as accessible via Session_adapter::core();

43 * - storing and invoking the central util::sync_io::Event_wait_func supplied by the user;

252 /// Set of result arg values from a successful passive-channel-open from a #Session_obj invoking #On_channel_func.
253 struct Channel_open_result
254 {
255 // Types.
256
257 /// Short-hand for pointer wrapper around a `*this`.
258 using Ptr = boost::movelib::unique_ptr<Channel_open_result>;
259
260 // Data.
261
262 /// Result 1/2 given about to be given to #m_on_channel_func_or_empty.
263 Channel_obj m_channel;
264
265 /// Result 2/2 given about to be given to #m_on_channel_func_or_empty.
266 Mdt_reader_ptr m_mdt_reader_ptr;
267 };
268
269 /// Queue of Channel_open_result.
270 using Channel_open_result_q = std::queue<typename Channel_open_result::Ptr>;
271
272 // Methods.
273
274 /**
275 * Signaled by the function returned by on_channel_func_sio(), it returns the IPC-pipe to steady-state (empty,
276 * not readable), invokes the original user handler passed to on_channel_func_sio(), and lastly
277 * begins the next async-wait for the procedure.
278 */
279 void on_ev_channel_open();
280
281 /**
282 * Returns the proper on-error handler to set up on the underlying #Session_obj (`Client_session`:
283 * via ctor; `Server_session`: via `init_handlers()`).
284 *
285 * The resulting handler must not be invoked before start_ops() and #m_on_err_func being set;
286 * else behavior undefined.
287 *
288 * @return See above.
289 */
290 flow::async::Task_asio_err on_err_func_sio();
291
292 /**
293 * Returns the proper on-passive-channel-open handler to set up on the underlying #Session_obj (`Client_session`:
294 * via ctor; `Server_session`: via `init_handlers()` 2-arg form).
295 *
296 * The resulting handler must not be invoked before start_ops() and #m_on_channel_func_or_empty being set;
297 * else behavior undefined.
298 *
299 * @return See above.
300 */
301 On_channel_func on_channel_func_sio();
302
303 // Data.
304
305 /**
306 * The `Task_engine` for `m_ready_*`. It is necessary to construct those pipe-end objects, but we never
307 * use that guy's `->async_*()` APIs -- only non-blocking operations, essentially leveraging boost.asio's
308 * portable transmission APIs but not its actual, um, async-I/O abilities in this case. Accordingly we
309 * never load any tasks onto #m_nb_task_engine and certainly never `.run()` (or `.poll()` or ...) it.
310 *
311 * In the `sync_io` pattern the user's outside event loop is responsible for awaiting readability/writability
312 * of a guy like #m_ready_reader_chan via our exporting of its `.native_handle()`.
313 */
314 flow::util::Task_engine m_nb_task_engine;
315
316 /**
317 * The `Task_engine` for `m_ev_wait_hndl_*`, unless it is replaced via replace_event_wait_handles().
318 *
319 * This is to fulfill the `sync_io` pattern.
320 */
321 flow::util::Task_engine m_ev_hndl_task_engine_unused;
322
323 /**
324 * Read-end of IPC-pipe used by `*this` used to detect that the error-wait has completed. The signal byte
325 * is detected in #m_ready_reader_err. There is no need to read it, as an error can occur at most once.
326 *
327 * @see #m_ready_writer_err.
328 */
329 util::Pipe_reader m_ready_reader_err;
330
331 /// Write-end of IPC-pipe together with #m_ready_reader_err.
332 util::Pipe_writer m_ready_writer_err;
333
334 /**
335 * Descriptor waitable by outside event loop async-waits -- storing the same `Native_handle` as (and thus being
336 * used to wait on events from) #m_ready_reader_err.
337 */
338 util::sync_io::Asio_waitable_native_handle m_ev_wait_hndl_err;
339
340 /**
341 * Read-end of IPC-pipe used by `*this` used to detect that a channel-open-wait has completed. The signal byte
342 * is read out of #m_ready_reader_chan, after it was written there via #m_ready_writer_chan. As explained in
343 * #m_target_channel_open_q doc header, channel-opens can occur at any time from another thread, even while
344 * `*this` is synchronously dealing with an existing one; therefore this is a rare case where more than 1 byte
345 * could be readable at a given time. Nevertheless we issue one async-wait on #m_ready_reader_chan per
346 * channel-open, in series; eventually all of them get popped, meaning all the channel-opens have been served.
347 *
348 * This is not a rare occurrence necessarily: an opposing `Session` user can perform 2+ `open_channel()` right in
349 * a row, and since creating a `Channel` with all its 2-4 fat peer objects can be relatively lengthy on this
350 * side, the 2nd/3rd/... on-channel-open firing can "step on" a preceding one.
351 *
352 * @see #m_ready_writer_chan.
353 */
354 util::Pipe_reader m_ready_reader_chan;
355
356 /// Write-end of IPC-pipe together with #m_ready_reader_chan.
357 util::Pipe_writer m_ready_writer_chan;
358
359 /**
360 * Descriptor waitable by outside event loop async-waits -- storing the same `Native_handle` as (and thus being
361 * used to wait on events from) #m_ready_reader_chan.
362 */
363 util::sync_io::Asio_waitable_native_handle m_ev_wait_hndl_chan;
364
365 /**
366 * Function (set forever in start_ops()) through which we invoke the outside event loop's
367 * async-wait facility for descriptors/events relevant to our ops. See util::sync_io::Event_wait_func
368 * doc header for a refresher on this mechanic.
369 */
370 util::sync_io::Event_wait_func m_ev_wait_func;
371
372 /// `on_err_func` from init_handlers(); `.empty()` until then.
373 flow::async::Task_asio_err m_on_err_func;
374
375 /**
376 * `on_passive_open_channel_func_or_empty` from init_handlers() (possibly `.empty()` if not supplied); until then
377 * `.empty()`.
378 */
379 On_channel_func m_on_channel_func_or_empty;
380
381 /// Result given to (or about to be given to) #m_on_err_func.
382 Error_code m_target_err_code_err;
383
384 /**
385 * Queue of #On_channel_func handler arg sets received from async-I/O #Session_obj -- meaning
386 * the `Session`, in unspecified background thread, informing us (which we signal via #m_ready_writer_chan)
387 * a channel has been passively open -- and not yet fed to #m_on_channel_func_or_empty.
388 *
389 * Protected by #m_target_channel_open_q_mutex.
390 *
391 * ### Rationale ###
392 * There's a queue here, which stands in contrast to (as of this writing) all other `sync_io` pattern impls
393 * in the project, including the on-error stuff in `*this`: those always deal with a single async-op at a time,
394 * and therefore no queue is necessary (just a single arg-set; usually just an `Error_code`, sometimes
395 * with a `size_t sz` though).
396 *
397 * The reason this one is different is as follows.
398 * - It's unlike, e.g., Client_session_adapter::async_connect(), because an async-connect is triggered by the
399 * user explicitly, which triggers a single async-wait, and we disallow in our API to trigger more, until
400 * that one completes.
401 * - It's unlike, e.g., #m_on_err_func stuff in `*this`, because while that one can indeed happen at any time
402 * from another thread -- without the user explicitly beginning the async-op (it sort of begins by itself) --
403 * it can also only happen at most once per `*this` (modulo move-assignment).
404 *
405 * At any rate: the handler we register with #m_async_io -- on_channel_func_sio() -- can be invoked
406 * from a background thread at any time. This triggers signaling #m_ready_reader_chan (via #m_ready_writer_chan),
407 * on which there is always a `sync_io`-pattern async-wait outstanding. We are informed of a completed
408 * async-wait (byte is available on #m_ready_reader_chan); we consume the byte and pop and pass the result
409 * from top of #m_target_channel_open_q. Then we start another async-wait. If there are more items in the queue
410 * that were added in the meantime, the same number of bytes have been pushed into #m_ready_writer_chan;
411 * so those async-waits are satisfied (in series) quickly, leading to the progressive emptying of this queue
412 * and passing along those results to `m_on_channel_func_or_empty()`.
413 */
414 Channel_open_result_q m_target_channel_open_q;
415
416 /// Protects #m_target_channel_open_q, accessed from user async-wait-reporter thread; and #Session_obj worker thread.
417 mutable flow::util::Mutex_non_recursive m_target_channel_open_q_mutex;
418
419 /**
420 * This guy does all the work. In our dtor this will be destroyed (hence thread stopped) first-thing,
421 * unless it is `.reset()` even earlier in dtor_stop().
422 */
423 std::optional<Async_io_obj> m_async_io;

536template<typename Session>
537void Session_adapter<Session>::init_pipe(util::Pipe_reader* reader, util::Pipe_writer* writer,
538 util::sync_io::Asio_waitable_native_handle* ev_wait_hndl)
539{
540 using util::Native_handle;
541 using boost::asio::connect_pipe;
542
543 Error_code sys_err_code;
544
545 connect_pipe(*reader, *writer, sys_err_code);
546 if (sys_err_code)
547 {
548 FLOW_LOG_FATAL("Session_adapter [" << *m_async_io << "]: Constructing: connect-pipe failed. Details follow.");
549 FLOW_ERROR_SYS_ERROR_LOG_FATAL();
550 assert(false && "We chose not to complicate the code given how unlikely this is, and how hosed you'd have to be.");
551 std::abort();
552 }
553
554 ev_wait_hndl->assign(Native_handle(reader->native_handle()));
555}

557template<typename Session>
558template<typename Event_wait_func_t>
559bool Session_adapter<Session>::start_ops(Event_wait_func_t&& ev_wait_func)
560{
561 using util::Task;
562
563 if (!m_ev_wait_func.empty())
564 {
565 FLOW_LOG_WARNING("Session_adapter [" << *m_async_io << "]: Start-ops requested, "
566 "but we are already started. Probably a user bug, but it is not for us to judge.");
567 return false;
568 }
569 // else
570
571 m_ev_wait_func = std::move(ev_wait_func);
572
573 /* No time to waste! on_err_func_sio() and on_channel_func_sio() had to have been called already; begin waiting
574 * for those events to be indicated via our IPC-pipes as set up in those methods. */
575
576 async_wait(&m_ev_wait_hndl_err,
577 false, // Wait for read.
578 boost::make_shared<Task>([this]()
579 {
580 FLOW_LOG_INFO("Session_adapter [" << *m_async_io << "]: Async-IO core on-error event: informed via IPC-pipe; "
581 "invoking handler.");
582 util::pipe_consume(get_logger(), &m_ready_reader_err); // No need really -- it's a one-time thing -- but just....
583
584 auto on_done_func = std::move(m_on_err_func);
585 m_on_err_func.clear(); // In case move() didn't do it. Might as well forget it.
586
587 on_done_func(m_target_err_code_err);
588 FLOW_LOG_TRACE("Handler completed.");
589 }));
590
591 /* Subtlety: It is tempting to guard this with `if (!m_on_channel_func_or_empty.empty()) {}` in the effort to
592 * avoid an an async-wait that will never get satisfied. However that only works in the Client_session case,
593 * when m_on_channel_func_or_empty is known from ction. Server_session still needs init_handlers() by user,
594 * and indeed that would be after start_ops(), so => bug. We could add fancier logic to account for this,
595 * but who really cares? So it'll register an extra FD in some boost.asio epoll-set. Meh. */
596 async_wait(&m_ev_wait_hndl_chan,
597 false, // Wait for read.
598 boost::make_shared<Task>([this]() { on_ev_channel_open(); }));
599
600 FLOW_LOG_INFO("Session_adapter [" << *m_async_io << "]: Start-ops requested. Done.");
601 return true;
602} // Session_adapter::start_ops()

604template<typename Session>
605void Session_adapter<Session>::on_ev_channel_open()
606{
607 using util::Task;
608 using flow::util::Lock_guard;
609
610 // We are in user calling thread.
611
612 typename Channel_open_result::Ptr result;
613 {
614 Lock_guard<decltype(m_target_channel_open_q_mutex)> lock(m_target_channel_open_q_mutex);
615
616 FLOW_LOG_INFO("Session_adapter [" << *m_async_io << "]: Async-IO core passively-opened channel event: "
617 "informed via IPC-pipe; invoking handler. Including this one "
618 "[" << m_target_channel_open_q.size() << "] are pending.");
619
620 assert((!m_target_channel_open_q.empty())
621 && "Algorithm bug? Result-queue elements and pipe signal bytes must be 1-1, so either something "
622 "failed to correctly push, or something overzealously popped.");
623
624 result = std::move(m_target_channel_open_q.front());
625 m_target_channel_open_q.pop();
626 } // Lock_guard lock(m_target_channel_open_q_mutex);
627
628 m_on_channel_func_or_empty(std::move(result->m_channel), std::move(result->m_mdt_reader_ptr));
629 FLOW_LOG_TRACE("Handler completed. Beginning next async-wait immediately. If more is/are pending "
630 "it/they'll be popped quickly due to immediately-completing async-wait(s).");
631
632 util::pipe_consume(get_logger(), &m_ready_reader_chan);
633 async_wait(&m_ev_wait_hndl_chan,
634 false, // Wait for read -- again.
635 boost::make_shared<Task>([this]() { on_ev_channel_open(); }));
636} // Session_adapter::on_ev_channel_open()

640bool Session_adapter<Session>::replace_event_wait_handles(const Create_ev_wait_hndl_func& create_ev_wait_hndl_func)

642 using util::Native_handle;
643
644 if (!m_ev_wait_func.empty())
645 {
646 FLOW_LOG_WARNING("Session_adapter [" << *m_async_io << "]: Cannot replace event-wait handles after "

648 return false;
649 }
650 // else
651
652 FLOW_LOG_INFO("Session_adapter [" << *m_async_io << "]: Replacing event-wait handles (probably to replace underlying "
653 "execution context without outside event loop's boost.asio Task_engine or similar).");
654

656 assert(m_ev_wait_hndl_chan.is_open());

662 saved = m_ev_wait_hndl_chan.release();
663 m_ev_wait_hndl_chan = create_ev_wait_hndl_func();

665
666 return true;
667} // Session_adapter::replace_event_wait_handles()

691template<typename Session>
692typename Session_adapter<Session>::On_channel_func
693 Session_adapter<Session>::on_channel_func_sio()
694{
695 using flow::util::Lock_guard;
696
697 // Careful in here! *this may not have been constructed yet.
698
699 return [this](Channel_obj&& new_channel, Mdt_reader_ptr&& new_channel_mdt)
700 {
701 // We are in Client_session_mv/Server_session_mv "unspecified" worker thread (called thread W in internal docs).
702 {
703 Lock_guard<decltype(m_target_channel_open_q_mutex)> lock(m_target_channel_open_q_mutex);
704
705 FLOW_LOG_INFO("Session_adapter [" << *m_async_io << "]: Async-IO core reports passively-opened channel event: "
706 "tickling IPC-pipe to inform user. This will make the # of pending such events "
707 "[" << (m_target_channel_open_q.size() + 1) << "].");
708 m_target_channel_open_q.emplace(boost::movelib::make_unique<Channel_open_result>());
709 auto& result = *(m_target_channel_open_q.back());
710 result.m_channel = std::move(new_channel);
711 result.m_mdt_reader_ptr = std::move(new_channel_mdt);
712 } // Lock_guard lock(m_target_channel_open_q_mutex);
713
714 /* By Session contract, handlers are never called concurrently with each other. Though this
715 * in POSIX is thread-safe even otherwise (w/r/t another pipe_produce() and most certainly w/r/t
716 * pipe_consume() of the other end). */
717 util::pipe_produce(get_logger(), &m_ready_writer_chan);
718 }; // return [](){}
719
720 /* start_ops() will set up the (first) async-wait for m_ready_reader_err being readable, as well as
721 * the action on that wait being satisfied; namely: to pipe_consume() that signal + begin the next wait. */
722} // Session_adapter::on_channel_func_sio()

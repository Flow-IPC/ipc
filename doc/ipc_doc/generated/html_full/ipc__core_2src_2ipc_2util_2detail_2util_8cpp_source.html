52/**
53 * Maps general Permissions_level specifier to low-level #Permissions value, when the underlying resource
54 * is in the file-system (e.g., a file) and is *produced* (and therefore always writable) by the owning (creating,
55 * updating) user; but *consumed* (and therefore at most readable, never writable) by potentially other processes,
56 * and therefore possibly other users (e.g., group access or unrestricted access -- but only for reading either way).

59 */

67void pipe_produce(flow::log::Logger* logger_ptr, Pipe_writer* pipe)
68{
69 using util::Blob_const;
70
71 /* As promised, we will block if the pipe is full.
72 *
73 * @todo It would still be nicer if we made pipe non-blocking and detected would-block here and at least
74 * assert()ed in that case. However their boost::asio::*able_pipe doesn't expose such a method
75 * (in contrast to various sockets)... which is a bit odd/annoying actually. We could use ::fcntl() but...
76 * anyway, that's the to-do. Actually we could perhaps use async_write_some() with, like, a timeout...
77 * or something.... */
78
79 const uint8_t PAYLOAD = '\0';
80
81 Error_code sys_err_code;
82 const bool ok = pipe->write_some(Blob_const(&PAYLOAD, 1), sys_err_code) == 1;
83
84 if (sys_err_code || (!ok))
85 {
86 FLOW_LOG_SET_CONTEXT(logger_ptr, Log_component::S_UTIL);
87
88 FLOW_LOG_FATAL("Pipe sync-write failed; this should never happen. Details follow.");
89 FLOW_ERROR_SYS_ERROR_LOG_FATAL();
90 assert(false && "Pipe-write bug?"); std::abort();
91 return;
92 }
93 // else
94} // pipe_produce()

119void remove_persistent_shm_pool(flow::log::Logger* logger_ptr, const Shared_name& pool_name, Error_code* err_code)
120{
121 using bipc::shared_memory_object;
122 using boost::system::system_category;
123
124 if (flow::error::exec_void_and_throw_on_error
125 ([&](Error_code* actual_err_code)
126 { remove_persistent_shm_pool(logger_ptr, pool_name, actual_err_code); },
127 err_code, "remove_persistent_shm_pool()"))
128 {
129 return;
130 }
131 // else
132
133 FLOW_LOG_SET_CONTEXT(logger_ptr, Log_component::S_UTIL);
134
135 FLOW_LOG_INFO("SHM pool [" << pool_name << "]: Removing persistent SHM pool if possible.");
136 const bool ok = shared_memory_object::remove(pool_name.native_str()); // Does not throw.
137
138 if (ok)
139 {
140 err_code->clear();
141 return;
142 }
143 /* smo::remove() is strangely gimped -- though I believe so are the other kernel-persistent remove()s
144 * throughout bipc -- it does not throw and merely returns true or false and no code. Odd, since there can
145 * be at least a couple of reasons one would fail to delete.... However, in POSIX, the Boost 1.78 source code
146 * shows that smo::remove() calls shared_memory_object::remove() which calls some internal
147 * ipcdetail::delete_file() which calls... drumroll... freakin' ::unlink(const char*). Hence we haxor: */
148#ifndef FLOW_OS_LINUX // @todo Should maybe check Boost version or something too?
149# error "Code in remove_persistent_shm_pool() relies on Boost invoking Linux unlink() with errno."
150#endif
151 const auto& sys_err_code = *err_code = Error_code(errno, system_category());
152 FLOW_ERROR_SYS_ERROR_LOG_WARNING();
153} // remove_persistent_shm_pool()

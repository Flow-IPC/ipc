43Shared_name build_conventional_shared_name
44 (const Shared_name& resource_type, const Shared_name& srv_app_name,
45 const Shared_name& srv_namespace, const Shared_name& cli_app_name,
46 const Shared_name& cli_namespace_or_sentinel)
47{
48 // Refer to Shared_name class doc header for the following. Make sure the two always match when modifying either.
49
50#ifndef NDEBUG
51 const auto& SENTINEL = Shared_name::S_SENTINEL;
52#endif
53
54 assert(!srv_namespace.empty());
55 assert(!cli_app_name.empty());
56 assert(!cli_namespace_or_sentinel.empty());
57
58 assert((srv_namespace != SENTINEL) && "Perhaps you want to use the other build_conventional_shared_name()?");
59 assert((cli_app_name != SENTINEL) && "Perhaps you want to use the other build_conventional_shared_name()? "
60 "It does support non-SENTINEL srv_namespace.");
61
62 auto name = build_conventional_shared_name_prefix(resource_type, srv_app_name);
63
64 assert(name.has_trailing_separator());
65 name += srv_namespace;
66
67 name /= cli_app_name;
68 name /= cli_namespace_or_sentinel;

70 return name;

93Shared_name build_conventional_shared_name_prefix(const Shared_name& resource_type, const Shared_name& srv_app_name)
94{
95 // Refer to Shared_name class doc header for the following. Make sure the two always match when modifying either.
96
97 const auto& ROOT_MAGIC = Shared_name::S_ROOT_MAGIC;
98#ifndef NDEBUG
99 const auto& SENTINEL = Shared_name::S_SENTINEL;
100#endif
101
102 assert(!resource_type.empty());
103 assert(!srv_app_name.empty());
104 assert((srv_app_name != SENTINEL) && "Perhaps you want to use build_conventional_non_session_based_shared_name()?");
105
106 Shared_name name;
107 name /= ROOT_MAGIC;
108 name /= resource_type;
109 name /= srv_app_name;
110 name /= Shared_name();
111 return name;

114bool decompose_conventional_shared_name(const Shared_name& name,
115 Shared_name* resource_type, Shared_name* srv_app_name,
116 Shared_name* srv_namespace, Shared_name* cli_app_name,
117 Shared_name* cli_namespace_or_sentinel, Shared_name* the_rest)
118{
119 using flow::util::ostream_op_string;
120 using std::regex_match;
121 using std::smatch;
122 using std::string;
123
124 /* This is thread-safe local `static`; actual local `static` is supposedly thread-safe in C++1x but with caveats.
125 * So just ensure it. Why not just make it global `static`? Answer: it relies on extern `static`s which
126 * may not yet be initialized (static init ordering problem). Anyway it's nice to keep the regex string local. */
127 std::call_once(non_global_scope_name_regex_built, [&]()
128 {
129 // Compatible with build_conventional_shared_name(1).
130 non_global_scope_name_regex.assign
131 (ostream_op_string(Shared_name::S_SEPARATOR, Shared_name::S_ROOT_MAGIC.str(), Shared_name::S_SEPARATOR,
132 "([^", Shared_name::S_SEPARATOR, "]+)", Shared_name::S_SEPARATOR, // resource_type
133 "([^", Shared_name::S_SEPARATOR, "]+)", Shared_name::S_SEPARATOR, // srv_app_name
134 "([^", Shared_name::S_SEPARATOR, "]+)", Shared_name::S_SEPARATOR, // srv_namespace
135 "([^", Shared_name::S_SEPARATOR, "]+)", Shared_name::S_SEPARATOR, // cli_app_name
136 "([^", Shared_name::S_SEPARATOR, "]+)", Shared_name::S_SEPARATOR, // cli_namespace_or_sentinel
137 "(.*)")); // the_rest
138 });
139
140 smatch matches;
141 if (!regex_match(name.str(), matches, non_global_scope_name_regex))
142 {
143 return false;
144 }
145 // else
146
147 // Just one little thing: srv_namespace is promised to never equal SENTINEL. Hence:
148 const auto srv_namespace_str = matches[3].str();
149 if (srv_namespace_str == Shared_name::S_SENTINEL.str())
150 {
151 return false;
152 }
153 // else
154
155 if (resource_type)
156 {
157 *resource_type = Shared_name::ct(matches[1].str());
158 }
159 if (srv_app_name)
160 {
161 *srv_app_name = Shared_name::ct(matches[2].str());
162 }
163 if (srv_namespace)
164 {
165 *srv_namespace = Shared_name::ct(srv_namespace_str);
166 }
167 if (cli_app_name)
168 {
169 *cli_app_name = Shared_name::ct(matches[4].str());
170 }
171 if (cli_namespace_or_sentinel)
172 {
173 *cli_namespace_or_sentinel = Shared_name::ct(matches[5].str());
174 }
175 if (the_rest)
176 {
177 *the_rest = Shared_name::ct(matches[6].str());
178 }
179
180 return true;
181} // decompose_conventional_shared_name(1)

183bool decompose_conventional_shared_name(const Shared_name& name,
184 Shared_name* resource_type, Shared_name* srv_app_name,
185 Shared_name* srv_namespace_or_sentinel, Shared_name* the_rest)
186{
187 using flow::util::ostream_op_string;
188 using std::regex_match;
189 using std::smatch;
190 using std::string;
191
192 // Analogous to the other overload, just simpler (fewer fields to decompose-to). Keeping comments light.
193
194 std::call_once(global_scope_name_regex_built, [&]()
195 {
196 // Compatible with build_conventional_shared_name(2).
197 global_scope_name_regex.assign
198 (ostream_op_string(Shared_name::S_SEPARATOR, Shared_name::S_ROOT_MAGIC.str(), Shared_name::S_SEPARATOR,
199 "([^", Shared_name::S_SEPARATOR, "]+)", Shared_name::S_SEPARATOR, // resource_type
200 "([^", Shared_name::S_SEPARATOR, "]+)", Shared_name::S_SEPARATOR, // srv_app_name
201 "([^", Shared_name::S_SEPARATOR, "]+)", Shared_name::S_SEPARATOR, // srv_namespace_or_sentinel
202 Shared_name::S_SENTINEL.str(), Shared_name::S_SEPARATOR,
203 "(.*)")); // the_rest
204 });
205
206 smatch matches;
207 if (!regex_match(name.str(), matches, global_scope_name_regex))

209 return false;
210 }
211 // else

213 if (resource_type)
214 {
215 *resource_type = Shared_name::ct(matches[1].str());
216 }
217 if (srv_app_name)
218 {
219 *srv_app_name = Shared_name::ct(matches[2].str());
220 }
221 if (srv_namespace_or_sentinel)
222 {
223 *srv_namespace_or_sentinel = Shared_name::ct(matches[3].str());
224 }
225 if (the_rest)
226 {

228 }

231} // decompose_conventional_shared_name(2)

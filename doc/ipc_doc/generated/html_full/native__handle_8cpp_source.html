47Native_handle& Native_handle::operator=(Native_handle&& src)
48{
49 using std::swap;
50
51 if (*this != src)
52 {
53 m_native_handle = S_NULL_HANDLE;
54 swap(m_native_handle, src.m_native_handle); // No-op if `&src == this`.

49/**
50 * Helper of async_write_with_native_handle() used as the callback executed when waiting for writability of
51 * the given connected local peer socket, with the idea to execute a native `sendmsg()` call and all further
52 * parts of the async send op started by async_write_with_native_handle(). This is used as the callback either
53 * by async_write_with_native_handle() or by itself upon `sendmsg()` encountering would-block, in which case
54 * it must wait again and invoke itself asynchronously (and so on).
55 *
56 * It is *not* used once at least 1 byte (and therefore also the native handle) have been successfuly sent.
57 *
58 * @tparam Task_err
59 * See async_write_with_native_handle().
60 * @param logger_ptr
61 * See async_write_with_native_handle().
62 * @param sys_err_code
63 * The error code yielded by the `Peer_socket::async_wait()` async call for which we are the handler arg.
64 * This can be success or error but not a would-block error.
65 * @param peer_socket
66 * See async_write_with_native_handle().
67 * @param payload_hndl
68 * See async_write_with_native_handle().
69 * @param payload_blob
70 * See async_write_with_native_handle().
71 * @param on_sent_or_error
72 * See async_write_with_native_handle(). Note that we, too, will likely blow it away via move semantics.
73 */
74template<typename Task_err>
75void on_wait_writable_or_error(flow::log::Logger* logger_ptr,
76 const Error_code& sys_err_code,
77 Native_handle payload_hndl,
78 const util::Blob_const& payload_blob,
79 Peer_socket* peer_socket,
80 Task_err&& on_sent_or_error);

82/**
83 * Helper of async_read_with_target_func() containing its core (asynchronously) recursive implementation.
84 * Call with template param `TARGET_TBD == true` to start the wait/async-read chain; then it asynchronously
85 * triggers the `TARGET_TBD == false` form which then repats that as many times as needed to complete
86 * the target buffer.
87 *
88 * It is also used directly in `TARGET_TBD == false` mode by async_read_interruptible() which is a simpler
89 * version of async_read_with_target_func() that knows the target buffer from the start.
90 *
91 * @tparam TARGET_TBD
92 * Must be `true` when async_read_with_target_func() async-invokes it, `false` when it async-invokes
93 * itself, or if async_read_interruptible() invokes it.
94 * @tparam Task_err_blob
95 * See async_read_with_target_func().
96 * @tparam Target_payload_blob_func
97 * See async_read_with_target_func(). However, it can be anything (suggest `int`),
98 * if `TARGET_TBD == false` and is ignored in that case.
99 * @tparam Should_interrupt_func
100 * See async_read_with_target_func().
101 * @param logger_ptr
102 * See async_read_with_target_func().
103 * @param sys_err_code
104 * The error code yielded by the `Peer_socket::async_wait()` async call for which we are the handler arg.
105 * This can be success or error but not a would-block error.
106 * @param peer_socket
107 * See async_read_with_target_func().
108 * @param should_interrupt_func
109 * See async_read_with_target_func().
110 * @param on_rcvd_or_error
111 * See async_read_with_target_func().
112 * @param target_payload_blob_func
113 * See async_read_with_target_func(). However, it can be anything (suggest `0`),
114 * if `TARGET_TBD == false` and is ignored in that case.
115 * @param target_payload_blob
116 * Ignored if `TARGET_TBD == true`; otherwise it must be the result of
117 * `target_payload_blob_func()` during the `TARGET_TBD == false` invocation in this operation.
118 * @param n_rcvd_so_far
119 * Must be 0 if `TARGET_TBD == true`; else it must equal the number of leading bytes of `target_payload_blob`
120 * that have already been filled out by reading from socket. (Note this may be 0 even so, though it's
121 * unlikely.)
122 */
123template<bool TARGET_TBD,
124 typename Task_err_blob, typename Target_payload_blob_func, typename Should_interrupt_func>
125void on_wait_readable_or_error(flow::log::Logger* logger_ptr, const Error_code& sys_err_code,
126 Peer_socket* peer_socket,
127 Should_interrupt_func&& should_interrupt_func,
128 Task_err_blob&& on_rcvd_or_error,
129 Target_payload_blob_func&& target_payload_blob_func,
130 util::Blob_mutable target_payload_blob,
131 size_t n_rcvd_so_far);

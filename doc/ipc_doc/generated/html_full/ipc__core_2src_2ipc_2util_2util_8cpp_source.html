46void set_resource_permissions(flow::log::Logger* logger_ptr, const fs::path& path,
47 const Permissions& perms, Error_code* err_code)
48{
49 using boost::system::system_category;
50 using ::open;
51 using ::close;
52 // using ::O_PATH; // It's a macro apparently.
53 // using ::errno; // It's a macro apparently.
54
55 if (flow::error::exec_void_and_throw_on_error
56 ([&](Error_code* actual_err_code)
57 { set_resource_permissions(logger_ptr, path, perms, actual_err_code); },
58 err_code, "util::set_resource_permissions(1)"))
59 {
60 return;
61 }
62 // else
63
64 FLOW_LOG_SET_CONTEXT(logger_ptr, Log_component::S_UTIL);
65
66 /* We can either reuse the handle-based overload, or we can do fs::permissions. The latter is perhaps more
67 * portable and certainly a few lines easier and logically simpler. Nevertheless I (ygoldfel) chose the former,
68 * and that is for consistency (and not just in a cosmetic sense; I'd like both routines to work and fail in
69 * similar ways with fewer degrees of freedom on how to differ based on some 3rd-party developer's whims).
70 * (@todo It is certainly arguable.) Thing is, there's an `fs` way of doing fs::permissions(path), but
71 * there's no fs::permissions(<FD>). So we'd have to use a native call for the latter but `fs` for the other
72 * former? Nah. Let's just do it native all the way but build one on top of the other. Also this repeats
73 * the technique in session::ensure_resource_owner_is_app(). */
74
75 int native_handle = open(path.c_str(), O_RDONLY); // Read-only is sufficient for ::fchmod().
76 if (native_handle == -1)
77 {
78 *err_code = Error_code(errno, system_category());
79 FLOW_LOG_WARNING("Tried to set permissions of resource at [" << path << "] but while obtaining info-only handle "
80 "encountered error [" << *err_code << "] [" << err_code->message() << "]; unable to proceed.");
81 return;
82 }
83 // else
84 Native_handle handle(native_handle);
85
86 // For nicer messaging add some more logging on error. A little code duplication, but it's OK.
87 set_resource_permissions(logger_ptr, handle, perms, err_code);
88 if (*err_code)
89 {
90 FLOW_LOG_WARNING("Tried to set permissions of resource at [" << path << "], upon successfully opening probe-only "
91 "descriptor/handle, but this resulted in error; see preceding WARNING referencing all "
92 "other details.");
93 }
94
95 close(native_handle);
96} // set_resource_permissions(1)

98void set_resource_permissions(flow::log::Logger* logger_ptr, Native_handle handle,
99 const Permissions& perms, Error_code* err_code)
100{
101 using boost::system::system_category;
102 using ::fchmod;
103 // using ::errno; // It's a macro apparently.
104
105 if (flow::error::exec_void_and_throw_on_error
106 ([&](Error_code* actual_err_code)
107 { set_resource_permissions(logger_ptr, handle, perms, actual_err_code); },
108 err_code, "util::set_resource_permissions(2)"))
109 {
110 return;
111 }
112 // else
113
114 assert((!handle.null()) && "Disallowed per contract.");
115
116 FLOW_LOG_SET_CONTEXT(logger_ptr, Log_component::S_UTIL);
117
118 // There's no Boost.filesystem or STL way to set mode, that I know of (via FD); so use POSIX-y thing.
119 const auto rc = fchmod(handle.m_native_handle, perms.get_permissions());
120 if (rc == -1)
121 {
122 *err_code = Error_code(errno, system_category());
123 FLOW_LOG_WARNING("Tried to set permissions of resources via descriptor/handle [" << handle << "] but encountered "
124 "error [" << *err_code << "] [" << err_code->message() << "]; unable to check.");
125 }
126 else
127 {
128 err_code->clear();
129 }
130
131 // As promised don't log anything else (not even TRACE) and leave that to the caller if desired.
132} // set_resource_permissions(2)

134bool process_running(process_id_t process_id)
135{
136 using boost::system::system_category;
137 using ::kill;
138 // using ::errno; // It's a macro apparently.
139
140 // The below, down to the impl, is described in the doc header. Note these are POSIX semantics.
141#if !defined(FLOW_OS_LINUX) && !defined(FLOW_OS_MAC)
142# error "Using POSIX ::kill() semantics. @todo Should check for POSIX/Unix; not Linux or Mac. Add a FLOW_OS_...."
143#endif
144
145 if (kill(process_id, 0) == 0)
146 {
147 return true;
148 }
149 // else
150 assert((Error_code(errno, system_category()) == boost::system::errc::no_such_process)
151 && "The fake signal zero is valid and is not actually sent so no permission problem possible; hence "
152 "ESRCH or equivalent must be the only possible error.");
153 return false;
154} // process_running()

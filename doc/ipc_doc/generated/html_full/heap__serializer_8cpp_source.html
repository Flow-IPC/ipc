71void Heap_fixed_builder::emit_serialization(Segment_ptrs* target_blobs, [[maybe_unused]] const Session& session,
72 Error_code* err_code) const
73{
74 using util::Blob_const;
75
76 if (flow::error::exec_void_and_throw_on_error
77 ([&](Error_code* actual_err_code) { emit_serialization(target_blobs, session, actual_err_code); },
78 err_code, "Heap_fixed_builder::emit_serialization()"))
79 {
80 return;
81 }
82 // If got here: err_code is not null.
83
84 assert(m_engine && "Are you operating on a moved-from `*this`?");
85
86 /* Reminder: concept API says emit_serialization() can be called >1 times. We do allow that in the below.
87 * Keep in mind that mutations can be executed in-between, so whatever caching for perf one imagines has
88 * to not mess up in the face of that. */
89
90 Segment_ptrs& blob_ptrs = *target_blobs;
91
92 size_t n_blobs = blob_ptrs.size();
93 m_engine->emit_segment_blobs(&blob_ptrs); // It appends, like we must. Just be careful in remembering that below.
94 n_blobs = blob_ptrs.size() - n_blobs;
95 assert((n_blobs > 0) && "Empty serialization? What went wrong?");
96
97 for (size_t abs_idx = blob_ptrs.size() - n_blobs, idx = 0;
98 idx != n_blobs;
99 ++abs_idx, ++idx)
100 {
101 const auto blob = blob_ptrs[abs_idx];
102 const auto seg_size = blob->size();
103
104 FLOW_LOG_TRACE("Heap_fixed_builder [" << *this << "]: "
105 "Serialization segment [" << idx << "] (0 based, of [" << n_blobs << "], 1-based): "
106 "Heap buffer @[" << static_cast<const void*>(blob->begin()) << "] sized [" << seg_size << "].");
107
108 // Check for the one error condition: a too-large segment.
109 if (seg_size > m_segment_sz)
110 {
111 FLOW_LOG_WARNING("Heap_fixed_builder [" << *this << "]: "
112 "Serialization segment [" << idx << "] (0 based, of [" << n_blobs << "], 1-based): "
113 "Heap buffer @[" << static_cast<const void*>(blob->begin()) << "] "
114 "sized [" << seg_size << "]: exceeded limit "
115 "[" << m_segment_sz << "]. Emitting error; will not return serialization.");
116 *err_code = error::Code::S_INTERNAL_ERROR_SERIALIZE_LEAF_TOO_BIG;
117 return;
118 }
119 // else
120 } // for (blob : blob_ptrs, starting from first appended one)
121 // Got here: no error. Done and done.
122} // Heap_fixed_builder::emit_serialization()

138Heap_reader::Heap_reader(const Config& config) :
139 flow::log::Log_context(config.m_logger_ptr, Log_component::S_TRANSPORT)
140{
141 FLOW_LOG_TRACE("Heap_reader [" << *this << "]: Heap reader started: "
142 "expecting [" << config.m_n_segments_est << "] segments (0 if unknown).");
143 m_serialization_segments.reserve(config.m_n_segments_est);
144}

146Heap_reader::~Heap_reader()
147{
148 if (!m_engine)
149 {
150 FLOW_LOG_TRACE("Heap_reader [" << *this << "]: Reader being destroyed including: "
151 "serialization containing [" << m_serialization_segments.size() << "] segments; "
152 "no deserialization yet attempted.");
153 }
154 else
155 {
156 // This *might* be perf-pricey, to some extent, so definitely be careful with the log level.
157 FLOW_LOG_TRACE("Heap_reader [" << *this << "]: Reader being destroyed including: "
158 "serialization containing [" << m_serialization_segments.size() << "] segments; "
159 "last deserialization reports [" << (m_engine->sizeInWords() * sizeof(::capnp::word)) << "] "
160 "input heap space total.");
161 }
162}

164flow::util::Blob* Heap_reader::add_serialization_segment(size_t max_sz)
165{
166 using flow::util::Blob;
167
168 const auto blob = new Blob(get_logger(), max_sz);
169 m_serialization_segments.emplace_back(blob);
170
171 FLOW_LOG_TRACE("Heap_reader [" << *this << "]: "
172 "Serialization segment [" << (m_serialization_segments.size() - 1) << "] (0-based) added/owned: "
173 "Heap buffer @[" << static_cast<const void*>(m_serialization_segments.back()->const_data()) << "] "
174 "max-sized [" << max_sz << "].");
175
176 return blob;
177}

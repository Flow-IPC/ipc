34size_t nb_write_some_with_native_handle(flow::log::Logger* logger_ptr,
35 Peer_socket* peer_socket_ptr,
36 Native_handle payload_hndl, const util::Blob_const& payload_blob,
37 Error_code* err_code)
38{
39 namespace bind_ns = flow::util::bind_ns;
40 using boost::system::system_category;
41 using boost::array;
42 namespace sys_err_codes = boost::system::errc;
43 using ::sendmsg;
44 using ::msghdr;
45 using ::iovec;
46 using ::cmsghdr;
47 // using ::SOL_SOCKET; // It's a macro apparently.
48 using ::SCM_RIGHTS;
49 using ::MSG_DONTWAIT;
50 using ::MSG_NOSIGNAL;
51 // using ::errno; // It's a macro apparently.
52
53 FLOW_ERROR_EXEC_FUNC_AND_THROW_ON_ERROR(size_t, nb_write_some_with_native_handle,
54 logger_ptr, peer_socket_ptr, payload_hndl, bind_ns::cref(payload_blob), _1);
55 // ^-- Call ourselves and return if err_code is null. If got to present line, err_code is not null.
56
57 /* To reader/maintainer: The below isn't that difficult, but if you need exact understanding and definitely if you
58 * plan to make changes -- even just clarifications or changes in the doc header -- then read the entire doc header
59 * of nb_read_some_with_native_handle() first. */
60
61 assert(peer_socket_ptr);
62 auto& peer_socket = *peer_socket_ptr;
63
64 FLOW_LOG_SET_CONTEXT(logger_ptr, Log_component::S_TRANSPORT);
65 FLOW_LOG_TRACE("Connected local peer socket wants to write from location @ [" << payload_blob.data() << "] "
66 "plus native handle [" << payload_hndl << "]. Will try to send.");
67
68 /* Let us send as much as possible of payload_blob; and the native socket payload_hndl.
69 * As explicitly documented in boost.asio docs: it does not provide an API for the latter (fairly hairy
70 * and Linux-specific) feature, but it is doable by using sendmsg() natively via peer_socket.native_handle()
71 * which gets the native handle (a/k/a FD).
72 *
73 * Side note: After I (ygoldfel) had written this and async_write_with_native_handle() (which calls us),
74 * I found validation for the approach in boost.asio docs in their example for Peer_socket::native_non_blocking(),
75 * which shows how to implement async_sendfile(), spiritually similar to our thing which is essentially
76 * a case of an analogous hypothetical async_sendmsg(). Subtlety: we, too, could have used native_non_blocking(),
77 * as their example does; and the only reason we did not is that `sendmsg()` happens to support a per-call flag
78 * that forces a non-blocking transmission, so one needn't deal with socket state. Subtlety: realize (as per
79 * boost.asio docs) that Peer_socket::non_blocking() is orthogonal to this entirely; it is orthogonal to
80 * native_non_blocking() and the per-call flag thing. */
81
82 iovec native_buf1 = { const_cast<void*>(payload_blob.data()), payload_blob.size() };
83 // (const_cast<> is OK, as sendmsg() is read-only. Probably the API is const-janky due to iovec's reuse in recvmsg().)
84
85 msghdr sendmsg_hdr =
86 {
87 0, // msg_name - Address; not used; we are connected.
88 0, // msg_namelen - Ditto.
89 /* msg_iov - Scatter/gather array. We just put the one blob here. It's natural to instead put a
90 * a buffer sequence, meaning N>=1 buffers, here. There's a to-do elsewhere as of this writing to
91 * make that API as well; as of now it's not needed. Take a look inside boost.asio for how they convert
92 * between the 2; it's not rocket science, but we can either reuse their code or learn from it;
93 * for example there's the matter of MAX_IOV (max size of this array). */
94 &native_buf1,
95 1, // msg_iovlen - # elements in msg_iov.
96
97 /* The other 3 fields can be uninitialized: msg_flags is unused; msg_control[len] are set below.
98 * The language won't let me leave them as garbage though so: */
99 0, 0, 0
100 };
101
102 /* What's left is to set up msg_control[len] which is how we send the native handle (payload_hndl).
103 * This is based on example snippet taken from Linux `man cmsg`. FWIW it's rather difficult to figure out
104 * how to write it from the rest of the documentation, which is likely complete and correct but opaque; the
105 * example is invaluable. Note we are particularly using the SOL_SOCKET/SCM_RIGHTS technique. */
106
107 constexpr size_t N_PAYLOAD_FDS = 1;
108 union
109 {
110 /* (Comment taken verbatim from aforementioned `man`.)
111 * Ancillary data buffer, wrapped in a union in order to ensure it is suitably aligned.
112 *
113 * ...If I (ygoldfel) understand correctly, and I think I do, the idea is that:
114 * - msg_control points to a contiguous data area (ancillary data); which is split up into variable-length
115 * sub-areas, each of which starts with a header, in the form of a `cmsghdr`, which contains a couple of
116 * enum-like ints, a cmsg_len specifying the length of the rest of that sub-area (within the sequence),
117 * and then that area which contains the stuff to be transmitted (whose exact size/meaning depends on the
118 * enums; in our case it'll be native handles).
119 * - A key caveat is that in order to be properly interpreted, the header -- probably cast by the kernel/something
120 * to `cmsghdr` -- must begin on an aligned address. So, in particular, if whatever comes before it
121 * (could be the preceding ancillary data sub-area, depending) ends just before an odd address, then
122 * the data sub-area must be a byte or a few to the right.
123 * - So this trick explicitly puts a cmsghdr there; so that its co-inhabitant m_buf (the actual area where we'll
124 * shove handles) will also begin on the same address/byte.
125 * - CMSG_ALIGN() is a Linux extension that might be usable to avoid having to use the trick, but seeing as
126 * how `man cmsg` in my Linux says it's not portable and itself foregoes it in favor of the trick in the
127 * example, I'll just also use the trick. */
128 array<uint8_t, CMSG_SPACE(sizeof(Native_handle::handle_t) * N_PAYLOAD_FDS)> m_buf;
129 cmsghdr m_align;
130 } msg_control_as_union;
131
132 sendmsg_hdr.msg_control = msg_control_as_union.m_buf.c_array();
133 sendmsg_hdr.msg_controllen = sizeof(msg_control_as_union.m_buf);
134 cmsghdr* const sendmsg_hdr_cmsg_ptr = CMSG_FIRSTHDR(&sendmsg_hdr); // Or cast from &m_buf; or use &m_align....
135 sendmsg_hdr_cmsg_ptr->cmsg_level = SOL_SOCKET;
136 sendmsg_hdr_cmsg_ptr->cmsg_type = SCM_RIGHTS;
137 sendmsg_hdr_cmsg_ptr->cmsg_len = CMSG_LEN(sizeof(Native_handle::handle_t) * N_PAYLOAD_FDS);
138
139 // Copy the FDs. We have just the one; simply assign (omit `memcpy`) but static_assert() to help future-proof.
140 static_assert(N_PAYLOAD_FDS == 1, "Should be only passing one native handle into sendmsg() as of this writing.");
141 *(reinterpret_cast<Native_handle::handle_t*>(CMSG_DATA(sendmsg_hdr_cmsg_ptr))) = payload_hndl.m_native_handle;
142
143 const auto n_sent_or_error
144 = sendmsg(peer_socket.native_handle(), &sendmsg_hdr,
145 /* If socket is un-writable then don't block;
146 * EAGAIN/EWOULDBLOCK instead. Doing it this way is easier than messing with fcntl(), particularly
147 * since we're working with a boost.asio-managed socket; that's actually fine -- there's a portable
148 * API native_non_blocking() in boost.asio for setting this -- but feels like the less interaction between
149 * portable boost.asio code we use and this native stuff, the better -- so just keep it local here. */
150 MSG_DONTWAIT | MSG_NOSIGNAL);
151 // ^-- Dealing with SIGPIPE is a pointless pain; if conn closed that way just give as an EPIPE error.
152
153 if (n_sent_or_error == -1)
154 {
155 /* Not even 1 byte of the blob was sent; and hence nor was payload_hndl. (Docs don't explicitly say that 2nd
156 * part after the semicolon, but it's essentially impossible that it be otherwise, as then it'd be unknowable.
157 * Update: Confirmed in kernel source and supported by this delightfully reassuring (in several ways) link:
158 * [ https://gist.github.com/kentonv/bc7592af98c68ba2738f4436920868dc ] (Googled "SCM_RIGHTS gist").) */
159
160 const Error_code sys_err_code(errno, system_category());
161 if ((sys_err_code == sys_err_codes::operation_would_block) || // EWOULDBLOCK
162 (sys_err_code == sys_err_codes::resource_unavailable_try_again)) // EAGAIN (same meaning)
163 {
164 FLOW_LOG_TRACE("Write attempt indicated would-block; not an error condition. Nothing sent.");
165 /* Subtlety: We could just set it to sys_err_code; but we specifically promised in contract we'd set it to the
166 * boost::asio::error::would_block representation of would-block condition. Why did we promise that? 2 reasons:
167 * 1, that is what boost.asio's own spiritually similar Peer_socket::write_some() would do in non_blocking() mode.
168 * 2, then we can promise a specific code instead of making them check for the above 2 errno values.
169 *
170 * Subtlety: net_flow::Peer_socket::sync_receive uses somewhat different semantics; it indicates would-block by
171 * returning 0 *but* a falsy *err_code. Why are we inconsistent with that? Answer: Because net_flow is not
172 * trying to be a boost.asio extension; we are. In net_flow's context (as of this writing) no one is surprised
173 * when semantics are somewhat different from boost.asio; but in our context they might be quite surprised
174 * indeed. */
175 *err_code = boost::asio::error::would_block;
176 return 0;
177 }
178 // else
179
180 assert(sys_err_code);
181
182 // All other errors are fatal.
183 FLOW_ERROR_SYS_ERROR_LOG_WARNING(); // Log based on sys_err_code.
184 FLOW_LOG_WARNING("Connected local peer socket tried to write from "
185 "location @ [" << payload_blob.data() << "] plus native handle [" << payload_hndl << "]; "
186 "but an unrecoverable error occurred. Nothing sent.");
187 *err_code = sys_err_code;
188 return 0;
189 } // if (n_sent_or_error == -1)
190 // else if (n_sent_or_error != -1)
191
192 assert (n_sent_or_error > 0);
193
194 FLOW_LOG_TRACE("sendmsg() reports the native handle [" << payload_hndl << "] was successfully sent; as "
195 "were [" << n_sent_or_error << "] of the blob's [" << payload_blob.size() << "] bytes.");
196 err_code->clear();
197 return n_sent_or_error;
198} // nb_write_some_with_native_handle()

200size_t nb_read_some_with_native_handle(flow::log::Logger* logger_ptr,
201 Peer_socket* peer_socket_ptr,
202 Native_handle* target_payload_hndl_ptr,
203 const util::Blob_mutable& target_payload_blob,
204 Error_code* err_code,
205 int message_flags)
206{
207 namespace bind_ns = flow::util::bind_ns;
208 using boost::system::system_category;
209 using boost::array;
210 namespace sys_err_codes = boost::system::errc;
211 using ::sendmsg;
212 using ::msghdr;
213 using ::iovec;
214 using ::cmsghdr;
215 // using ::SOL_SOCKET; // It's a macro apparently.
216 using ::SCM_RIGHTS;
217 using ::MSG_DONTWAIT;
218 using ::MSG_CTRUNC;
219 // using ::errno; // It's a macro apparently.
220
221 FLOW_ERROR_EXEC_FUNC_AND_THROW_ON_ERROR(size_t, nb_read_some_with_native_handle,
222 logger_ptr, peer_socket_ptr, target_payload_hndl_ptr,
223 bind_ns::cref(target_payload_blob), _1, message_flags);
224 // ^-- Call ourselves and return if err_code is null. If got to present line, err_code is not null.
225
226 /* To reader/maintainer: The below isn't that difficult, but if you need exact understanding and definitely if you
227 * plan to make changes -- even just clarifications or changes in the doc header -- then read the entire doc header
228 * of nb_read_some_with_native_handle() first. */
229
230 assert(peer_socket_ptr);
231 assert(target_payload_hndl_ptr);
232
233 auto& peer_socket = *peer_socket_ptr;
234 auto& target_payload_hndl = *target_payload_hndl_ptr;
235
236 FLOW_LOG_SET_CONTEXT(logger_ptr, Log_component::S_TRANSPORT);
237 FLOW_LOG_TRACE("Connected local peer socket wants to read up to [" << target_payload_blob.size() << "] bytes "
238 "to location @ [" << target_payload_blob.data() << "] "
239 "plus possibly a native handle. Will try to receive.");
240 target_payload_hndl = Native_handle();
241 assert(target_payload_hndl.null()); // We promised to set to this if no socket received (including on error).
242
243 /* Let us receive as much as possible into target_payload_blob up to its size; and a native socket (if any) into
244 * target_payload_hndl (if none, then it'll remain null()).
245 * As explicitly documented in boost.asio docs: it does not provide an API for the latter (fairly hairy
246 * and Linux-specific) feature, but it is doable by using recvmsg() natively via peer_socket.native_handle()
247 * which gets the native handle (a/k/a FD).
248 *
249 * Recommending first looking at nb_write_some_with_native_handle(), as we operate symmetrically/similarly. */
250
251 iovec native_buf1 = { target_payload_blob.data(), target_payload_blob.size() };
252 msghdr recvmsg_hdr =
253 {
254 0, // msg_name - Address; not used; we are connected.
255 0, // msg_namelen - Ditto.
256 /* msg_iov - Scatter/gather array. We just put the one target blob here. See similarly-themed notes at similar
257 * spot in nb_write_some_with_native_handle(); they apply here too, more or less. */
258 &native_buf1,
259 1, // msg_iovlen - # elements in msg_iov. This is an *input* arg only; output is total length read: the ret value.
260 /* The following fields may or may need not be initialized; but let's discuss them generally:
261 * msg_control, msg_controllen - These are out-args, indirectly set/reserved and interpreted with CMSG* below.
262 * msg_flags - This is an out-arg containing special feature flags. These may be checked below after call.
263 * The language won't let me leave them as garbage though so: */
264 0, 0, 0
265 };
266 // May not be needed, but at least some examples online do it; seems prudent and cheap.
267 recvmsg_hdr.msg_flags = 0;
268
269 /* Set up .msg_control*. This, and then interpreting the output after the call, is based on the out-equivalent in
270 * nb_write_some_with_native_handle() as well as cross-referencing with some Internet sources.
271 *
272 * Before the call, we must reserve space for the ancillary out-data, if any; set msg_control to point to that;
273 * and msg_controllen to the size of that thing. */
274
275 constexpr size_t N_PAYLOAD_FDS = 1;
276 union
277 {
278 // See nb_write_some_with_native_handle(). It explains the m_align thing.
279 array<uint8_t, CMSG_SPACE(sizeof(Native_handle::handle_t) * N_PAYLOAD_FDS)> m_buf;
280 cmsghdr m_align;
281 } msg_control_as_union;
282 // Probably paranoia. Just in case pre-fill it with zeroes (0x00... is not a valid FD) for sanity check later.
283 msg_control_as_union.m_buf.fill(0);
284
285 recvmsg_hdr.msg_control = msg_control_as_union.m_buf.c_array();
286 recvmsg_hdr.msg_controllen = sizeof(msg_control_as_union.m_buf);
287
288 const auto n_rcvd_or_error
289 = recvmsg(peer_socket.native_handle(), &recvmsg_hdr,
290 /* If socket is un-writable then don't block; EAGAIN/EWOULDBLOCK instead. ...Further comment omitted;
291 * see sendmsg() elsewhere in this .cpp. Same thing here. */
292 MSG_DONTWAIT | message_flags); // <-- Re. message_flags please read notes in our doc header.
293
294 // Carefully check all the outputs. Return value first.
295
296 if (n_rcvd_or_error == -1)
297 {
298 /* Not even 1 byte of a blob was read; and hence nor was any target_payload_hndl. (See comment in similar
299 * sport in nb_write_some_with_native_handle(); applies here equally.) */
300
301 const Error_code sys_err_code(errno, system_category());
302 if ((sys_err_code == sys_err_codes::operation_would_block) || // EWOULDBLOCK
303 (sys_err_code == sys_err_codes::resource_unavailable_try_again)) // EAGAIN (same meaning)
304 /* Reader: "But, ygoldfel, why didn't you just check errno itself against those 2 actual Evalues?"
305 * ygoldfel: "Because it makes me feel better about portable-ish style. Shut up, that's why. Get off my lawn." */
306 {
307 FLOW_LOG_TRACE("Read attempt indicated would-block; not an error condition. Nothing received.");
308 // Subtlety x 2: ...omitted. See similar spot in nb_write_some_with_native_handle(). Same here.
309 *err_code = boost::asio::error::would_block;
310 return 0; // target_payload_hndl already set.
311 }
312 // else
313
314 assert(sys_err_code);
315
316 // All other errors are fatal.
317 FLOW_ERROR_SYS_ERROR_LOG_WARNING(); // Log based on sys_err_code.
318 FLOW_LOG_WARNING("Connected local peer socket tried to read up to [" << target_payload_blob.size() << "] bytes at "
319 "location @ [" << target_payload_blob.data() << "] plus possibly a native handle; "
320 "but an unrecoverable error occurred. Nothing received.");
321 *err_code = sys_err_code;
322 return 0; // target_payload_hndl already set.
323 } // if (n_rcvd_or_error == -1)
324 // else if (n_rcvd_or_error != -1)
325
326 if (n_rcvd_or_error == 0)
327 {
328 /* WARNING doesn't feel right: it's a graceful connection end.
329 * INFO could be good, but it might be too verbose depending on the application.
330 * Use TRACE to be safe; caller can always log differently if desired. */
331 FLOW_LOG_TRACE("Connected local peer socket tried to read up to [" << target_payload_blob.size() << "] bytes at "
332 "location @ [" << target_payload_blob.data() << "] plus possibly a native handle; "
333 "but it returned EOF meaning orderly connection shutdown by peer. Nothing received.");
334 *err_code = boost::asio::error::eof;
335 return 0; // target_payload_hndl already set.
336 }
337 // else
338 assert(n_rcvd_or_error > 0);
339
340 /* Next, target_payload_blob... which is already written to (its first n_rcvd_or_error bytes).
341 *
342 * Next, recvmsg_hdr.msg_flags. Basically only the following is relevant: */
343 if (recvmsg_hdr.msg_flags != 0)
344 {
345 FLOW_LOG_INFO("Connected local peer socket tried to read up to [" << target_payload_blob.size() << "] bytes at "
346 "location @ [" << target_payload_blob.data() << "] plus possibly a native handle; "
347 "and it returned it read [" << n_rcvd_or_error << "] bytes successfully but also returned raw "
348 "out-flags value [0x" << std::hex << recvmsg_hdr.msg_flags << std::dec << "]. "
349 "Will check for relevant flags but otherwise "
350 "ignoring if nothing bad. Logging at elevated level because it's interesting; please investigate.");
351
352 if ((recvmsg_hdr.msg_flags & MSG_CTRUNC) != 0)
353 {
354 FLOW_LOG_WARNING("Connected local peer socket tried to read up to [" << target_payload_blob.size() << "] bytes "
355 "at location @ [" << target_payload_blob.data() << "] plus possibly a native handle; "
356 "and it returned it read [" << n_rcvd_or_error << "] bytes successfully but also returned raw "
357 "out-flags value [0x" << recvmsg_hdr.msg_flags << "] which includes MSG_CTRUNC. "
358 "That flag indicates more stuff was sent as ancillary data; but we expect at most 1 native "
359 "handle. Other side sent something strange. Acting as if nothing received + error.");
360 *err_code = error::Code::S_LOW_LVL_UNEXPECTED_STREAM_PAYLOAD_BEYOND_HNDL;
361 return 0; // target_payload_hndl already set.
362 }
363 // else
364 }
365
366 /* Lastly examine ancillary data (and note MSG_CTRUNC already eliminated above).
367 * Use, basically, the method from `man cmsg` in Linux. */
368 cmsghdr* const recvmsg_hdr_cmsg_ptr = CMSG_FIRSTHDR(&recvmsg_hdr);
369 if (recvmsg_hdr_cmsg_ptr)
370 {
371 // There is some ancillary data. It can only (validly according to our expected protocol) be one thing.
372 if ((recvmsg_hdr_cmsg_ptr->cmsg_level == SOL_SOCKET) &&
373 (recvmsg_hdr_cmsg_ptr->cmsg_type == SCM_RIGHTS))
374 {
375 static_assert(N_PAYLOAD_FDS == 1, "Should be only dealing with one native handle with recvmsg() "
376 "as of this writing.");
377 target_payload_hndl.m_native_handle
378 = *(reinterpret_cast<const Native_handle::handle_t*>(CMSG_DATA(recvmsg_hdr_cmsg_ptr)));
379 }
380 else
381 {
382 FLOW_LOG_WARNING("Connected local peer socket tried to read up to [" << target_payload_blob.size() << "] bytes "
383 "at location @ [" << target_payload_blob.data() << "] plus possibly a native handle; "
384 "and it returned it read [" << n_rcvd_or_error << "] bytes successfully but also "
385 "unexpected ancillary data of csmg_level|cmsg_type "
386 "[" << recvmsg_hdr_cmsg_ptr->cmsg_level << '|' << recvmsg_hdr_cmsg_ptr->cmsg_type << "]. "
387 "Acting as if nothing received + error.");
388 *err_code = error::Code::S_LOW_LVL_UNEXPECTED_STREAM_PAYLOAD_BEYOND_HNDL;
389 return 0; // target_payload_hndl already set.
390 }
391 // else
392 if (CMSG_NXTHDR(&recvmsg_hdr, recvmsg_hdr_cmsg_ptr))
393 {
394 /* This is rather strange: we didn't provide enough space for more ancillary data; yet there is somehow more,
395 * even though we didn't detect MSG_CTRUNC earlier. Well, whatever. It's bad just like MSG_CTRUNC. */
396 FLOW_LOG_WARNING("Connected local peer socket tried to read up to [" << target_payload_blob.size() << "] bytes "
397 "at location @ [" << target_payload_blob.data() << "] plus possibly a native handle; "
398 "and it returned it read [" << n_rcvd_or_error << "] bytes and native handle "
399 "[" << target_payload_hndl << "] but also more ancillary data; but we expect at most 1 native "
400 "handle. Other side sent something strange. Acting as if nothing received + error.");
401 target_payload_hndl = Native_handle(); // Undo the above (to match promised semantics on error).
402 *err_code = error::Code::S_LOW_LVL_UNEXPECTED_STREAM_PAYLOAD_BEYOND_HNDL;
403 return 0;
404 }
405 // else { No more ancillary data, as expected. }
406 }
407 // else { No ancillary data, meaning no native handle; that's quite normal. }
408
409 FLOW_LOG_TRACE("recvmsg() reports receipt of possible native handle [" << target_payload_hndl << "]; as well as "
410 "[" << n_rcvd_or_error << "] of the blob's [" << target_payload_blob.size() << "]-byte capacity.");
411 err_code->clear();
412 return n_rcvd_or_error;
413} // nb_read_some_with_native_handle()

415void release_native_peer_socket(Native_handle&& peer_socket_native_or_null)
416{
417 using flow::util::Task_engine;
418
419 // As promised:
420 if (peer_socket_native_or_null.null())
421 {
422 return;
423 }
424 // else
425
426 /* Purely for style reasons let's wrap it in a boost.asio `socket` and let its dtor
427 * take care of it. ::close() would have worked too, but I suppose this is less "native" and more
428 * consistent. */
429 Task_engine task_engine;
430 [[maybe_unused]] Peer_socket sock(task_engine, Protocol(), peer_socket_native_or_null.m_native_handle);
431
432 peer_socket_native_or_null = Native_handle(); // As promised nullify it.
433
434 // Now destroy `sock`, then destroy task_engine.
435} // release_native_peer_socket()

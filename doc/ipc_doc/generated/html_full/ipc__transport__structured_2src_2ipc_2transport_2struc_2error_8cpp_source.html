100std::string Category::message(int val) const // Virtual.
101{
102 using std::string;
103
104 // KEEP THESE STRINGS IN SYNC WITH COMMENT IN error.hpp ON THE INDIVIDUAL ENUM MEMBERS!
105
106 /* Just create a string (unless compiler is smart enough to do this only once and reuse later)
107 * based on a static char* which is rapidly indexed from val by the switch() statement.
108 * Since the error_category interface requires that message() return a string by value, this
109 * is the best we can do speed-wise... but it should be fine.
110 *
111 * Some error messages can be fancier by specifying outside values (e.g., see
112 * net_flow's S_INVALID_SERVICE_PORT_NUMBER). */
113 switch (static_cast<Code>(val))
114 {
115 case Code::S_STRUCT_CHANNEL_INTERNAL_PROTOCOL_MISMATCH_GOT_NO_HNDL:
116 return "Structured channel: received unexpected payload: message sans native handle along handles-only pipe.";
117 case Code::S_STRUCT_CHANNEL_INTERNAL_PROTOCOL_MISMATCH_GOT_NO_BLOB:
118 return "Structured channel: received unexpected payload: message with empty blob (not even an encoded length).";
119 case Code::S_STRUCT_CHANNEL_INTERNAL_PROTOCOL_MISMATCH_GOT_HNDL:
120 return "Structured channel: received unexpected payload: continuation message contains native handle.";
121 case Code::S_STRUCT_CHANNEL_INTERNAL_PROTOCOL_MISUSED_SCHEMA:
122 return "Structured channel: received structured message with invalid internally-set/used fields.";
123 case Code::S_INTERNAL_ERROR_SERIALIZE_LEAF_TOO_BIG:
124 return "Structured message serialization (e.g., when sending over channel): A user-mutated datum (e.g., "
125 "blob or string) is so large as to require a too-large segment to fit into the prescribed max size (such "
126 "as max blob size in an IPC channel, when using direct serialization sans SHM). Consider using a "
127 "different serialization mechanism.";
128 case Code::S_INTERNAL_ERROR_DESERIALIZE_TARGET_ALLOC_FAILED:
129 return "Structured message deserialization (e.g., when receiving from channel): Reader engine, when asked "
130 "to supply a target memory area for a serialization-storing segment, reports it is unable to obtain an "
131 "area of sufficient size, such as due to running out of mem-pool space. Consider using a different "
132 "deserialization mechanism.";
133 case Code::S_STRUCT_CHANNEL_INTERNAL_PROTOCOL_LOG_IN_MISUSED_SCHEMA:
134 return "Structured channel: log-in phase: received structured message with invalid internally-set/used fields.";
135 case Code::S_STRUCT_CHANNEL_INTERNAL_PROTOCOL_BAD_AUTH:
136 return "Structured channel: "
137 "auth session token in in-message does not match expected value established during log-in.";
138 case Code::S_STRUCT_CHANNEL_INTERNAL_PROTOCOL_INTERNAL_MSG_TYPE_UNKNOWN:
139 return "Structured channel: in internally-generated message the internal message type is unknown.";
140 case Code::S_STRUCT_CHANNEL_GOT_UNEXPECTED_LOG_IN_REQUEST:
141 return "Structured channel: "
142 "log-in phase as server: actual log-in request contents differ from local user expectation.";
143 case Code::S_DESERIALIZE_FAILED_INSUFFICIENT_SEGMENTS:
144 return "Structured message deserialization: Tried to deserialize without enough input segments (e.g., none).";
145 case Code::S_DESERIALIZE_FAILED_SEGMENT_MISALIGNED:
146 return "Structured message deserialization: An input segment is not aligned as required.";
147
148 case Code::S_END_SENTINEL:
149 assert(false && "SENTINEL: Not an error. "
150 "This Code must never be issued by an error/success-emitting API; I/O use only.");
151 }
152 assert(false);
153 return "";
154} // Category::message()

156util::String_view Category::code_symbol(Code code) // Static.
157{
158 // Note: Must satisfy istream_to_enum() requirements.
159
160 switch (code)
161 {
162 case Code::S_STRUCT_CHANNEL_INTERNAL_PROTOCOL_MISMATCH_GOT_NO_HNDL:
163 return "STRUCT_CHANNEL_INTERNAL_PROTOCOL_MISMATCH_GOT_NO_HNDL";
164 case Code::S_STRUCT_CHANNEL_INTERNAL_PROTOCOL_MISMATCH_GOT_NO_BLOB:
165 return "STRUCT_CHANNEL_INTERNAL_PROTOCOL_MISMATCH_GOT_NO_BLOB";
166 case Code::S_STRUCT_CHANNEL_INTERNAL_PROTOCOL_MISMATCH_GOT_HNDL:
167 return "STRUCT_CHANNEL_INTERNAL_PROTOCOL_MISMATCH_GOT_HNDL";
168 case Code::S_STRUCT_CHANNEL_INTERNAL_PROTOCOL_MISUSED_SCHEMA:
169 return "STRUCT_CHANNEL_INTERNAL_PROTOCOL_MISUSED_SCHEMA";
170 case Code::S_INTERNAL_ERROR_SERIALIZE_LEAF_TOO_BIG:
171 return "INTERNAL_ERROR_SERIALIZE_LEAF_TOO_BIG";
172 case Code::S_INTERNAL_ERROR_DESERIALIZE_TARGET_ALLOC_FAILED:
173 return "INTERNAL_ERROR_DESERIALIZE_TARGET_ALLOC_FAILED";
174 case Code::S_STRUCT_CHANNEL_INTERNAL_PROTOCOL_LOG_IN_MISUSED_SCHEMA:
175 return "STRUCT_CHANNEL_INTERNAL_PROTOCOL_LOG_IN_MISUSED_SCHEMA";
176 case Code::S_STRUCT_CHANNEL_INTERNAL_PROTOCOL_BAD_AUTH:
177 return "STRUCT_CHANNEL_INTERNAL_PROTOCOL_BAD_AUTH";
178 case Code::S_STRUCT_CHANNEL_INTERNAL_PROTOCOL_INTERNAL_MSG_TYPE_UNKNOWN:
179 return "STRUCT_CHANNEL_INTERNAL_PROTOCOL_INTERNAL_MSG_TYPE_UNKNOWN";
180 case Code::S_STRUCT_CHANNEL_GOT_UNEXPECTED_LOG_IN_REQUEST:
181 return "STRUCT_CHANNEL_GOT_UNEXPECTED_LOG_IN_REQUEST";
182 case Code::S_DESERIALIZE_FAILED_INSUFFICIENT_SEGMENTS:
183 return "DESERIALIZE_FAILED_INSUFFICIENT_SEGMENTS";
184 case Code::S_DESERIALIZE_FAILED_SEGMENT_MISALIGNED:
185 return "DESERIALIZE_FAILED_SEGMENT_MISALIGNED";
186
187 case Code::S_END_SENTINEL:
188 return "END_SENTINEL";
189 }
190 assert(false);
191 return "";
192}

200std::istream& operator>>(std::istream& is, Code& val)
201{
202 /* Range [<1st Code>, END_SENTINEL); no match => END_SENTINEL;
203 * allow for number instead of ostream<< string; case-insensitive. */
204 val = flow::util::istream_to_enum(&is, Code::S_END_SENTINEL, Code::S_END_SENTINEL, true, false,
205 Code(S_CODE_LOWEST_INT_VALUE));
206 return is;
207}

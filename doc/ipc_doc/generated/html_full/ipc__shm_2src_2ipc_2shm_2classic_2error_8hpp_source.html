99/**
100 * Ummm -- it specializes this `struct` to -- look -- the end result is boost.system uses this as
101 * authorization to make `enum` `Code` convertible to `Error_code`. The non-specialized
102 * version of this sets `value` to `false`, so that random arbitary `enum`s can't just be used as
103 * `Error_code`s. Note that this is the offical way to accomplish that, as (confusingly but
104 * formally) documented in boost.system docs.
105 */
106template<>
107struct is_error_code_enum<::ipc::shm::classic::error::Code>
108{
109 /// Means `Code` `enum` values can be used for `Error_code`.

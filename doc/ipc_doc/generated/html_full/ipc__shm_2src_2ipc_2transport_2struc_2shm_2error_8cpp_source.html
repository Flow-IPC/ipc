93std::string Category::message(int val) const // Virtual.
94{
95 using std::string;
96
97 // KEEP THESE STRINGS IN SYNC WITH COMMENT IN error.hpp ON THE INDIVIDUAL ENUM MEMBERS!
98
99 /* Just create a string (unless compiler is smart enough to do this only once and reuse later)
100 * based on a static char* which is rapidly indexed from val by the switch() statement.
101 * Since the error_category interface requires that message() return a string by value, this
102 * is the best we can do speed-wise... but it should be fine.
103 *
104 * Some error messages can be fancier by specifying outside values (e.g., see
105 * net_flow's S_INVALID_SERVICE_PORT_NUMBER). */
106 switch (static_cast<Code>(val))
107 {
108 case Code::S_SERIALIZE_FAILED_SESSION_HOSED:
109 return "Structured message serialization: SHM-session `lend_object()` indicated failure when preparing an encoding "
110 "of the location-in-SHM of the message's serialization; the session should be assumed to be down, and "
111 "the opposing process cannot be reached.";
112 case Code::S_DESERIALIZE_FAILED_SESSION_HOSED:
113 return "Structured message deserialization: SHM-session `borrow_object()` indicated failure when interpreting "
114 "an encoding of the location-in-SHM of the message's serialization; the session should be assumed to "
115 "be down, and the opposing process cannot be reached.";
116
117 case Code::S_END_SENTINEL:
118 assert(false && "SENTINEL: Not an error. "
119 "This Code must never be issued by an error/success-emitting API; I/O use only.");
120 }
121 assert(false);
122 return "";
123} // Category::message()

125util::String_view Category::code_symbol(Code code) // Static.
126{
127 // Note: Must satisfy istream_to_enum() requirements.
128
129 switch (code)
130 {
131 case Code::S_SERIALIZE_FAILED_SESSION_HOSED:
132 return "SERIALIZE_FAILED_SESSION_HOSED";
133 case Code::S_DESERIALIZE_FAILED_SESSION_HOSED:
134 return "DESERIALIZE_FAILED_SESSION_HOSED";
135
136 case Code::S_END_SENTINEL:
137 return "END_SENTINEL";
138 }
139 assert(false);
140 return "";
141}

149std::istream& operator>>(std::istream& is, Code& val)
150{
151 /* Range [<1st Code>, END_SENTINEL); no match => END_SENTINEL;
152 * allow for number instead of ostream<< string; case-insensitive. */
153 val = flow::util::istream_to_enum(&is, Code::S_END_SENTINEL, Code::S_END_SENTINEL, true, false,
154 Code(S_CODE_LOWEST_INT_VALUE));
155 return is;
156}

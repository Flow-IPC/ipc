28void capnp_set_lent_shm_handle(schema::ShmHandle::Builder* shm_handle_root,
29 const flow::util::Blob_sans_log_context& lend_result)
30{
31 using util::Blob_mutable;
32
33 assert(!lend_result.empty());
34 assert(shm_handle_root);
35
36 const auto n = lend_result.size();
37
38 /* Avoid wasting internal serialization space if already init...()ed.
39 * (capnp docs state that initX() when `x` is already initX()ed will zero but otherwise leave the previously
40 * initialized field inside the serialization. capnp does not reuse such space for reasons (trade-off).) */
41 const bool field_inited_already = shm_handle_root->hasSerialization();
42 auto capnp_blob_builder = field_inited_already ? shm_handle_root->getSerialization()
43 : shm_handle_root->initSerialization(n);
44 assert(((!field_inited_already) || (capnp_blob_builder.size() == n))
45 && "Please only pass-in uninitialized ShmHandle field or one filled-out by this same function earlier.");
46
47 lend_result.sub_copy(lend_result.begin(), Blob_mutable(capnp_blob_builder.begin(), n));
48} // capnp_set_lent_shm_handle()

50void capnp_get_shm_handle_to_borrow(const schema::ShmHandle::Reader& shm_handle_root,
51 flow::util::Blob_sans_log_context* arg_to_borrow)
52{
53 using util::Blob_const;
54
55 assert(arg_to_borrow);
56
57 const auto capnp_blob_reader = shm_handle_root.getSerialization();
58 arg_to_borrow->assign_copy(Blob_const(capnp_blob_reader.begin(), capnp_blob_reader.size()));
59
60 assert(!arg_to_borrow->empty());
61} // capnp_get_shm_handle_to_borrow()

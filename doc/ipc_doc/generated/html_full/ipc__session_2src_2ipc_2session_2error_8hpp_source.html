36/// Numeric value of the lowest Code.
37constexpr int S_CODE_LOWEST_INT_VALUE = 1;
38
39/**
40 * All possible errors returned (via `Error_code` arguments) by ipc::session functions/methods *outside of*
41 * ipc::transport-triggered errors involved in transport involved in doing session-related work; and possibly
42 * system-triggered errors.
43 *
44 * All notes from transport::error::Code doc header apply here.
45 */
46enum class Code
47{
48 /// User called an API with 1 or more arguments against the API spec.
49 S_INVALID_ARGUMENT = S_CODE_LOWEST_INT_VALUE,
50
51 /// Async completion handler is being called prematurely, because underlying object is shutting down, as user desires.
52 S_OBJECT_SHUTDOWN_ABORTED_COMPLETION_HANDLER,
53
54 /**
55 * Session master channel: log-in as client: The Client Namespace Store (CNS) file (a/k/a PID file) lacks
56 * leading newline-terminated line with numeric PID.
57 */
58 S_CLIENT_NAMESPACE_STORE_BAD_FORMAT,
59
60 /// Session master channel: log-in as client: received log-in response is not the expected internal message.
61 S_CLIENT_MASTER_LOG_IN_RESPONSE_BAD,
62
63 /**
64 * Session master channel: log-in as server: client identifies self as unknown client-app or one that is not
65 * registered as an allowed partner to this server-app.
66 */
67 S_SERVER_MASTER_LOG_IN_REQUEST_CLIENT_APP_DISALLOWED_OR_UNKNOWN,
68
69 /**
70 * Session master channel: log-in as server: client application-level process credentials (UID, etc.)
71 * do not match the registered values and/or the OS-reported client peer values from socket stream.
72 */
73 S_SERVER_MASTER_LOG_IN_REQUEST_CLIENT_APP_INCONSISTENT_CREDS,
74
75 /// Session channel opening: User code chose to reject all passive-open requests.
76 S_SESSION_OPEN_CHANNEL_REMOTE_PEER_REJECTED_PASSIVE_OPEN,
77
78 /// Session channel opening: Server peer is responsible for low-level resource acquisition but failed to do so.
79 S_SESSION_OPEN_CHANNEL_SERVER_CANNOT_PROCEED_RESOURCE_UNAVAILABLE,
80
81 /**
82 * Session channel opening: Other peer must reply quickly enough as-if non-blocking but did not do so in time.
83 * Session continues; this is not a fatal error.
84 */
85 S_SESSION_OPEN_CHANNEL_ACTIVE_TIMEOUT,
86
87 /**
88 * Low-level Boost.ipc.mutex: boost.interprocess emitted miscellaneous library exception sans a system code;
89 * a WARNING message at throw-time should contain all possible details.
90 */
91 S_MUTEX_BIPC_MISC_LIBRARY_ERROR,
92
93 /**
94 * Session master channel: post-log-in setup (such as SHM setup): send failed because channel hosed around that time.
95 * Timing was such that exact reason for this could not be obtained, but logs will show it.
96 */
97 S_SERVER_MASTER_POST_LOG_IN_SEND_FAILED_MISC,
98
99 /**
100 * Session master channel: log-in as server: opposing client's compile-time type config differs from
101 * local server's counterpart config; did you use the proper Server/Client_session paired templates and
102 * mutually equal template parameters?
103 */
104 S_SERVER_MASTER_LOG_IN_REQUEST_CONFIG_MISMATCH,
105
106 /**
107 * A resource in the file system (file, SHM pool, MQ, etc.) has or could have unexpected owner; ipc::session may
108 * emit this when acquiring resources and/or opening session if the server process is not running as the
109 * configured user, or a previous iteration was not running as that user.
110 */
111 S_RESOURCE_OWNER_UNEXPECTED,
112
113 /// SENTINEL: Not an error. This Code must never be issued by an error/success-emitting API; I/O use only.
114 S_END_SENTINEL
115}; // enum class Code

159/**
160 * Ummm -- it specializes this `struct` to -- look -- the end result is boost.system uses this as
161 * authorization to make `enum` `Code` convertible to `Error_code`. The non-specialized
162 * version of this sets `value` to `false`, so that random arbitary `enum`s can't just be used as
163 * `Error_code`s. Note that this is the offical way to accomplish that, as (confusingly but
164 * formally) documented in boost.system docs.
165 */
166template<>
167struct is_error_code_enum<::ipc::session::error::Code>
168{
169 /// Means `Code` `enum` values can be used for `Error_code`.
170 static const bool value = true;
171 // See note in similar place near transport::error.
172};

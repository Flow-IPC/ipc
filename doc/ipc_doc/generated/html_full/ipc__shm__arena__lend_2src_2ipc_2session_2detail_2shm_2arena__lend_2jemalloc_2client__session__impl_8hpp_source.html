278TEMPLATE_JEM_CLI_SESSION_IMPL
279template<typename Task_err>
280bool CLASS_JEM_CLI_SESSION_IMPL::async_connect
281 (const typename Base::Base::Base::Mdt_builder_ptr& mdt,
282 typename Base::Base::Base::Channels* init_channels_by_cli_req_pre_sized,
283 typename Base::Base::Base::Mdt_reader_ptr* mdt_from_srv_or_null,
284 typename Base::Base::Base::Channels* init_channels_by_srv_req,
285 Task_err&& on_done_func)
286{
287 using flow::util::ostream_op_string;
288 using boost::make_shared;
289 using boost::shared_ptr;
290 using std::to_string;
291 using Channels = typename Base::Base::Base::Channels;
292 using Mdt_reader_ptr = typename Base::Base::Base::Mdt_reader_ptr;
293 using Master_structured_channel = typename Base::Base::Base::Master_structured_channel;
294
295 /* This being an arena-lending style of SHM-provider, it's quite different and somewhat more complex than
296 * shm::classic to deal with; but basically there are two items to set up: the Arena(s) where stuff is allocated
297 * on *this* side; and the Shm_session which can lend from said Arena(s) and borrow from *any* opposing Arena(s)
298 * Session_server sets up. They're just objects which are synchronously constructed, and that's it. However
299 * Shm_session requires a Channel over which to conduct internal communications needed by this and opposing
300 * Shm_session. So that's the main difficulty, as it's asynchronous. That said the base vanilla Client_impl
301 * provides some protected methods -- cancel_peer_state_to_connecting() on --
302 * that make it pretty simple. The Channel, specifically, just needs a single
303 * bidirectional Native_socket_stream. Same as when opening that type of channel for the user, Session_server
304 * creates a socket-pair and sends us one of the `Native_handle`s (FDs) to us. So we simply expect_msg() it
305 * on the session-master-channel; we do it after the vanilla async_connect() completes successfully (the log-in
306 * and all that). OK: let's go. First do the vanilla async_connect() but substitute our additional steps
307 * as the handler, remembering the real user-provided handler to invoke eventually, god willing.
308 *
309 * Okay, another wrinkle (same as in shm::classic equivalent async_connect()); don't set the out-args, until
310 * we are really successful; so put stuff into some intermediate targets first. */
311
312 auto temp_init_channels_by_cli_req_pre_sized
313 = init_channels_by_cli_req_pre_sized ? make_shared<Channels>(init_channels_by_cli_req_pre_sized->size())
314 : shared_ptr<Channels>();
315 auto temp_mdt_from_srv_or_null
316 = mdt_from_srv_or_null ? make_shared<Mdt_reader_ptr>() : shared_ptr<Mdt_reader_ptr>();
317 auto temp_init_channels_by_srv_req
318 = init_channels_by_srv_req ? make_shared<Channels>() : shared_ptr<Channels>();
319
320 return Base::Base::async_connect(mdt,
321 temp_init_channels_by_cli_req_pre_sized.get(),
322 temp_mdt_from_srv_or_null.get(),
323 temp_init_channels_by_srv_req.get(),
324 [this, on_done_func = std::move(on_done_func),
325 init_channels_by_cli_req_pre_sized, mdt_from_srv_or_null, init_channels_by_srv_req,
326 temp_init_channels_by_cli_req_pre_sized, // @todo Avoid s_p<> copies here.
327 temp_mdt_from_srv_or_null,
328 temp_init_channels_by_srv_req]
329 (const Error_code& async_err_code)
330 {
331 // We are in thread W (of vanilla Client_session_impl).
332
333 if (async_err_code)
334 {
335 // No questions asked. GTFO.
336 on_done_func(async_err_code);
337 return;
338 }
339 // else: Cool; now do our added steps.
340
341 /* We're in PEER state... but as foreshadowed above, undo that. Provide, as required, the new handler (like us...
342 * but this time really-really final) that will run once either *we*
343 * eventually complete_async_connect_after_canceling_peer_state(success/failure); or a master-channel error
344 * hoses *this. So this is the final stage of this whole thing. */
345 const auto master_channel
346 = Base::Base::cancel_peer_state_to_connecting([on_done_func = std::move(on_done_func),
347 init_channels_by_cli_req_pre_sized,
348 mdt_from_srv_or_null,
349 init_channels_by_srv_req,
350 // @todo Avoid s_p<> copies here --v.
351 temp_init_channels_by_cli_req_pre_sized,
352 temp_mdt_from_srv_or_null,
353 temp_init_channels_by_srv_req]
354 (const Error_code& err_code)
355 {
356 // We are in thread W (of vanilla Client_session_impl).
357
358 if (!err_code)
359 {
360 // Still good! Finalize out-args.
361 if (init_channels_by_cli_req_pre_sized)
362 {
363 *init_channels_by_cli_req_pre_sized = std::move(*temp_init_channels_by_cli_req_pre_sized);
364 }
365 if (mdt_from_srv_or_null)
366 {
367 *mdt_from_srv_or_null = std::move(*temp_mdt_from_srv_or_null);
368 }
369 if (init_channels_by_srv_req)
370 {
371 *init_channels_by_srv_req = std::move(*temp_init_channels_by_srv_req);
372 }
373 } // if (!err_code)
374
375 on_done_func(err_code); // err_code could be truthy.
376 }); // cancel_peer_state_to_connecting()
377
378 /* That was aspirational and hopeful. Now, by cancel_peer_state_to_connecting() contract, actually kick off the
379 * the actions that will async-lead to complete_async_connect_after_canceling_peer_state() which would
380 * cause the above handler to finally run (unless a session-master-channel error does so first).
381 *
382 * Namely we just async-await the message with the Native_handle for the internal-use Channel. So:
383 * m_master_channel->expect_msg(..., [...](Msg_in_ptr m) { post([..., m]() { handle in-message m; }); });
384 * But m_master_channel is private in super-class, and that super-class instead gave us this nice
385 * protected utility to use: */
386 master_channel->expect_msg(Master_structured_channel::Msg_which_in::JEMALLOC_SHM_SETUP,
387 [this, master_channel]
388 (typename Master_structured_channel::Msg_in_ptr&& msg) mutable
389 {
390 // We are in thread Wc (unspecified, really struc::Channel async callback thread).
391 Base::Base::async_worker()->post([this, master_channel, msg = std::move(msg)]() mutable
392 {
393 // We are in thread W.
394 if (Base::Base::Base::hosed())
395 {
396 // Must have emitted error somehow elsewhere. async_connect() will fail.
397 return;
398 }
399 // else
400
401 // That's the entire payload of the in-message.
402 auto local_hndl_or_null = msg->native_handle_or_null();
403
404 if (local_hndl_or_null.null())
405 {
406 FLOW_LOG_WARNING("Client session [" << *this << "]: Session-connect request: Vanilla async-connect"
407 "succeeded, but opposing server failed to create the resources (pre-connected native handle "
408 "pair for local stream socket connection) necessary for arena-lending SHM provider's "
409 "internal IPC needs. Will go back to NULL state and report to user via "
410 "on-async-connect handler.");
411 Base::Base::complete_async_connect_after_canceling_peer_state
412 (session::error::Code::S_SESSION_OPEN_CHANNEL_SERVER_CANNOT_PROCEED_RESOURCE_UNAVAILABLE);
413 return;
414 }
415 /* else: We can make Channel, and upgrade to struc::Channel, with no further potential for errors.
416 * Our super-class (shm::...::Session_mv) can handle those details and everything else including
417 * setting up the Arena. That stuff is the same on the server and client, except server also sets
418 * up an extra arena; but we don't have any extra steps. So that's it!
419 *
420 * Well, one last thing actually: The cleanup algorithm started/contained in our ctor depends on being able
421 * to parse-out the PID of the creating process from the pool name as obtained by
422 * doing Pool_arena::for_each_persistent() which trolls /dev/shm (in Linux at least). Hence we shall place
423 * our PID at the expect place in the name via the mechanism provided for such things by init_shm(). */
424
425 FLOW_LOG_INFO("Client session [" << *this << "]: Session-connect request: Vanilla async-connect succeeded, "
426 "and opposing server sent a resource (pre-connected native handle for "
427 "local stream socket connection) necessary for arena-lending SHM provider's internal "
428 "IPC needs. Will send ack and complete local SHM setup synchronously.");
429
430 // Don't forget to ack it.
431 Error_code err_code;
432 if (!master_channel->send(master_channel->create_msg(), msg.get(), &err_code))
433 {
434 if (err_code)
435 {
436 // The send() detected new master_channel hosing.
437 Base::Base::complete_async_connect_after_canceling_peer_state(err_code);
438 }
439 // else: Must have emitted error somehow elsewhere. async_connect() will fail.
440 return;
441 }
442 // else
443
444 /* So just init_shm() left.
445 * It might have failed; if so, then that's that (nothing else to undo); or conversely, if it succeeds,
446 * on this side there's nothing further to do either. So just report the result of init_shm(). */
447 err_code = Base::init_shm(std::move(local_hndl_or_null),
448 Shared_name::ct(to_string(util::Process_credentials::own_process_id())));
449
450 if (!err_code)
451 {
452 /* I lied (Arnie voice from Commando). We also need to do this for each arena (we have just the one).
453 * Need this for arena->construct<T>(), where T is allocator-compliant thing such as STL containers,
454 * to work. */
455#ifndef NDEBUG
456 const bool ok =
457#endif
458 Base::session_shm()->add_shm_pool_listener(&m_shm_arena_pool_listener);
459 assert(ok && "Maintenance bug? It should only fail if already registered.");
460 }
461
462 Base::Base::complete_async_connect_after_canceling_peer_state(err_code);
463 }); // async_worker()->post()
464 }); // master_channel->expect_msg()
465
466 // if (returned false) { Must have emitted error somehow elsewhere. async_connect() will fail. }
467 }); // return async_connect()
468} // Client_session_impl::async_connect()

470TEMPLATE_JEM_CLI_SESSION_IMPL
471void CLASS_JEM_CLI_SESSION_IMPL::cleanup()
472{
473 // `classic`? Pool_arena?! But this is jemalloc!?!?! Worry not. See remove_each_persistent_if() doc header.
474 using ipc::shm::classic::Pool_arena;
475 using util::String_view;
476 using util::process_id_t;
477 using boost::chrono::seconds;
478 using boost::lexical_cast;
479
480 constexpr util::Fine_duration CLEANUP_PERIOD = seconds(30);
481
482 FLOW_LOG_TRACE("Client session [" << *this << "]: Periodic (or initial) cleanup starting.");
483
484 /* This is just like Session_server::cleanup(), so see that guy for discussion first. The differences here:
485 * - Like Session_server, we decode the pool creator's PID from a given pool's name; but in our case the
486 * creator is a client, so the srv_namespace() -- the server process's PID -- is not the right thing.
487 * That's why we added it specifically into the name in async_connect() when invoking Base::init_shm().
488 * So now we just parse it right out of there instead of the standard srv_namespace component.
489 * - Not a really-real difference but good to note: In practice (in production anyway) there's only one
490 * Session_server per process ever, so probably one Session_server's async cleanup algorithm won't
491 * be running concurrently with another. There can definitely be multiple `Client_session`s though, albeit
492 * (in production anyway) not usually concurrently (but even then it's quite conceivable one Client_session
493 * is hosed but not destroyed as a C++ object; while another has been created; it's not recommended but should
494 * be harmless). So there's a higher potential, arguably, for our async cleanup()-started algorithm
495 * concurrently running with another for the same Client_app. It is fine however: again, the whole point of it
496 * is to only touch pools created by PIDs that are not around, not even in zombie form; and if one is indeed
497 * identified, then the worst thing that can possibly happen is the remove_persistent() (unlink) attempt
498 * will fail due to being unnecessary -- this is perfectly fine (we even suppress logs/errors, IIRC).
499 * Anyway: Noting here FYI.
500 *
501 * That said proceed in the (almost) same way as Session_server. Keeping comments light. */
502
503 const auto n_removed
504 = util::remove_each_persistent_if<Pool_arena>(get_logger(),
505 [&](const Shared_name& shm_pool_name)
506 {
507 Shared_name resource_type;
508 Shared_name cli_app_name;
509 Shared_name the_rest;
510
511 if (!(decompose_conventional_shared_name(shm_pool_name,
512 &resource_type,
513 nullptr, // srv_app_name: Needs to be there, but we don't care what it is.
514 nullptr, // srv_namespace: Ditto.
515 &cli_app_name,
516 nullptr, // cli_namespace_or_sentinel: Ditto.
517 &the_rest)
518 && (resource_type == Shared_name::S_RESOURCE_TYPE_ID_SHM)
519 && (cli_app_name == Base::Base::Base::cli_app_ptr()->m_name)
520 && String_view(the_rest.str()).starts_with((SHM_SUBTYPE_PREFIX + Shared_name::S_SEPARATOR).str())))
521 {
522 // Not our pool to possibly delete. Misnamed; or not by this Client_app; or not from this SHM-provider.
523 return false;
524 }
525 // else
526
527 /* We must parse out the client-PID (as noted in big comment above).
528 * the_rest should look like: SHM_SUBTYPE_PREFIX / -->client-PID<-- / more-stuff.
529 * Hence make past_prefix point to: ---------------^ */
530 String_view past_prefix(the_rest.str());
531 past_prefix.remove_prefix(SHM_SUBTYPE_PREFIX.str().size() + 1);
532 // Now cut off from here -------------------------------------------^.
533 const size_t sep_pos = past_prefix.find_first_of(Shared_name::S_SEPARATOR);
534 if (sep_pos == String_view::npos)
535 {
536 // Created by server; not us.
537 return false;
538 }
539 // else
540 past_prefix.remove_suffix(past_prefix.size() - sep_pos);
541
542 process_id_t pid = 0;
543 try
544 {
545 pid = lexical_cast<process_id_t>(past_prefix);
546 }
547 catch (...) {}
548
549 if (pid == 0)
550 {
551 FLOW_LOG_WARNING("Client session [" << *this << "]: During periodic cleanup found seemingly relevant "
552 "(based on prefix/form) SHM-pool named [" << shm_pool_name << "]; but client-PID term "
553 "[" << past_prefix << "] is not a valid-looking PID. Ignoring; but this warrants "
554 "investigation.");
555 return false;
556 }
557 // else
558
559 if (util::process_running(pid))
560 {
561 return false;
562 }
563 // else
564
565 FLOW_LOG_INFO("Client session [" << *this << "]: During periodic cleanup found SHM-pool "
566 "named [" << shm_pool_name << "]; appears to have been created by a process of this application "
567 "that is no longer running, even as a zombie; perhaps it exited without graceful cleanup. Will "
568 "attempt to remove it. Competing cleanup(s) may be occurring concurrently; that is fine.");
569 return true;
570 }); // remove_each_persistent_if()
571 if (n_removed == 0)
572 {
573 FLOW_LOG_TRACE("Client session [" << *this << "]: Cleanup finished: remove none.");
574 }
575 else
576 {
577 FLOW_LOG_INFO("Client session [" << *this << "]: Cleanup finished: Removed [" << n_removed << "] pools "
578 "successfully.");
579 }
580
581 m_async_cleanup_worker.schedule_from_now(CLEANUP_PERIOD, [this](auto&&) { cleanup(); });
582} // Client_session_impl::cleanup()

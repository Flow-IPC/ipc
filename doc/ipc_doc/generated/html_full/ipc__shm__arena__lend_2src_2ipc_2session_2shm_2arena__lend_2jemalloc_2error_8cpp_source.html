100std::string Category::message(int val) const // Virtual.
101{
102 using std::string;
103
104 // KEEP THESE STRINGS IN SYNC WITH COMMENT IN error.hpp ON THE INDIVIDUAL ENUM MEMBERS!
105
106 // See notes in transport::error in same spot.
107 switch (static_cast<Code>(val))
108 {
109 case Code::S_SHM_ARENA_CREATION_FAILED:
110 return "Session opening: While setting up the session's locally-managed SHM-arena(s) of an arena-lending "
111 "SHM-provider, an error occurred thus hosing the session before it could be opened. Logs may "
112 "indicate the reason.";
113 case Code::S_SHM_ARENA_LEND_FAILED:
114 return "Session opening: While registering the session's locally-managed SHM-arena(s) of an arena-lending "
115 "SHM-provider with the lend/borrow engine, an error occurred thus hosing the session before it could "
116 "be opened. Logs may indicate the reason. Most likely it was an internal-IPC failure when sending "
117 "arena info.";
118 case Code::S_SERVER_MASTER_SHM_UNEXPECTED_TRANSPORT_ERROR:
119 return "Session opening: While trying to transmit resource for internal-IPC use in an arena-lending SHM-provider, "
120 "server encountered incoming-direction channel error whose exact nature could not be determined at that "
121 "exact moment. Logs will indicate that exact nature; meanwhile session opening failed.";
122
123 case Code::S_END_SENTINEL:
124 assert(false && "SENTINEL: Not an error. "
125 "This Code must never be issued by an error/success-emitting API; I/O use only.");
126 }
127 assert(false);
128 return "";
129} // Category::message()

131util::String_view Category::code_symbol(Code code) // Static.
132{
133 // Note: Must satisfy istream_to_enum() requirements.
134
135 switch (code)
136 {
137 case Code::S_SHM_ARENA_CREATION_FAILED:
138 return "SHM_ARENA_CREATION_FAILED";
139 case Code::S_SHM_ARENA_LEND_FAILED:
140 return "SHM_ARENA_LEND_FAILED";
141 case Code::S_SERVER_MASTER_SHM_UNEXPECTED_TRANSPORT_ERROR:
142 return "SERVER_MASTER_SHM_UNEXPECTED_TRANSPORT_ERROR";
143
144 case Code::S_END_SENTINEL:
145 return "END_SENTINEL";
146 }
147 assert(false);
148 return "";
149}

157std::istream& operator>>(std::istream& is, Code& val)
158{
159 /* Range [<1st Code>, END_SENTINEL); no match => END_SENTINEL;
160 * allow for number instead of ostream<< string; case-insensitive. */
161 val = flow::util::istream_to_enum(&is, Code::S_END_SENTINEL, Code::S_END_SENTINEL, true, false,
162 Code(S_CODE_LOWEST_INT_VALUE));
163 return is;
164}

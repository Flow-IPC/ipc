109Native_socket_stream Native_socket_stream::Impl::release()
110{
111 using util::sync_io::Asio_waitable_native_handle;
112 using util::String_view;
113 using asio_local_stream_socket::Peer_socket;
114 using flow::log::Log_context;
115
116 // Keep in sync with NULL-state ctor!
117
118 Error_code sys_err_code;
119 const auto abort_on_fail = [&](String_view context) // Little helper.
120 {
121 if (sys_err_code)
122 {
123 FLOW_LOG_FATAL("Socket stream [" << *this << "]: "
124 "Socket op [" << context << "] failed; this should never happen. Details follow. Look into it.");
125 FLOW_ERROR_SYS_ERROR_LOG_FATAL();
126 assert(false && "Socket op failed; this should never happen. Look into it.");
127 std::abort();
128 }
129 };
130
131 /* Our mission is to (1) create a fresh guy like *this managing the same Native_handle; and (2) make us
132 * as-if we were just cted in NULL state (with null Logger and empty nickname).
133 *
134 * (1) is the easy (and important) part: we just need to invoke our ctor feeding it the raw Native_handle.
135 * (2) is the hard/brittle (and not super-important, but hey, we did promise to make it work a certain way) part:
136 * we need to act like the NULL-state ctor path on existing *this.
137 *
138 * It does help that by contract we have to be in nothing-has-happened-since-entering-PEER-state state. So
139 * our expected state is pretty rigid; so it is reasonably easy to reason about it.
140 *
141 * Maybe it's self-evident but to be clear -- things like m_nb_task_engine, m_ev_hndl_task_engine_unused,
142 * m_snd_auto_ping_timer, m_rcv_idle_timer can simply be left alone; nothing wrong with them.
143 * The timers haven't been used (by contract, no auto_ping() or idle_timer_run()); the `Task_engine`s
144 * are dissociated/re-associated with their asio objects' FDs in civilized fashion via .release() + reassignment. */
145
146 FLOW_LOG_TRACE("Socket stream [" << *this << "]: Releasing idle-state object to new socket-stream core object.");
147
148 assert((m_state == State::S_PEER) && "By contract must be in PEER state."); // This is just important.
149 assert(m_peer_socket && "By contract must be not be in hosed state."); // Ditto.
150 // These following ones mean we won't have to reset them to be as-if-default-cted.
151 assert(m_snd_pending_payloads_q.empty() && "By contract must be in idle state.");
152 assert((!m_snd_pending_err_code) && "By contract must be in idle state.");
153 assert((!m_snd_finished) && "By contract must be in idle state.");
154 assert(m_snd_pending_on_last_send_done_func_or_empty.empty() && "By contract must be in idle state.");
155 assert(m_snd_auto_ping_period == util::Fine_duration::zero());
156 assert((!m_rcv_user_request) && "By contract must be in idle state.");
157 assert((!m_rcv_pending_err_code) && "By contract must not be in a hosed state.");
158 assert(m_rcv_idle_timeout == util::Fine_duration::zero());
159
160 // As-if-default-cted means these will all be empty.
161 m_conn_ev_wait_func.clear();
162 m_snd_ev_wait_func.clear();
163 m_rcv_ev_wait_func.clear();
164
165 // And this guy.
166 m_state = State::S_NULL;
167
168 // We'll need this key FD; get it out of m_peer_socket in civilized fashion; it becomes FD-less (!.is_open()).
169 auto peer_socket_raw_hndl = m_peer_socket->release(sys_err_code);
170 abort_on_fail("release");
171
172 /* Now we're ready to set up m_*peer_socket to as-if-default-cted state.
173 * Follow what ctor does. Note that, indeed, we generate a new FD. */
174 assert((!m_peer_socket->is_open()) && "We ->release()d it above.");
175 m_peer_socket->open(asio_local_stream_socket::local_ns::stream_protocol(),
176 sys_err_code);
177 abort_on_fail("open");
178
179 m_ev_wait_hndl_peer_socket.assign(Native_handle(m_peer_socket->native_handle()));
180
181 // Logging done; can blow up Log_context super-object to as-if-default-cted state.
182 const auto logger_ptr = get_logger();
183 *(static_cast<Log_context*>(this)) = Log_context(nullptr, Log_component::S_TRANSPORT);
184 // This no longer matters for logging; might as well empty it here. Obv save it for the new guy.
185 auto nickname_str = std::move(m_nickname);
186 m_nickname.clear(); // Just in case move() didn't do it.
187
188 // The easy+important part is to finally make a fresh core like ex-*this.
189 return Native_socket_stream(logger_ptr, nickname_str, Native_handle(peer_socket_raw_hndl));
190} // Native_socket_stream::Impl::release()

202 if ((!m_conn_ev_wait_func.empty()) || (!m_snd_ev_wait_func.empty()) || (!m_rcv_ev_wait_func.empty()))

205 "a start-*-ops procedure has been executed. Ignoring.");
206 return false;

208 // else
209
210 FLOW_LOG_INFO("Socket stream [" << *this << "]: Replacing event-wait handles (probably to replace underlying "
211 "execution context without outside event loop's boost.asio Task_engine or similar).");
212
213 assert(m_ev_wait_hndl_peer_socket.is_open());
214 assert(m_snd_ev_wait_hndl_auto_ping_timer_fired_peer.is_open());
215 assert(m_rcv_ev_wait_hndl_idle_timer_fired_peer.is_open());
216

218 m_ev_wait_hndl_peer_socket = create_ev_wait_hndl_func();
219 m_ev_wait_hndl_peer_socket.assign(saved);
220

223 m_snd_ev_wait_hndl_auto_ping_timer_fired_peer.assign(saved);
224

226 m_rcv_ev_wait_hndl_idle_timer_fired_peer = create_ev_wait_hndl_func();
227 m_rcv_ev_wait_hndl_idle_timer_fired_peer.assign(saved);
228
229 return true;
230} // Native_socket_stream::Impl::replace_event_wait_handles()

237bool Native_socket_stream::Impl::async_connect(const Shared_name& absolute_name, Error_code* sync_err_code_ptr,
238 flow::async::Task_asio_err&& on_done_func)
239{
240 namespace bind_ns = flow::util::bind_ns;
241 using asio_local_stream_socket::Endpoint;
242 using asio_local_stream_socket::Peer_socket;
243 using asio_local_stream_socket::endpoint_at_shared_name;
244 using flow::util::ostream_op_string;
245 using flow::async::Task_asio_err;
246 using util::Task;
247
248 if (m_state != State::S_NULL)
249 {
250 FLOW_LOG_WARNING("Socket stream [" << *this << "]: Wanted to connect to [" << absolute_name << "] "
251 "but already not in NULL state. Ignoring.");
252 return false;
253 }
254 // else
255 if (!op_started<Op::S_CONN>("async_connect"))
256 {
257 return false;
258 }
259 // else
260
261 m_state = State::S_CONNECTING;
262
263 FLOW_LOG_INFO("Socket stream [" << *this << "]: Will attempt connect to [" << absolute_name << "].");
264
265 Error_code sync_err_code;
266 const auto remote_endpoint = endpoint_at_shared_name(get_logger(), absolute_name, &sync_err_code);
267 assert((remote_endpoint == Endpoint()) == bool(sync_err_code)); // (By the way it WARNs on error.)
268
269 if (!sync_err_code)
270 {
271 // Endpoint was fine; do the actual async connect attempt.
272
273 /* Normally we'd do m_peer_socket->async_connect(F), but in the sync_io pattern we have to break it down into
274 * some non-blocking operation(s) with an async-wait delegated to the user via m_conn_ev_wait_func().
275 * (Internally that's what ->async_connect() does, but the wait is done by boost.asio.)
276 * Namely that's:
277 * -# Set socket to non-blocking mode: ->non_blocking(true).
278 * -# Synchronous ->connect().
279 * -# This might succeed -- or fail -- immediately, as it's a local socket. Done. Otherwise:
280 * -# It will yield would-block indicating connection in-progress.
281 * -# Wait for writability of socket.
282 * -# Once that's ready, done. (If it really failed, it'll be "writable" -- but as soon as they try using it,
283 * the true error shall be revealed.)
284 *
285 * We will do just that, for maximum resiliency in the face of who-knows-what. In reality (as tested in Linux)
286 * the situation is simultaneously simpler -- in practice -- and more complicated (the reasons for why it works
287 * that way). I explain... it's pretty messy. This is with boost.asio from Boost 1.81.
288 *
289 * The trickiness is in ->connect(). If the opposing acceptor is listen()ing, and there is sufficient backlog
290 * space, then it just immediately succeeds. If it is not listen()ing, then it just immediately fails.
291 * Now suppose it is listen()ing, but it ran out of backlog space. Then: Whether ->non_blocking(true) or not
292 * (but, like, yes, it's true for us), it internally does the following.
293 * -# ::connect(). If returns error.
294 * -# If that error is would-block (EWOULDBLOCK or EAGAIN or EINPROGRESS; but really-really EAGAIN is what happens
295 * in Linux with Unix domain sockets):
296 * -# ::poll(), awaiting (with infinite timeout) writability of socket.
297 * - Yes, it does this totally ignoring ->non_blocking(). That reads like a bug, and no comments explain
298 * it, but in practice it apparently is not a bug; at least it's not for Unix domain sockets in my
299 * (ygoldfel) testing.
300 * -# The ::poll() immediately returns 1 active FD, meaning it's writable.
301 * - I have no idea why, exactly. In point of fact in blocking mode the ::connect() will actually sit there
302 * and wait for the backlog to clear. It does make sense non-blocking ::connect() immediately yields
303 * EAGAIN. But then ::poll() reports immediate writability? That is rather strange. Why wouldn't
304 * the connect() just return an error then?
305 * -# The code then tries to get SO_ERROR through a ::getsockopt(). This might be a TCP-stack thing more;
306 * in any case it yields no error in practice with Unix domain socket. (I really traced through the code to
307 * confirm this; socket_ops.ipp is the file.)
308 * So, ostensibly the ->connect() succeeds in this situation. However, ->write_some() at that point just
309 * hilariously yields ENOTCONN. So all is well that ends... poorly? Point is, that's what happens; at least
310 * it doesn't just sit around blocking inside ->connect() despite non_blocking(true). Is it okay behavior
311 * given the situation (backlog full)? Firsly backlog really shouldn't be full given a properly
312 * operating opposing acceptor (by the way it default to like 4096). But if it is, I'd say this behavior is
313 * fine, and my standard is this: Even using a nice, civilized, normal m_peer_socket->async_connect(), the
314 * resulting behavior is *exactly* the same: The ->async_connect() quickly "succeeds"; then
315 * ->async_write() immediately ENOTCONNs. If we do equally well as a properly operated ->async_connect(),
316 * then what else can they ask for really?
317 *
318 * That said, again, out of a preponderance of caution and future-proofness (?) we don't rely on the above
319 * always happening; and in fact have a clause for getting would_block from ->connect() in which case
320 * we do the whole m_conn_ev_wait_func() thing. For now it won't be exercised is all. */
321
322 assert((!m_peer_socket->non_blocking()) && "New NULL-state socket should start as not non-blocking.");
323 m_peer_socket->non_blocking(true, sync_err_code);
324 if (!sync_err_code)
325 {
326 m_peer_socket->connect(remote_endpoint, sync_err_code);
327 if (sync_err_code == boost::asio::error::would_block)
328 {
329 FLOW_LOG_INFO("Socket stream [" << *this << "]: boost::asio::connect() (non-blocking) got would-block; "
330 "while documented as possible this is actually rather surprising based on our testing "
331 "and understanding of internal boost.asio code. Proceeding to wait for writability.");
332 m_conn_ev_wait_func(&m_ev_wait_hndl_peer_socket,
333 true, // Wait for write.
334 // Once writable do this.
335 boost::make_shared<Task>
336 ([this, on_done_func = std::move(on_done_func)]() mutable
337 {
338 conn_on_ev_peer_socket_writable(std::move(on_done_func));
339 }));
340
341 sync_err_code = error::Code::S_SYNC_IO_WOULD_BLOCK;
342 }
343 else if (sync_err_code)
344 {
345 FLOW_LOG_WARNING("Socket stream [" << *this << "]: boost::asio::connect() (non-blocking) completed "
346 "immediately but with error [" << sync_err_code << "] [" << sync_err_code.message() << "]. "
347 "Connect request failing. Will emit error via sync-args.");
348 }
349 // else { Success. Reminder: this and fatal error are both much likelier than would-block. }
350 } // if (!err_code) [m_peer_socket->non_blocking(true)] (but it may have become truthy inside)
351 else // if (err_code) [m_peer_socket->non_blocking(true)]
352 {
353 FLOW_LOG_WARNING("Socket stream [" << *this << "]: Trying to set non-blocking mode yielded error, "
354 "which is pretty crazy; details: [" << sync_err_code << "] "
355 "[" << sync_err_code.message() << "]. Connect request failing. "
356 "Will emit error via sync-args.");
357 }
358 } // if (!sync_err_code) [endpoint construction] (but it may have become truthy inside)
359 // else if (sync_err_code) [endpoint construction] { It logged. }
360
361 // If got here, sync_err_code indicates immediate success or failure of async_connect().
362 m_state = sync_err_code ? State::S_NULL : State::S_PEER;
363
364 // Standard error-reporting semantics.
365 if ((!sync_err_code_ptr) && sync_err_code)
366 {
367 throw flow::error::Runtime_error(sync_err_code, "Native_socket_stream::Impl::async_connect()");
368 }
369 // else
370 sync_err_code_ptr && (*sync_err_code_ptr = sync_err_code);
371 // And if (!sync_err_code_ptr) + no error => no throw.
372
373 return true;
374} // Native_socket_stream::Impl::async_connect()

376void Native_socket_stream::Impl::conn_on_ev_peer_socket_writable(flow::async::Task_asio_err&& on_done_func)
377{
378 assert((m_state == State::S_CONNECTING) && "Only we can get out of CONNECTING state in the first place.");
379
380 /* The wait indicates it's writable... or "writable," meaning in error state. While we could try some
381 * trick like ::getsockopt(SO_ERROR), it's all academic anyway: in Linux at least, as noted in long
382 * comment in async_connect(), ->connect() will either succeed or fail right away; and in the one
383 * case where one can force a connectable-but-not-immediately situation -- hitting a backlog-full acceptor --
384 * it'll still just succeed. Anyway, if it weren't academic, then even if we falsely assume PEER state here,
385 * when really m_peer_socket is hosed underneath, it'll just get exposed the moment we try to read or write.
386 * So just relax. */
387
388 FLOW_LOG_INFO("Socket stream [" << *this << "]: Writable-wait upon would-blocked connect attempt: done. "
389 "Entering PEER state. Will emit to completion handler.");
390 m_state = State::S_PEER;
391 on_done_func(Error_code());
392 FLOW_LOG_TRACE("Handler completed.");
393} // Native_socket_stream::Impl::conn_on_ev_peer_socket_writable()

32/**
33 * Common data and logic for shm::classic::Server_session_impl and shm::classic::Client_session_impl.
34 * SHM-classic is such that the two sides are exactly the same, once the two shm::classic::Pool_arena handles
35 * have been set up. So that stuff that is all the same is here in this internally-used super-class of the 2
36 * `*_impl`. The latter add the initial assignment of the two arena handles and, relatedly, the
37 * client/server-specific APIs (e.g., Client_session_impl::sync_connect()).
38 *
39 * @tparam Session_impl_t
40 * Our base that our sub-class wants to build on-top-of. E.g., `Server_session_impl<...>`.
41 */

276TEMPLATE_CLSC_SESSION_IMPL
277template<typename T>
278typename CLASS_CLSC_SESSION_IMPL::Blob
279 CLASS_CLSC_SESSION_IMPL::lend_object(const typename Arena::template Handle<T>& handle)
280{
281 using util::Blob_const;
282 using flow::util::buffers_dump_string;
283
284 // This may look somewhat cheesy, but it's quite quick with only a couple scopes to check, so why not?
285 scope_id_t scope_id;
286 Blob serialization;
287
288 if (session_shm()->is_handle_in_arena(handle))
289 {
290 serialization = session_shm()->lend_object(handle);
291 scope_id = S_SCOPE_ID_SESSION;
292 }
293 else // (Do not use `else if (is_handle_in_arena())` to avoid maybe-uninitialized warning in some compilers.)
294 {
295 assert(app_shm()->is_handle_in_arena(handle)
296 && "lend_object() called on invalid value? Or bug -- forgot to update this method?");
297 serialization = app_shm()->lend_object(handle);
298 scope_id = S_SCOPE_ID_APP;
299 }
300
301 /* Encode which Arena it came from, so borrow_object() can decode this on the other side. Without this it's
302 * quite impossible for it to know. Even if one understands what's inside `serialization` (spoiler alert -- currently
303 * Pool_arena places there offset from pool base addr), its meaning is only meaningful when knowing
304 * the Arena. */
305 Blob real_serialization(serialization.size() + sizeof(scope_id));
306 real_serialization.emplace_copy(real_serialization.begin(), serialization.const_buffer());
307 real_serialization.emplace_copy(real_serialization.begin() + serialization.size(),
308 Blob_const(&scope_id, sizeof(scope_id)));
309
310 FLOW_LOG_TRACE("Session [" << *this << "]: SHM-classic-lend serialization: "
311 "[" << buffers_dump_string(real_serialization.const_buffer(), " ") << "].");
312
313 return real_serialization;
314} // Session_impl::lend_object()

316TEMPLATE_CLSC_SESSION_IMPL
317template<typename T>
318typename CLASS_CLSC_SESSION_IMPL::Arena::template Handle<T>
319 CLASS_CLSC_SESSION_IMPL::borrow_object(const Blob& serialization)
320{
321 using util::Blob_const;
322 using flow::util::buffers_dump_string;
323 using Value = T;
324
325 // This should be possible to understand if one has grokked borrow_object().
326
327 FLOW_LOG_TRACE("Session [" << *this << "]: SHM-classic-borrow serialization: "
328 "[" << buffers_dump_string(serialization.const_buffer(), " ") << "].");
329
330 Blob real_serialization(serialization); // Copy (it's small).
331 const auto real_serialization_sz = real_serialization.size() - sizeof(scope_id_t);
332 real_serialization.start_past_prefix_inc(real_serialization_sz);
333 // Copy it out of there (that should realign it at the target stack location).
334 const auto scope_id = *(reinterpret_cast<const scope_id_t*>(real_serialization.const_data()));
335 real_serialization.start_past_prefix_inc(-real_serialization_sz);
336 real_serialization.resize(real_serialization_sz);
337 // real_serialization is now the original Arena::lend_object() result.
338
339 if (scope_id == S_SCOPE_ID_SESSION)
340 {
341 return session_shm()->template borrow_object<Value>(real_serialization);
342 }
343 // else
344 if (scope_id == S_SCOPE_ID_APP)
345 {
346 return app_shm()->template borrow_object<Value>(real_serialization);
347 }
348 // else
349
350 assert(false && "borrow_object() called on invalid value? Or bug -- forgot to update this method? IPC fail?");
351 return nullptr;
352} // Session_impl::borrow_object()

177TEMPLATE_JEM_SRV_SESSION_IMPL
178CLASS_JEM_SRV_SESSION_IMPL::~Server_session_impl()
179{
180 /* Since we are subclass of Base::Base (session::Server_session_impl), and we do post our own tasks onto thread W
181 * (otherwise maintained by Base::Base), this protected guy's doc header says we must call it. So formally speaking
182 * that's why we do it. More in English, though: if a subclass of session::Server_session_impl (us, or perhaps
183 * Base) has queued a task F on thread W (async_worker()), and this dtor is called, and F() touches some part of
184 * *this or (Base&)(*this), then there can be undefined-behavior and/or concurrent-access trouble. Once
185 * Base::Base::~Base() is reached, we're safe, as it stop()s thread W in civilized fashion first-thing, synchronously,
186 * before letting its own destruction continue. So this causes the stop()page to occur somewhat earlier, so that
187 * we too can feel peace: */
188 Base::Base::dtor_async_worker_stop();
189 // Thread W has been joined.
190}

193template<typename Session_server_impl_t,
194 typename Task_err, typename Cli_app_lookup_func, typename Cli_namespace_func, typename Pre_rsp_setup_func,
195 typename N_init_channels_by_srv_req_func, typename Mdt_load_func>
196void CLASS_JEM_SRV_SESSION_IMPL::async_accept_log_in
197 (Session_server_impl_t* srv,
198 typename Base::Base::Channels* init_channels_by_srv_req,
199 typename Base::Base::Mdt_reader_ptr* mdt_from_cli_or_null,
200 typename Base::Base::Channels* init_channels_by_cli_req,
201 Cli_app_lookup_func&& cli_app_lookup_func,
202 Cli_namespace_func&& cli_namespace_func, Pre_rsp_setup_func&& pre_rsp_setup_func,
203 N_init_channels_by_srv_req_func&& n_init_channels_by_srv_req_func, Mdt_load_func&& mdt_load_func,
204 Task_err&& on_done_func)
205{
206 using util::Native_handle;
207 namespace asio_local = transport::asio_local_stream_socket::local_ns;
208 using asio_local::connect_pair;
209 using transport::asio_local_stream_socket::Peer_socket;
210 using boost::movelib::make_unique;
211
212 // The extra stuff to do on top of the base vanilla Server_session_impl.
213 auto real_pre_rsp_setup_func
214 = [this,
215 // Get the Session_server<> such that its core comprises the arg `srv`.
216 srv = srv->this_session_srv(),
217 pre_rsp_setup_func = std::move(pre_rsp_setup_func)]
218 () -> Error_code
219 {
220 // We are in thread W.
221
222 auto err_code = pre_rsp_setup_func();
223 if (err_code)
224 {
225 // Any Session_server-given setup failed => no point in doing our SHM-classic per-session setup.
226 return err_code;
227 }
228 // else
229
230 /* Do our extra stuff on top of the base vanilla Server_session_impl. That is Base::session_shm() must
231 * work (return non-null), as must app_shm() (ditto, cached in this->m_app_shm); plus lend_object()ing
232 * objects constructed from either must work, and borrow_object()ing from anything the opposing process
233 * transmits must work. The lend/borrow stuff will work only if session_shm() and app_shm() have
234 * been registered with the lend/borrow engine, namely shm_session().
235 * Incidentally Base::shm_session() must also be available (return non-null), in case the user wants to
236 * access that guy directly.
237 *
238 * m_app_shm first: please see our class doc header; then come back here. To summarize:
239 * - pre_rsp_setup_func() had to have created what we want app_shm() to return.
240 * - If it already existed by then (Client_app seen already), even better.
241 * - If that creation failed, then pre_rsp_setup_func() just failed, so we are not here.
242 * - Parent session server's app_shm(<the Client_app>) = what we want. */
243 m_app_shm = srv->app_shm_ptr(*(Base::Base::Base::cli_app_ptr()));
244 assert(m_app_shm && "How can it be null, if pre_rsp_setup_func() returned success? Contract broken internally?");
245
246 /* Now session_shm(), shm_session(), and registering of the former (and of app_shm()) into shm_session().
247 * Base::init_shm() takes care of all of it. However it's not time
248 * for that yet: the log-in is not yet complete. We can't do other master_channel() traffic yet (and we need
249 * to for init_shm() purposes). See below for the post-vanilla-async_accept_log_in() handler steps. */
250
251 return err_code; // == Error_code().
252 }; // auto real_pre_rsp_setup_func =
253
254 Base::Base::async_accept_log_in(srv,
255 init_channels_by_srv_req, mdt_from_cli_or_null, init_channels_by_cli_req,
256 std::move(cli_app_lookup_func),
257 std::move(cli_namespace_func),
258 std::move(real_pre_rsp_setup_func),
259 std::move(n_init_channels_by_srv_req_func),
260 std::move(mdt_load_func),
261
262 [this, srv = srv->this_session_srv(), on_done_func = std::move(on_done_func)]
263 (const Error_code& async_err_code)
264 {
265 // We are in thread W (from Base).
266
267 if (async_err_code == session::error::Code::S_OBJECT_SHUTDOWN_ABORTED_COMPLETION_HANDLER)
268 {
269 return; // Stuff is shutting down. GTFO.
270 }
271 // else
272
273 auto err_code = async_err_code; // We need to modify it below.
274
275 if (!err_code)
276 {
277 // Now resume from inside real_pre_rsp_setup_func(). init_shm() needs our side of the internal-use channel.
278
279 FLOW_LOG_INFO("Server session [" << *this << "]: Vanilla accept-log-in succeeded; now we will send "
280 "a resource to opposing session-client to enable an internal-IPC channel for arena-lending "
281 "SHM-provider; and set up local SHM objects synchronously. If it all succeeds we will "
282 "complete the accept-log-in.");
283
284 Peer_socket local_hndl_asio(*(Base::async_worker()->task_engine()));
285 Peer_socket remote_hndl_asio(std::move(local_hndl_asio));
286 Error_code sys_err_code;
287 connect_pair(local_hndl_asio, remote_hndl_asio, sys_err_code);
288 if (sys_err_code)
289 {
290 FLOW_LOG_WARNING("Server session [" << *this << "]: While setting up SHM internal-use channel: "
291 "connect_pair() failed. Details follow.");
292 FLOW_ERROR_SYS_ERROR_LOG_WARNING(); // Log based on sys_err_code.
293 err_code = sys_err_code;
294 }
295 else // if (!sys_err_code)
296 {
297 auto local_hndl = Native_handle(local_hndl_asio.release());
298 auto remote_hndl = Native_handle(remote_hndl_asio.release());
299
300 /* Before we do our local stuff we should enable the other side to do the complementary thing:
301 * send them (with return ack) `remote_hndl`. */
302 auto msg = Base::master_channel()->create_msg();
303 msg.body_root()->initJemallocShmSetup();
304 msg.store_native_handle_or_null(std::move(remote_hndl));
305 // We don't care about contents of the ack (it's just an ack).
306 if ((!Base::master_channel()->sync_request(msg, nullptr, &err_code)) // @todo Consider using finite timeout.
307 && (!err_code))
308 {
309 /* Annoying corner case. Incoming-direction error occurred on master_channel() before we could
310 * sync_request(); it'll be reported through its on-error handler. We really don't want to enter
311 * async-land and wait for that, though: we're technically in almost-PEER state already and merely
312 * haven't informed user of that via their on_done_func(). Let's just report a catch-all thing;
313 * in any case the true error is logged when detected earlier inside struc::Channel. */
314 err_code = error::Code::S_SERVER_MASTER_SHM_UNEXPECTED_TRANSPORT_ERROR;
315 }
316 /* else if ((!sync_request()) && err_code) { New error detected via sync_request(). }
317 * else { All good so far. Ready for init_shm()! } */
318
319 if (!err_code)
320 {
321 err_code = Base::init_shm(std::move(local_hndl), Shared_name(), app_shm_ptr());
322 /* If that returned falsy: Fantastic. To quote our mission from earlier:
323 * "Now session_shm(), shm_session(), and registering of the former (and of app_shm()) into shm_session().
324 * Base::init_shm() takes care of all of it." It INFO-logged too. */
325
326 if (!err_code)
327 {
328 /* One last thing. We also need to do this for each arena (we have the two; but Session_server takes care
329 * of it for app_shm() upon creating it; which -- reminder -- occurs only the first time a session for
330 * that Client_app appears; we may be the 2nd/3rd/... one).
331 * Need this for arena->construct<T>(), where T is allocator-compliant thing such as STL containers,
332 * to work. */
333#ifndef NDEBUG
334 const bool ok =
335#endif
336 Base::session_shm()->add_shm_pool_listener(srv->shm_arena_pool_listener());
337 assert(ok && "Maintenance bug? It should only fail if already registered.");
338 }
339
340 } // if (!err_code) (but it may have become truthy inside)
341 } // if (!sys_err_code)
342 } // if (!err_code) (but it may have become truthy inside)
343
344 // Be Aggressive in cleaning up early (early versus waiting for our dtor that is).
345 if (err_code)
346 {
347 Base::reset_shm();
348 // For cleanliness/does not matter. (m_app_shm *pointee* can and must continue to exist for other sessions.)
349 m_app_shm.reset();
350 }
351
352 on_done_func(err_code);
353 }); // Base::Base::async_accept_log_in()
354} // Server_session_impl::async_accept_log_in()

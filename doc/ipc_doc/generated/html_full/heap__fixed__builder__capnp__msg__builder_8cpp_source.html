42void Heap_fixed_builder_capnp_message_builder::emit_segment_blobs(Segment_ptrs* target_blob_ptrs_ptr)
43{
44 // We'll need to adjust it a bit below.
45 auto& blobs = m_serialization_segments_plus_frame_space;
46 auto& target_blob_ptrs = *target_blob_ptrs_ptr;
47
48 assert((!blobs.empty()) && "Should not be possible for serialization to be empty with our use cases. Investigate.");
49 target_blob_ptrs.reserve(target_blob_ptrs.size() + blobs.size()); // Remember: we are appending; not replacing.
50
51 const auto capnp_segs = getSegmentsForOutput();
52 assert((capnp_segs.size() == n_segments())
53 && "Somehow our MessageBuilder created fewer or more segments than allocateSegment() was called?!");
54
55 for (size_t idx = 0; idx != capnp_segs.size(); ++idx)
56 {
57 // Pointer-to-word + size in words => pointer-to-byte + size in bytes.
58 const auto capnp_seg = capnp_segs[idx].asBytes();
59 const auto seg_sz = capnp_seg.size();
60
61 auto& blob = *(blobs[idx]);
62
63 assert((capnp_seg.begin() == blob.begin())
64 && "Somehow capnp-returned segments are out of order to allocateSegment() calls; or something....");
65 assert((seg_sz != 0)
66 && "capnp shouldn't be generating zero-sized segments.");
67 assert((seg_sz <= (blob.capacity() - m_frame_prefix_sz - m_frame_postfix_sz))
68 && "capnp somehow overflowed the area we gave it.");
69
70 blob.resize(seg_sz); // Just pull up end() to immediately follow the serialization.
71
72 /* @todo Maybe output some stuff, like the segment dump, here, like
73 * shm::capnp_message_builder::lend() does? Though, in practice, this is always the
74 * top serialization and hence is printed out in pretty good detail as the actual blob is transmitted over
75 * ipc::transport core layer. So maybe skip it to avoid unneeded verbosity. */
76
77 /* Lastly output a *pointer*, as promised, to the Blob we hold. They are free to do what they want
78 * with the pointed-to `blob`, including sliding around its begin() and end() to make use of the
79 * frame space. They must not access the dereferenced ptr past *this lifetime, nor past the
80 * next mutation (if any) (whichever happens first). emit_segment_blobs() can be called agains after the
81 * latter for an update serialization. */
82 target_blob_ptrs.push_back(&blob);
83 } // for (idx in [0, size()))
84} // Heap_fixed_builder_capnp_message_builder::emit_segment_blobs()

86kj::ArrayPtr<capnp::word> Heap_fixed_builder_capnp_message_builder::allocateSegment(unsigned int min_sz) // Virtual.
87{
88 using Word = capnp::word;
89 using Capnp_word_buf = kj::ArrayPtr<Word>;
90 using flow::util::Blob;
91 using std::memset;
92 constexpr size_t WORD_SZ = sizeof(Word);
93
94 /* It'd be nice not to realloc m_serialization_segments_plus_frame_space's internal buffer, as that involves
95 * moving `unique_ptr<Blob>`s around: constant-time/cheap though it is, if we can avoid it, good.
96 * This value is pretty decent; 1 is really most typical, while 4 shallow `unique_ptr<Blob>`s is not a ton of RAM. */
97 constexpr size_t N_SEGS_GUESS = 4;
98 m_serialization_segments_plus_frame_space.reserve(N_SEGS_GUESS); // No-op after 1st time.
99
100 /* Background from capnp: They're saying the need the allocated space for serialization to store at least min_sz:
101 * probably they're going to store some object that needs at least this much space. So typically it's some
102 * scalar leaf thing, like 4 bytes or whatever; but it could be larger -- or even huge (e.g., a Data or List
103 * of huge size, because the user mutated it so via a ::Builder). Oh, and it has to be zeroed, as by calloc().
104 *
105 * So all we *have* to allocate is min_sz exactly in that sense. But the idea is to try to allocate more, so that
106 * capnp can efficiently shove more objects in there too without calling allocateSegment() for each one.
107 * And a fixed-heap builder we promised to make that the fixed size m_segment_sz.
108 * (MallocMessageBuilder has that mode too: FIXED_SIZE allocation-strategy.) Of course, if min_sz exceeds that,
109 * then we have no choice but to allocate the larger amount min_sz. */
110
111 const size_t seg_sz
112 = std::max(min_sz * WORD_SZ, // Don't forget: in their API min_sz is in `word`s.
113 /* Seems prudent to give capnp an area that is a multiple of `word`s. Maybe required. Probably even.
114 * However we must *not* exceed m_segment_sz, if we can help it. A little less is okay. */
115 (m_segment_sz / WORD_SZ) * WORD_SZ);
116
117 // Now just follow the logic we expounded on in the various doc headers. Keeping explanations light here.
118
119 // By contract we can't let the prefix frame change the start of the area given to capnp to be non-aligned.
120 assert(((m_frame_prefix_sz % sizeof(void*)) == 0)
121 && "N=frame_prefix_sz passed to ctor, by contract, must be such that (aligned_ptr + N) is also aligned.");
122
123 m_serialization_segments_plus_frame_space.emplace_back(new Blob(m_logger_for_blobs));
124 auto& blob = *(m_serialization_segments_plus_frame_space.back());
125 blob.reserve(m_frame_prefix_sz + seg_sz + m_frame_postfix_sz);
126 blob.resize(seg_sz, m_frame_prefix_sz); // Set: size(), start(), respectively.
127 assert(blob.size() != 0);
128 // capnp requires: it must be zeroed. (Blob intentionally doesn't zero. @todo Would be a nice *optional* feature.)
129 memset(blob.begin(), 0, blob.size());
130
131 return Capnp_word_buf(reinterpret_cast<Word*>(blob.begin()),
132 reinterpret_cast<Word*>(blob.end()));
133} // Heap_fixed_builder_capnp_message_builder::allocateSegment()

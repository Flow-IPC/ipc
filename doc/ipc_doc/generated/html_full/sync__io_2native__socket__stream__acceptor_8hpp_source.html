34/**
35 * `sync_io`-pattern counterpart to async-I/O-pattern transport::Native_socket_stream_acceptor.
36 *
37 * @see util::sync_io doc header -- describes the general `sync_io` pattern we are following.
38 * @see transport::Native_socket_stream_acceptor -- generally all notes, other than the mechanism of how
39 * async_accept() reports readiness, in that doc header apply here.
40 *
41 * As is generally the case when choosing `sync_io::X` versus `X`, we recommend using `X` due to it being easier.
42 * In this particular case (see below) there is no perf benefit to using `sync_io::X`, either, so the only reason
43 * to use `sync_io::X` in this case would be because you've got an old-school reactor event loop with
44 * a `poll()` or `epoll_wait()`, in which case the `sync_io` API may be easier to integrate.
45 *
46 * ### Internal implementation ###
47 * Normally this would not be in the public docs for this public-use class, but indulge us.
48 *
49 * In perf-critical situations, such as the various transport::Blob_sender / transport::Blob_receiver / etc.
50 * impls, typically `sync_io::X` contains the core implementation, then `X` adapts a `sync_io::X` *core*
51 * to provide background-thread-driven work and signaling of completion. We do not consider
52 * the present `X = Native_socket_stream_acceptor` to be perf-critical; as such as of this writing
53 * transport::Native_socket_stream_acceptor has the logic, while sync_io::Native_socket_stream_acceptor
54 * adapts it into the `sync_io` pattern. Internally this is accomplished using an unnamed IPC-pipe, where
55 * an internal background thread W tickles said IPC-pipe which is waited-on by the user of
56 * sync_io::Native_socket_stream_acceptor.
57 */

113 /**
114 * Sets up the `sync_io`-pattern interaction between `*this` and the user's event loop; required before
115 * async_accept() will work (as opposed to no-op/return `false`).
116 *
117 * `ev_wait_func()` -- with signature matching util::sync_io::Event_wait_func -- is a key function memorized
118 * by `*this`. It shall be invoked by `*this` operations when some op cannot complete synchronously and requires
119 * a certain event (readable/writable) to be active on a certain native-handle.
120 *
121 * @see util::sync_io::Event_wait_func doc header for useful and complete instructions on how to write an
122 * `ev_wait_func()` properly. Doing so correctly is the crux of using the `sync_io` pattern.
123 *
124 * This is a standard `sync_io`-pattern API per util::sync_io doc header.
125 *
126 * @tparam Event_wait_func_t
127 * Function type matching util::sync_io::Event_wait_func.
128 * @param ev_wait_func
129 * See above.
130 * @return `false` if this has already been invoked; no-op logging aside. `true` otherwise.
131 */
132 template<typename Event_wait_func_t>
133 bool start_accept_ops(Event_wait_func_t&& ev_wait_func);

147 /**
148 * Starts background attempt at a peer connection being established, contingent on the user reporting
149 * (via `sync_io` pattern) the necessary native handle event(s); completion handler `on_done_func()` shall
150 * execute synchronously during such a future report. Or perhaps in plainer English:
151 * -# Call this.
152 * -# It will synchronously invoke util::sync_io::Event_wait_func from start_accept_ops(),
153 * causing you to synchronously begin an async-wait for some status of some native handle (FD).
154 * -# Once that async-wait completes, call `(*on_active_ev_func)()` to report that fact to `*this`.
155 * -# It will synchronously invoke `on_done_func()` saved from the present call,
156 * indicating via the `Error_code` arg success or failure. The semantics of this are equivalent
157 * to those of #Async_io_obj `async_accept()`.
158 *
159 * @note There is no mode of async_accept() wherein `*target_peer` is synchronously (immediately) available.
160 * Completion shall always be reported via `on_done_func()` after a mandatory async-wait.
161 * This is in contrast to perf-critical operations such as, say, Native_socket_stream::async_receive_blob().
162 *
163 * @tparam Task_err
164 * See #Async_io_obj API.
165 * @param target_peer
166 * See #Async_io_obj API.
167 * @param on_done_func
168 * See #Async_io_obj API.
169 * @return `true` if op started; `false` if one is already in progress, and therefore this one is ignored (no-op).
170 */
171 template<typename Task_err>
172 bool async_accept(Peer* target_peer, Task_err&& on_done_func);

183 * The `Task_engine` for `m_ready_*`. It is necessary to construct those pipe-end objects, but we never
184 * use that guy's `->async_*()` APIs -- only non-blocking operations, essentially leveraging boost.asio's

186 * never load any tasks onto #m_nb_task_engine and certainly never `.run()` (or `.poll()` or ...) it.
187 *
188 * In the `sync_io` pattern the user's outside event loop is responsible for awaiting readability/writability
189 * of a guy like #m_ready_reader via our exporting of its `.native_handle()`.

202 * is read out of #m_ready_reader, making it empty again (the steady-state before the next time acceptability-wait
203 * begins, and a byte is written to it making it non-empty).
204 *
205 * @see #m_ready_writer.

207 util::Pipe_reader m_ready_reader;

214
215 /**
216 * Descriptor waitable by outside event loop async-waits -- storing the same `Native_handle` as (and thus being
217 * used to wait on events from) #m_ready_reader.

222 * Function (set forever in start_accept_ops()) through which we invoke the outside event loop's

224 * doc header for a refresher on this mechanic.
225 */
226 util::sync_io::Event_wait_func m_ev_wait_func;

228 /**
229 * `on_done_func` from async_accept() if one is pending; otherwise `.empty()`. We maybe could have kept it as
230 * a lambda capture only, but this way it can double as a guard against allowing 2+ async-accepts (which goes
231 * against the spirit of what `sync_io`-pattern objects provide).
232 */
233 On_peer_accepted_func m_on_done_func_or_empty;
234
235 /// Result of last `m_async_io.async_accept()`.
236 Error_code m_target_err_code;
237
238 /// This guy does all the work. We merely report async_accept() completion via #m_ready_reader and #m_ready_writer.
239 Async_io_obj m_async_io;

246template<typename Task_err>
247bool Native_socket_stream_acceptor::async_accept(Peer* target_peer, Task_err&& on_done_func)
248{
249 using util::Task;
250
251 if (!m_on_done_func_or_empty.empty())
252 {
253 FLOW_LOG_WARNING("Acceptor [" << *this << "]: Async-accept requested, but one is already in progress. Ignoring.");
254 return false;
255 }
256 // else
257
258 m_on_done_func_or_empty = std::move(on_done_func);
259
260 m_async_io.async_accept(target_peer, [this](const Error_code& err_code)
261 {
262 if (err_code == error::Code::S_OBJECT_SHUTDOWN_ABORTED_COMPLETION_HANDLER)
263 {
264 return; // Stuff is shutting down. GTFO.
265 }
266 // else
267
268 FLOW_LOG_TRACE("Acceptor [" << *this << "]: Async-IO core async-accept done; tickling IPC-pipe to inform user.");
269 m_target_err_code = err_code;
270
271 util::pipe_produce(get_logger(), &m_ready_writer);
272 }); // m_async_io.async_accept()
273
274 m_ev_wait_func(&m_ev_wait_hndl,
275 false, // Wait for read.
276 boost::make_shared<Task>
277 ([this]()
278 {
279 FLOW_LOG_INFO("Acceptor [" << *this << "]: Async-IO core async-accept finished: informed via IPC-pipe; "
280 "invoking handler.");
281 util::pipe_consume(get_logger(), &m_ready_reader);
282
283 auto on_done_func = std::move(m_on_done_func_or_empty);
284 m_on_done_func_or_empty.clear(); // In case move() didn't do it.
285
286 on_done_func(m_target_err_code);
287 FLOW_LOG_TRACE("Handler completed.");
288 }));
289
290 return true;
291} // Native_socket_stream_acceptor::async_accept()

293template<typename Event_wait_func_t>
294bool Native_socket_stream_acceptor::start_accept_ops(Event_wait_func_t&& ev_wait_func)
295{
296 if (!m_ev_wait_func.empty())
297 {
298 FLOW_LOG_WARNING("Acceptor [" << *this << "]: Start-ops requested, "
299 "but we are already started. Probably a user bug, but it is not for us to judge.");
300 return false;
301 }
302 // else
303
304 m_ev_wait_func = std::move(ev_wait_func);
305 FLOW_LOG_INFO("Acceptor [" << *this << "]: Start-ops requested. Done.");
306 return true;
307}

309template<typename Create_ev_wait_hndl_func>
310bool Native_socket_stream_acceptor::replace_event_wait_handles(const Create_ev_wait_hndl_func& create_ev_wait_hndl_func)
311{
312 if (!m_ev_wait_func.empty())
313 {
314 FLOW_LOG_WARNING("Acceptor [" << *this << "]: Cannot replace event-wait handles after "
315 "a start-*-ops procedure has been executed. Ignoring.");
316 return false;
317 }
318 // else
319
320 FLOW_LOG_INFO("Acceptor [" << *this << "]: Replacing event-wait handles (probably to replace underlying "
321 "execution context without outside event loop's boost.asio Task_engine or similar).");
322
323 assert(m_ev_wait_hndl.is_open());
324
325 Native_handle saved(m_ev_wait_hndl.release());
326 m_ev_wait_hndl = create_ev_wait_hndl_func();
327 m_ev_wait_hndl.assign(saved);
328
329 return true;
330} // Native_socket_stream::Impl::replace_event_wait_handles()

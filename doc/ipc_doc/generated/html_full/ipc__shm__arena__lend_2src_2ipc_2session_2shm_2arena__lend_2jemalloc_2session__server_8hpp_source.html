40/**
41 * This is to vanilla Session_server what shm::arena_lend::jemalloc::Server_session is to vanilla #Server_session:
42 * it is the session-server type that starts SHM-enabled sessions with SHM-jemalloc provider
43 * (ipc::shm::arena_lend::jemalloc). Its API is identical to that of Session_server, except that it emits
44 * #Server_session_obj that are shm::arena_lend::jemalloc::Server_session and not vanilla #Server_session.
45 *
46 * @internal
47 * ### Implementation ###
48 * See similar section of session::Session_server. It explains why we sub-class Session_server_impl and even how
49 * how that's used for this SHM-jemalloc scenario. To reiterate:
50 *
51 * We use 2 of 2 available customization points of `private` super-class Session_server_impl. We:
52 * - pass-up a `per_app_setup_func()` that, given the new session's desired Client_app, creates-if-needed the per-app
53 * SHM-arena and keeps it open as well as available via `this->app_shm(Client_app::m_name)`; and
54 * - parameterize Session_server_impl on shm::arena_lend::jemalloc::Server_session which, during log-in, creates
55 * the per-session SHM-arena and keeps it open; saves `this->app_shm(Client_app::m_name)`;
56 * and partners with Client_session_impl to set up the lend/borrowing capability between us.
57 *
58 * shm::arena_lend::jemalloc::Server_session doc header delves deeply into the entire impl strategy for setting up
59 * these arenas. If you read/grok that, then the present class's impl should be straightforward to follow.
60 * @endinternal
61 *
62 * @tparam S_MQ_TYPE_OR_NONE
63 * See vanilla #Session_server.
64 * @tparam S_TRANSMIT_NATIVE_HANDLES
65 * See vanilla #Session_server.
66 * @tparam Mdt_payload
67 * See vanilla #Session_server.
68 */

181 /**
182 * Returns pointer to the per-`app` SHM-arena, whose lifetime extends until `*this` is destroyed;
183 * or null if the given Client_app has not yet opened at least 1 shm::arena_lend::jemalloc::Server_session via
184 * async_accept(). Alternatively you may use shm::arena_lend::jemalloc::Session_mv::app_shm() off any session object
185 * filled-out by `*this` async_accept(), as long as its Server_session_mv::client_app() equals
186 * `app` (by App::m_name equality).
187 *
188 * If non-null is returned, then the same pointer value shall be returned for all subsequent calls
189 * with the same (by App::m_name equality) `app`. The non-null pointers returned for any 2 calls, where `app`
190 * is different (by App::m_name equality) among them, shall always differ.
191 *
192 * See shm::arena_lend::jemalloc::Session_mv::Arena doc header for useful instructions on working with #Arena,
193 * `lend_object()`, and `borrow_object()`.
194 *
195 * ### Perf ###
196 * Given the choice between Server_session_mv::app_shm() and the present method, the latter is somewhat
197 * slower; internally it involves a mutex-protected map lookup, while the former simply returns a cached
198 * pointer as of this writing.
199 *
200 * Generally it is also quite fast for the user to save any non-null value returned by either `app_shm()`;
201 * the pointer returned shall always be the same after all.
202 *
203 * @internal
204 * ### Thread safety ###
205 * For internal use, namely by shm::arena_lend::jemalloc::Server_session_impl::async_accept_log_in() at least,
206 * it is guaranteed the app_shm() may be called on the same `*this` concurrently to itself
207 * and init_app_shm_as_needed(). Formally speaking this isn't publicly documented, as I (ygoldfel) didn't want
208 * to get users into any bad habit, but internally it does have this property -- as it is required.
209 * @endinternal
210 *
211 * @param app
212 * Client_app whose segregated SHM-arena to return, if a session for a client of the app has been
213 * opened prior to this call.
214 * @return Pointer to `*this`-held per-`app` SHM-arena, if it has been created; null otherwise.
215 * See above.
216 */
217 Arena* app_shm(const Client_app& app);

396TEMPLATE_JEM_SESSION_SRV
397Error_code CLASS_JEM_SESSION_SRV::init_app_shm_as_needed(const Client_app& app)
398{
399 using ipc::shm::arena_lend::jemalloc::Memory_manager;
400 using boost::movelib::make_unique;
401 using std::make_shared;
402 using std::to_string;
403 using std::string;
404
405 /* We are in some unspecified thread; actually *a* Session_server_impl thread Ws (a Server_session_impl thread W).
406 * Gotta lock at least to protect from concurrent calls to ourselves on behalf of other async_accept()s. */
407
408 Lock_guard app_shm_lock(m_app_shm_mutex);
409
410 auto& app_shm = m_app_shm_by_name[app.m_name]; // Find; insert if needed.
411 if (app_shm)
412 {
413 // Cool; already exists, as app.m_name seen already (and successfully set-up, as seen below) by us.
414 return Error_code();
415 }
416 // else
417
418 // Below should INFO-log already; let's not litter logs explaining why this is being created; context = sufficient.
419
420 /* Arena needs a pool namer it will invoke each time the memory manager decides it wants a new mmap()ped vaddr
421 * space (in the SHM context, <-> SHM-pool). We use our standard semantics for per-session items. To save some
422 * compute cycles pre-compute the prefix and remember it via capture.
423 *
424 * This is the same as Session_mv::init_shm() but simpler due to being a server-side thing (as opposed to possibly
425 * client-side) and thus not needing to add any client-PID for cleanup purposes (server-PID is already
426 * the srv_namespace baked into every Shared_name). Also it's per-app, not per-session, hence cli_namespace is
427 * not needed. */
428 auto pool_naming_func = [shm_pool_name_prefix
429 = build_conventional_shared_name(Shared_name::S_RESOURCE_TYPE_ID_SHM,
430 Shared_name::ct(m_srv_app_ref.m_name), m_srv_namespace,
431 Shared_name::ct(app.m_name))
432 / SHM_SUBTYPE_PREFIX / ""] // Might as well remember the last separator too.
433 (auto pool_id) -> string
434 {
435 // @todo Arena expects string at the moment; might/should change to Shared_name at some point.
436
437 string shm_pool_name = shm_pool_name_prefix.str();
438
439 /* Per contract pool_id is unique entirely globally and is informally recommended to encode here; so we do.
440 * We could instead keep our own ++ing pool-ID, since we guarantee non-conflicting names from other
441 * apps/servers/etc. using our Shared_name-building conventions; but since they're giving us this
442 * thing for free (as of this writing it is globally-unique -- even across apps, etc. -- for
443 * SHM-jemalloc's own internal purposes), we might as well use it. */
444 return shm_pool_name += to_string(pool_id);
445 // @todo to_string() deals with locales which may be slow. Consider std::to_chars().
446 }; // auto pool_naming_func =
447
448 app_shm
449 = Arena::create(get_logger(),
450 // See equally-relevant note/@todo in Session_mv::init_shm().
451 make_shared<Memory_manager>(get_logger()),
452 typename Arena::Shm_object_name_generator(std::move(pool_naming_func)),
453 util::shared_resource_permissions(m_srv_app_ref.m_permissions_level_for_client_apps));
454 if (!app_shm)
455 {
456 /* app_shm, which is in the map directly, is null. Just leave null in the map; meh.
457 * .erase()ing it from there is just pedantic at best. (The [] lookup above will do the right thing next time.) */
458
459 FLOW_LOG_WARNING("Session_server [" << * this << "]: Failed to create session-scope Arena; "
460 "details may be found above. Session will not open.");
461
462 // See @todo on this Code; in short if create() emitted an Error_code, we'd just emit that instead here.
463 return error::Code::S_SHM_ARENA_CREATION_FAILED;
464 }
465 // else: Cool! One last thing (which cannot really fail).
466
467 /* Need this for arena->construct<T>(), where T is allocator-compliant thing such as STL containers,
468 * to work. */
469#ifndef NDEBUG
470 const bool ok =
471#endif
472 app_shm->add_shm_pool_listener(&m_shm_arena_pool_listener);
473 assert(ok && "Maintenance bug? It should only fail if already registered.");
474
475 return Error_code();
476 // Lock_guard app_shm_lock(m_app_shm_mutex): unlocks here.
477} // Session_server::init_app_shm_as_needed()

485TEMPLATE_JEM_SESSION_SRV
486std::shared_ptr<typename CLASS_JEM_SESSION_SRV::Arena> CLASS_JEM_SESSION_SRV::app_shm_ptr(const Client_app& app)
487{
488 using std::shared_ptr;
489
490 // We are in some unspecified thread; we promised thread safety form any concurrency situation.
491
492 Lock_guard app_shm_lock(m_app_shm_mutex);
493
494 /* Subtlety: Due to an intentional quirk of init_app_shm_as_needed(), if it's in the map, the ptr may still be null:
495 * init_app_shm_as_needed() failed for app.m_name but does not erase in that case and just leaves null. */
496 const auto map_it = m_app_shm_by_name.find(app.m_name);
497 return (map_it == m_app_shm_by_name.end()) ? shared_ptr<Arena>() : map_it->second;
498
499 // Lock_guard app_shm_lock(m_app_shm_mutex): unlocks here.
500}

512TEMPLATE_JEM_SESSION_SRV
513typename CLASS_JEM_SESSION_SRV::Structured_msg_reader_config
514 CLASS_JEM_SESSION_SRV::app_shm_reader_config(const Client_app& app)

560TEMPLATE_JEM_SESSION_SRV
561void CLASS_JEM_SESSION_SRV::cleanup()
562{
563 // `classic`? Pool_arena?! But this is jemalloc!?!?! Worry not. See remove_each_persistent_if() doc header.
564 using ipc::shm::classic::Pool_arena;
565 using util::String_view;
566 using util::process_id_t;
567 using boost::chrono::seconds;
568 using boost::lexical_cast;
569
570 constexpr util::Fine_duration CLEANUP_PERIOD = seconds(30);
571
572 FLOW_LOG_TRACE("Client session [" << *this << "]: Periodic (or initial) cleanup starting.");
573
574 /* This is just like Client_session_impl::cleanup(), except for some small differences explained therein; but
575 * the main explanation is right here. Read on.
576 *
577 * As with SHM-classic, cleanup of SHM-pools = 2 tasks; graceful cleanup, when objects are destroyed normally
578 * on process exit() or earlier; and cleanup after a process dies or zombifies/later dies and thus the
579 * regular destructor don't run. Graceful cleanup for SHM-jemalloc is accomplished in each Arena's dtor
580 * and is, to us, a black box; the point is it takes care of it, so we need not do anything further.
581 * (This is in contrast to SHM-classic which centers on classic::Pool_arena which is so minimalistic that it leaves
582 * that task to the user of Pool_arena; in our case ipc::session::shm::classic code is that user and takes care of it.
583 * Point is, with SHM-jemalloc we need not.) So that leaves ungraceful cleanup. In point of fact, SHM-jemalloc
584 * explicitly leaves the task of handling to that to its user (that's us). So:
585 *
586 * In short, SHM-jemalloc at its core is symmetric in the sense that in an A-B session, where (say) A happens
587 * to be the Session_server side (SHM-jemalloc is not aware of that, but you and I are here), A maintains
588 * SHM-arena(s); and B maintains its own SHM-arena(s) (and the two cross-borrow objects from each other).
589 * If A aborts before destructor(s) completely run, then its arena(s) shall leak, and it's up to us to
590 * remove_persistent() (really, SHM-unlink) them; and just the same for B. While it is conceivable to try to
591 * impose the ipc::session ethos, where session-server is responsible for cleaning up everything -- including
592 * stuff leaked by session-client(s) -- in this (rare) case that's not really natural. It is difficult to make
593 * any assumptions about what state some creator-process of a given SHM-pool may be in: SHM-jemalloc is a (complex)
594 * black box. Instead it is better to treat each side as similarly as possible, mirroring SHM-jemalloc's own
595 * design (even though it left this task to us).
596 *
597 * Exact rationale aside, here's the deal. If a given creator (owner) process of a SHM-jemalloc-managed pool
598 * is not running, then it is correct and safe to remove_persistent() that pool: any existing handles to that
599 * pool in other processes that may be alive will continue to be valid; and no more such processes shall arise.
600 * Why? Answer: because they only *can* arise through the Shm_session::lend_arena(Arena*) operation within the
601 * creator (owner) of that Arena (which = pool collection = all SHM pools in arena)... but that process is dead.
602 * Nor does SHM-jemalloc support an owner re-opening (in the owner capacity) a previously created SHM-pool.
603 * Therefore, logically, *if* we can determine that a given pool's owner is dead, *then* we can/should SHM-unlink
604 * that pool. We'll discuss how to do that next. First, though, is it ever safe/correct to SHM-unlink a pool,
605 * while its creator (owner) is *not* dead? The answer is no, except if that process is zombified (not "really"
606 * active). We have no good way of determining that, as of this writing anyway. There are some exceptions to
607 * this within ipc::session, and we are indeed in ipc::session right now. Namely:
608 * - If the owner of a pool was a Session_server (or Server_session borne of one), then in our paradigm there
609 * must be at most one active Session_server; so if we are alive, then the owner of the pool must be inactive,
610 * assuming things are working properly. (SHM-classic cleanup relies on that in its cleanup algorithm.)
611 * - However no such inference can be on Client_session side: Multiple `Client_session`s of one Client_app
612 * may live concurrently.
613 * Since we can't really rely on that stuff, we elect to instead go with the simple aforementioned rule:
614 * - A SHM-pool was created by this app's SHM-jemalloc; and its owner (creator) process is now dead
615 * <=> remove_persistent() that SHM-pool.
616 *
617 * That leaves only the following challenges:
618 * - List all SHM-pools created by this app's SHM-jemalloc. Solution: Pool_arena::for_each_persistent(), etc.,
619 * which can scan /dev/shm (in Linux anyway) and find the pools with a certain name pattern; our Shared_name
620 * ipc::session semantics (formally implemented in session_shared_name.hpp and described in Shared_name doc
621 * header) name things in such a way as to make this pretty simple.
622 * - Determine PID of creator of given pool. Solution: Our PID is encoded into the Shared_name due to the
623 * aforementioned ipc::session Shared_name semantics.
624 * - Determne whether process with a certain PID is alive. Solution: util::process_running().
625 *
626 * Let's do it. */
627
628 const auto n_removed
629 = util::remove_each_persistent_if<Pool_arena>(get_logger(),
630 [&](const Shared_name& shm_pool_name)
631 {
632 Shared_name resource_type;
633 Shared_name srv_app_name;
634 Shared_name srv_namespace_aka_pid;
635 Shared_name the_rest;
636
637 if (!(decompose_conventional_shared_name(shm_pool_name,
638 // We always identify resource type even though redundant in this case.
639 &resource_type,
640 &srv_app_name, // This must be us, not some other app.
641 &srv_namespace_aka_pid, // This is what must be dead.
642 nullptr, // cli_app_name: Needs to be there, but we don't care what it is.
643 nullptr, // cli_namespace_or_sentinel: Ditto.
644 &the_rest)
645 && (resource_type == Shared_name::S_RESOURCE_TYPE_ID_SHM)
646 && (srv_app_name == m_srv_app_ref.m_name)
647 && String_view(the_rest.str()).starts_with((SHM_SUBTYPE_PREFIX + Shared_name::S_SEPARATOR).str())))
648 {
649 /* Not our pool to possibly delete. Misnamed; or not from a split with our Server_app;
650 * or not from this SHM-provider. */
651 return false;
652 }
653 // else
654
655 /* We could actually skip this next check -- then we'd be deleting client-created stuff too. It doesn't seem
656 * like there's any practical downside. Just... I (ygoldfel) have really convinced myself by that text
657 * in the big comment above regarding arena-lending SHM-providers being symmetrical in how each side deals
658 * with the resources they owned. So going out of our way to clean server stuff by server, client stuff by
659 * by client. @todo Reconsider maybe. It's also not of huge import, this question: If it gets cleaned, cool. */
660 if (String_view(the_rest.str()).find_first_of(Shared_name::S_SEPARATOR,
661 SHM_SUBTYPE_PREFIX.size() + 1) != String_view::npos)
662 {
663 return false; // Not our pool to possibly delete. From a split with our Server_app but created by client side.
664 }
665 // else
666
667 process_id_t pid = 0;
668 try
669 {
670 pid = lexical_cast<process_id_t>(srv_namespace_aka_pid.str());
671 }
672 catch (...) {}
673 if (pid == 0)
674 {
675 FLOW_LOG_WARNING("Session server [" << *this << "]: During periodic cleanup found seemingly relevant "
676 "(based on prefix/form) SHM-pool named [" << shm_pool_name << "]; but server-PID (a/k/a "
677 "server-namespace) term [" << srv_namespace_aka_pid << "] is not a valid-looking PID. "
678 "Ignoring; but this warrants investigation.");
679 return false;
680 }
681 // else
682
683 if (util::process_running(pid))
684 {
685 return false;
686 }
687 // else
688
689 FLOW_LOG_INFO("Session server [" << *this << "]: During periodic cleanup found SHM-pool "
690 "named [" << shm_pool_name << "]; appears to have been created by a process of this application "
691 "that is no longer running, even as a zombie; perhaps it exited without graceful cleanup. Will "
692 "attempt to remove it. Competing cleanup(s) may be occurring concurrently; that is fine.");
693 return true;
694 }); // remove_each_persistent_if()
695 if (n_removed == 0)
696 {
697 FLOW_LOG_TRACE("Session server [" << *this << "]: Cleanup finished: remove none.");
698 }
699 else
700 {
701 FLOW_LOG_INFO("Session server [" << *this << "]: Cleanup finished: Removed [" << n_removed << "] pools "
702 "successfully.");
703 }
704
705 m_async_cleanup_worker.schedule_from_now(CLEANUP_PERIOD, [this](auto&&) { cleanup(); });
706} // Session_server::cleanup()
